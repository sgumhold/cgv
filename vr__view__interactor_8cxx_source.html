342 bool vr_view_interactor::handle(cgv::gui::event& e)
343 {
344  if ((e.get_flags() & cgv::gui::EF_VR) != 0) {
345  if (debug_vr_events) {
346  e.stream_out(std::cout);
347  std::cout << std::endl;
348  }
349  return handle_vr_events(e);
350  }
351 
352  if (head_tracker != -1) {
353  if ( ((e.get_flags() & cgv::gui::EF_VR) != 0) && (e.get_kind() == cgv::gui::EID_POSE) ) {
354  cgv::gui::vr_pose_event& vrpe = dynamic_cast<cgv::gui::vr_pose_event&>(e);
355  if (vrpe.get_trackable_index() == head_tracker) {
356  const mat3& O = reinterpret_cast<const mat3&>(vrpe.get_orientation());
357  const vec3& p = reinterpret_cast<const vec3&>(vrpe.get_position());
358  mat3 V = head_tracker_orientation * O;
359  set_view_up_dir(V.col(1));
360  set_view_dir(-V.col(2));
361 
362  vec3 dv = get_focus() - get_eye();
363  vec3 e = head_tracker_orientation * p + head_tracker_position;
364  set_focus(e+dv);
365  }
366  }
367  }
368  if (e.get_kind() == cgv::gui::EID_KEY) {
369  cgv::gui::key_event& ke = static_cast<cgv::gui::key_event&>(e);
370  if ((ke.get_action() == cgv::gui::KA_PRESS)) {
371  if (ke.get_modifiers() == cgv::gui::EM_CTRL) {
372  if (ke.get_key() >= '0' && ke.get_key() < '4') {
373  unsigned player_index = ke.get_key() - '0';
374  if (player_index < kits.size()) {
375  current_vr_handle_index = player_index;
376  current_vr_handle = kits[player_index];
377  update_member(&current_vr_handle_index);
378  return true;
379  }
380  }
381  if (ke.get_key() == cgv::gui::KEY_Space) {
382  separate_view = !separate_view;
383  on_set(&separate_view);
384  return true;
385  }
386  }
387  else if (ke.get_modifiers() == cgv::gui::EM_CTRL + cgv::gui::EM_SHIFT) {
388  if (ke.get_key() >= '0' && ke.get_key() < '1'+vr::max_nr_controllers) {
389  int ci = ke.get_key() - '0';
390  if (ci == vr::max_nr_controllers)
391  ci = 0;
392  if (current_vr_handle_index >= 0) {
393  vr::vr_kit_state& state = kit_states[current_vr_handle_index];
394  if (state.controller[ci].status == vr::VRS_TRACKED) {
395  // p = R * (q-r) + t
396  // f = R * (q-r') + t'
397  // p - f = R*(r'-r)+t-t'
398  // t' = f
399  // p - f = R*(r'-r)+t-f
400  // r' = r = R^*(p - t)
401  vec3& p = reinterpret_cast<vec3&>(state.controller[ci].pose[9]);
402  mat3 invR = cgv::math::rotate3<float>(-tracking_rotation, vec3(0, 1, 0));
403  tracking_rotation_origin += invR* (p - tracking_origin);
404  tracking_origin = get_focus();
405  p = get_focus();
406  for (int i = 0; i < 3; ++i) {
407  update_member(&tracking_origin[i]);
408  update_member(&tracking_rotation_origin[i]);
409  }
410  calibrate_driver();
411  }
412  }
413  }
414  }
415  }
416  }
417  return stereo_view_interactor::handle(e);

499  }

501  }

503  if (iter != kits.end()) {

506  }
507  old_kits.pop_back();
508  }
509  if (current_vr_handle == 0 && !kits.empty())
510  current_vr_handle = kits.front();

512  while (!new_kits.empty()) {

514  if (kit_ptr) {
515  if (!kit_ptr->fbos_initialized())

518  if (current_vr_handle == 0)
519  current_vr_handle = new_kits.back();
520  }
521  kits.push_back(new_kits.back());
522  update_kits = true;
523  }
524  new_kits.pop_back();
525  }

527  //

541  kit_enum_definition += kit_name;
542  }

549  if (current_vr_handle == 0)
550  current_vr_handle_index = -1;
551  else {
552  for (unsigned i = 0; i < kits.size(); ++i)
553  if (kits[i] == current_vr_handle) {
554  current_vr_handle_index = i;

590 // render the views for the kits in nested render passes
591 void vr_view_interactor::render_vr_kits(cgv::render::context& ctx)
592 {
593  cgv::render::RenderPassFlags rpf = ctx.get_render_pass_flags();
594  // first render all but current vr kit views
595  for (rendered_display_index = 0; rendered_display_index<int(kits.size()); ++rendered_display_index) {
596  if (rendered_display_index == current_vr_handle_index)
597  continue;
598  rendered_display_ptr = get_vr_kit_from_index(rendered_display_index);
599  if (!rendered_display_ptr)
600  continue;
601  if (kit_states[rendered_display_index].hmd.status == vr::VRS_DETACHED)
602  continue;
603  void* fbo_handle;
604  ivec4 cgv_viewport;
605  for (rendered_eye = 0; rendered_eye < 2; ++rendered_eye) {
606  rendered_display_ptr->enable_fbo(rendered_eye);
607  ctx.announce_external_frame_buffer_change(fbo_handle);
608  ctx.announce_external_viewport_change(cgv_viewport);
609  ctx.render_pass(cgv::render::RP_USER_DEFINED, cgv::render::RenderPassFlags(rpf & ~cgv::render::RPF_HANDLE_SCREEN_SHOT), this);
610  rendered_display_ptr->disable_fbo(rendered_eye);
611  ctx.recover_from_external_viewport_change(cgv_viewport);
612  ctx.recover_from_external_frame_buffer_change(fbo_handle);
613  }
614  }
615  // then render current vr kit
616  rendered_display_index = current_vr_handle_index;
617  rendered_display_ptr = get_vr_kit_from_index(rendered_display_index);
618  if (rendered_display_ptr && kit_states[rendered_display_index].hmd.status != vr::VRS_DETACHED) {
619  void* fbo_handle;
620  ivec4 cgv_viewport;
621  //rendered_eye: 0...monitor,1...left eye,2...right eye
622  for (rendered_eye = 0; rendered_eye < 2; ++rendered_eye) {
623  rendered_display_ptr->enable_fbo(rendered_eye);
624  ctx.announce_external_frame_buffer_change(fbo_handle);
625  ctx.announce_external_viewport_change(cgv_viewport);
626  // we break here in the case of no separate to use main render pass to render second eye of current vr kit
627  if (rendered_eye == 1 && !separate_view) {
628  this->fbo_handle = fbo_handle;
629  this->cgv_viewport = cgv_viewport;
630  break;
631  }
632  ctx.render_pass(cgv::render::RP_USER_DEFINED, cgv::render::RenderPassFlags(rpf & ~cgv::render::RPF_HANDLE_SCREEN_SHOT));
633  rendered_display_ptr->disable_fbo(rendered_eye);
634  ctx.recover_from_external_viewport_change(cgv_viewport);
635  ctx.recover_from_external_frame_buffer_change(fbo_handle);
636  }
637  }
638  else {
639  rendered_display_ptr = 0;
640  rendered_display_index = -1;
641  }
642  if (separate_view) {
643  rendered_display_ptr = 0;
644  rendered_display_index = -1;
645  }

697 void vr_view_interactor::draw_vr_kits(cgv::render::context& ctx)
698 {
699  std::vector<vec4> spheres;
700  std::vector<rgb> sphere_colors;
701  cgv::render::mesh_render_info* MI_hmd_ptr = 0;
702  cgv::render::mesh_render_info* MI_controller_ptr = 0;
703  cgv::render::mesh_render_info* MI_tracker_ptr = 0;
704  cgv::render::mesh_render_info* MI_base_ptr = 0;
705  std::set<const vr::vr_driver*> driver_set;
706  for (int i = 0; i < (int)kits.size(); ++i) {
707  // skip removed kits
708  vr::vr_kit* kit_ptr = get_vr_kit_from_index(i);
709  if (!kit_ptr)
710  continue;
711  // extract drivers of which the bases should be drawn
712  if (base_vis_type != VVT_NONE)
713  driver_set.insert(kit_ptr->get_driver());
714  // ignore completely hidden vr kits
715  if (hmd_vis_type == VVT_NONE && controller_vis_type == VVT_NONE && tracker_vis_type == VVT_NONE)
716  continue;
717  // determine state of vr kit and skip kit if no state is available
718  vr::vr_kit_state state;
719  vr::vr_kit_state* state_ptr = &state;
720  if (i == current_vr_handle_index)
721  state_ptr = &kit_states[current_vr_handle_index];
722  else
723  if (!kit_ptr->query_state(state, 1))
724  continue;
725  // render other hmds
726  if (kit_ptr != rendered_display_ptr) {
727  // either as mesh
728  if ((hmd_vis_type & VVT_MESH) != 0) {
729  if (!MI_hmd_ptr) {
730  MI_hmd_ptr = vr::get_vrmesh_render_info(ctx, vr::VRM_HMD);
731  // try to read meshes and turn off mesh rendering and potentially switch to sphere rendering in case meshes are not available
732  if (MI_hmd_ptr) {
733  hmd_mesh_file_name = vr::get_vrmesh_file_name(vr::VRM_HMD);
734  update_member(&hmd_mesh_file_name);
735  }
736  else {
737  hmd_vis_type = VVT_SPHERE;
738  on_set(&hmd_vis_type);
739  }
740  }
741  if (MI_hmd_ptr != 0) {
742  ctx.push_modelview_matrix();
743  ctx.mul_modelview_matrix(
744  cgv::math::pose4<float>(reinterpret_cast<const mat34&>(state_ptr->hmd.pose[0]))*
745  cgv::math::translate4<float>(0, 0.1f, -0.1f)*
746  cgv::math::scale4<float>(vec3(mesh_scales[vr::VRM_HMD]))
747  );
748  MI_hmd_ptr->draw_all(ctx);
749  ctx.pop_modelview_matrix();
750  }
751  }
752  // and or as spheres
753  if ((hmd_vis_type & VVT_SPHERE) != 0) {
754  float left_eye_to_head[12];
755  float right_eye_to_head[12];
756  kit_ptr->put_eye_to_head_matrix(0, left_eye_to_head);
757  kit_ptr->put_eye_to_head_matrix(1, right_eye_to_head);
758  const mat3& R_w_h = reinterpret_cast<const mat3&>(state_ptr->hmd.pose[0]);
759  const vec3& p_w_h = reinterpret_cast<const vec3&>(state_ptr->hmd.pose[9]);
760  const mat3& R_h_l = reinterpret_cast<const mat3&>(left_eye_to_head[0]);
761  const vec3& p_h_l = reinterpret_cast<const vec3&>(left_eye_to_head[9]);
762  const mat3& R_h_r = reinterpret_cast<const mat3&>(right_eye_to_head[0]);
763  const vec3& p_h_r = reinterpret_cast<const vec3&>(right_eye_to_head[9]);
764  vec4 s_l(0, 0, 0, 0.012f);
765  vec4 s_r = s_l;
766  reinterpret_cast<vec3&>(s_l) = R_w_h * p_h_l + p_w_h;
767  reinterpret_cast<vec3&>(s_r) = R_w_h * p_h_r + p_w_h;
768  spheres.push_back(s_l);
769  sphere_colors.push_back(rgb(1, 0, 0));
770  spheres.push_back(s_r);
771  sphere_colors.push_back(rgb(0, 0, 1));
772  }
773  }
774  for (unsigned ci = 0; ci < vr::max_nr_controllers; ++ci) {
775  if (state_ptr->controller[ci].status != vr::VRS_TRACKED || state_ptr->controller[ci].time_stamp == 0)
776  continue;
777  bool show_trackable_spheres;
778  cgv::render::mesh_render_info* M_info = 0;
779  float mesh_scale = 1;
780  if (kit_ptr->get_device_info().controller[ci].type == vr::VRC_CONTROLLER) {
781  show_trackable_spheres = (controller_vis_type & VVT_SPHERE) != 0;
782  if ((controller_vis_type & VVT_MESH) != 0) {
783  if (!MI_controller_ptr) {
784  MI_controller_ptr = vr::get_vrmesh_render_info(ctx, vr::VRM_CONTROLLER);
785  if (MI_controller_ptr) {
786  controller_mesh_file_name = vr::get_vrmesh_file_name(vr::VRM_CONTROLLER);
787  update_member(&controller_mesh_file_name);
788  }
789  else {
790  controller_vis_type = VVT_SPHERE;
791  on_set(&controller_vis_type);
792  }
793  }
794  if (MI_controller_ptr) {
795  M_info = MI_controller_ptr;
796  mesh_scale = mesh_scales[vr::VRM_CONTROLLER];
797  }
798  }
799  }
800  else {
801  show_trackable_spheres = (tracker_vis_type & VVT_SPHERE) != 0;
802  if ((tracker_vis_type & VVT_MESH) != 0) {
803  if (!MI_tracker_ptr) {
804  MI_tracker_ptr = vr::get_vrmesh_render_info(ctx, vr::VRM_TRACKER);
805  if (MI_tracker_ptr) {
806  tracker_mesh_file_name = vr::get_vrmesh_file_name(vr::VRM_TRACKER);
807  update_member(&tracker_mesh_file_name);
808  }
809  else {
810  tracker_vis_type = VVT_SPHERE;
811  on_set(&tracker_vis_type);
812  }
813  }
814  if (MI_tracker_ptr) {
815  M_info = MI_tracker_ptr;
816  mesh_scale = mesh_scales[vr::VRM_TRACKER];
817  }
818  }
819  }
820  if (show_trackable_spheres)
821  add_trackable_spheres(state_ptr->controller[ci].pose, i, spheres, sphere_colors);
822  if (M_info) {
823  ctx.push_modelview_matrix();
824  ctx.mul_modelview_matrix(cgv::math::pose4<float>(reinterpret_cast<const mat34&>(state_ptr->controller[ci].pose[0])));
825  ctx.mul_modelview_matrix(cgv::math::scale4<float>(vec3(mesh_scale)));
826  M_info->draw_all(ctx);
827  ctx.pop_modelview_matrix();
828  }
829  }
830  }
831  for (auto& dp : driver_set) {
832  auto ss = dp->get_tracking_reference_states();
833  for (const auto& s : ss) {
834  if (s.second.status == vr::VRS_TRACKED) {
835  if ((base_vis_type & VVT_SPHERE) != 0)
836  add_trackable_spheres(s.second.pose, -1, spheres, sphere_colors);
837  if ((base_vis_type & VVT_MESH) != 0) {
838  if (!MI_base_ptr) {
839  MI_base_ptr = vr::get_vrmesh_render_info(ctx, vr::VRM_BASE);
840  if (MI_base_ptr) {
841  base_mesh_file_name = vr::get_vrmesh_file_name(vr::VRM_BASE);
842  update_member(&base_mesh_file_name);
843  }
844  else {
845  base_vis_type = VVT_SPHERE;
846  on_set(&base_vis_type);
847  }
848  }
849  if (MI_base_ptr) {
850  ctx.push_modelview_matrix();
851  ctx.mul_modelview_matrix(cgv::math::pose4<float>(reinterpret_cast<const mat34&>(s.second.pose[0])));
852  ctx.mul_modelview_matrix(cgv::math::scale4<float>(vec3(mesh_scales[vr::VRM_BASE])));
853  MI_base_ptr->draw_all(ctx);
854  ctx.pop_modelview_matrix();
855  }
856  }
857  }
858  }
859  }
860  if (!spheres.empty()) {
861  cgv::render::sphere_renderer& sr = cgv::render::ref_sphere_renderer(ctx);
862  sr.set_y_view_angle(float(get_y_view_angle()));
863  sr.set_render_style(srs);
864  sr.set_sphere_array(ctx, spheres);
865  sr.set_color_array(ctx, sphere_colors);
866  sr.render(ctx, 0, spheres.size());
867  }
868 }

871 void vr_view_interactor::draw_action_zone(cgv::render::context& ctx)
872 {
873  if (show_action_zone && current_vr_handle) {
874  double time = cgv::gui::trigger::get_current_time();
875  vr::vr_kit* kit_ptr = get_vr_kit_from_index(current_vr_handle_index);
876  if (kit_ptr) {
877  const vr::vr_driver* driver_ptr = kit_ptr->get_driver();
878  if (driver_ptr) {
879  float h = driver_ptr->get_action_zone_height();
880  vec3 up_dir;
881  driver_ptr->put_up_direction(&up_dir[0]);
882  std::vector<float> boundary;
883  driver_ptr->put_action_zone_bounary(boundary);
884  size_t n = boundary.size() / 3;
885  std::vector<vec3> G;
886  G.resize(5 * n);
887  unsigned i;
888  for (i = 0; i < 5; ++i) {
889  for (size_t j = 0; j < n; ++j) {
890  vec3 p = reinterpret_cast<const vec3&>(boundary[3 * j]);
891  p += 0.25f*i*h*up_dir;
892  G[i*n + j] = p;
893  }
894  }
895  glLineStipple(3, GLushort(0xF0F0));
896  glEnable(GL_LINE_STIPPLE);
897  glLineWidth(fence_line_width);
898  auto& prog = ctx.ref_default_shader_program();
899  int pos_idx = prog.get_position_index();
900  cgv::render::attribute_array_binding::set_global_attribute_array(ctx, pos_idx, G);
901  cgv::render::attribute_array_binding::enable_global_array(ctx, pos_idx);
902  prog.enable(ctx);
903  float lambda = float(cos(fence_frequency*time));
904  lambda *= lambda;
905  ctx.set_color((1 - lambda)*fence_color1 + lambda * fence_color2);
906  for (i = 0; i < 5; ++i)
907  glDrawArrays(GL_LINE_LOOP, GLint(i*n), (GLsizei)n);
908  prog.disable(ctx);
909  cgv::render::attribute_array_binding::disable_global_array(ctx, pos_idx);
910  glDisable(GL_LINE_STIPPLE);
911  glLineWidth(1.0f);
912  post_redraw();
913  }
914  }
915  }
