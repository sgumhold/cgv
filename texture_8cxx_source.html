234 template <class int_type>
235 bool is_power_of_two(int_type i)
236 {
237  do {
238  if (i == 1)
239  return true;
240  if ((i & 1) != 0)
241  return false;
242  i /= 2;
243  }
244  while (true);
245  return false;
246 }

248 template <class int_type>
249 int_type power_of_two_ub(int_type i)
250 {
251  int_type res = 2;
252  while (res < i)
253  res *= 2;
254  return res;
255 }

257 bool texture::create_from_image(cgv::data::data_format& df, cgv::data::data_view& dv, const context& ctx,
258  const std::string& file_name, unsigned char* clear_color_ptr, int level, int cube_side)
259 {
260  bool ensure_power_of_two = clear_color_ptr != 0;
261  std::string fn = file_name;
262  if (df.empty()) {
263  if (fn.empty()) {
264  cgv::render::render_component::last_error = "attempt to create texture from empty file name";
265  return false;
266  }
267  }
268  std::vector<data_format> palette_formats;
269  std::vector<data_view> palettes;
270  if (!fn.empty()) {
271  image_reader ir(df, &palette_formats);
272  if (!ir.read_image(fn, dv)) {
273  cgv::render::render_component::last_error = "error reading image file ";
274  cgv::render::render_component::last_error += fn;
275  cgv::render::render_component::last_error += ": ";
276  cgv::render::render_component::last_error += ir.get_last_error();
277  return false;
278  }
279  for (unsigned i=0; i<palette_formats.size(); ++i) {
280  palettes.push_back(data_view());
281  if (!ir.read_palette(i, palettes.back())) {
282  cgv::render::render_component::last_error = "error reading palette";
283  return false;
284  }
285  }
286  }
287  if (cube_side < 1)
288  destruct(ctx);
289  size_t w = df.get_width(), h = df.get_height();
290  size_t W = w, H = h;
291  data_format df1(df);
292  if (ensure_power_of_two && (!is_power_of_two(w) || !is_power_of_two(h))) {
293  W = power_of_two_ub(df.get_width());
294  H = power_of_two_ub(df.get_height());
295  df1.set_width(W);
296  df1.set_height(H);
297  }
298  //float ext[2] = { (float)w/W, (float)h/H };
299  data_view dv1(&df1);
300  unsigned entry_size = df.get_entry_size();
301  const unsigned char* src_ptr = dv.get_ptr<unsigned char>();
302  unsigned char* dest_ptr = dv1.get_ptr<unsigned char>();
303  unsigned char* dest_ptr_end = dest_ptr + entry_size * W*H;
304  for (unsigned y = 0; y < h; ++y) {
305  dest_ptr_end -= W * entry_size;
306  memcpy(dest_ptr_end, src_ptr, w*entry_size);
307  if (clear_color_ptr) {
308  for (size_t x = w; x < W; ++x)
309  memcpy(dest_ptr_end + x * entry_size, clear_color_ptr, entry_size);
310  }
311  else
312  std::fill(dest_ptr_end + w * entry_size, dest_ptr_end + W*entry_size, 0);
313  src_ptr += w * entry_size;
314  }
315  if (H > h) {
316  size_t N = (H - h)*W;
317  if (clear_color_ptr) {
318  for (size_t i = 0; i < N; ++i)
319  memcpy(dest_ptr + i * entry_size, clear_color_ptr, entry_size);
320  }
321  else
322  std::fill(dest_ptr, dest_ptr + N*entry_size, 0);
323  }
324  return create(ctx, dv1, level, cube_side, false, &palettes);
325 }

344 bool texture::deduce_file_names(const std::string& file_names, std::vector<std::string>& deduced_names)
345 {
346  if (std::find(file_names.begin(), file_names.end(), '{') != file_names.end()) {
347  std::vector<token> toks;
348  deduced_names.resize(6);
349  tokenizer(file_names).set_ws("").set_sep("{,}").bite_all(toks);
350  int selection = -1;
351  for (unsigned i=0; i<toks.size(); ++i) {
352  if (toks[i] == "{") {
353  if (selection != -1) {
354  std::cerr << "warning: nested {} not allowed in cubemap file names specification " << file_names << std::endl;
355  return false;
356  }
357  selection = 0;
358  }
359  else if (toks[i] == "}") {
360  if (selection == -1) {
361  std::cerr << "warning: } without opening { in cubemap file names specification " << file_names << std::endl;
362  return false;
363  }
364  if (selection != 5) {
365  std::cerr << "warning: no 6 file names specified for creating cubemap from images " << file_names << std::endl;
366  return false;
367  }
368  selection = -1;
369  }
370  else if (toks[i] == ",") {
371  if (selection == -1) {
372  std::cerr << "warning: , arising outside {} in cubemap file names specification " << file_names << std::endl;
373  return false;
374  }
375  ++selection;
376  }
377  else {
378  if (selection == -1) {
379  for (unsigned j=0; j<6; ++j)
380  deduced_names[j] += to_string(toks[i]);
381  }
382  else {
383  if (selection == 6) {
384  std::cerr << "warning: more than 6 files names specified for cubemap creation " << file_names << std::endl;
385  return false;
386  }
387  deduced_names[selection] += to_string(toks[i]);
388  }
389  }
390  }
391  }
392  else {
393  std::string path_prefix = get_path(file_names);
394  if (!path_prefix.empty())
395  path_prefix += '/';
396  void* handle = find_first(file_names);
397  while (handle != 0) {
398  deduced_names.push_back(path_prefix+find_name(handle));
399  handle = find_next(handle);
400  }
401  if (deduced_names.size() != 6) {
402  std::cerr << "warning: not exactly 6 files names specified for cubemap creation " << file_names << std::endl;
403  return false;
404  }
405  }
406  return true;
407 }

409 bool texture::create_from_images(const context& ctx, const std::string& file_names, int level)
410 {
411  std::vector<std::string> deduced_names;
412  if (!deduce_file_names(file_names, deduced_names))
413  return false;
414  bool success = true;
415  for (unsigned i=0; success && i<6; ++i)

422  return success;

426 bool texture::write_to_file(context& ctx, const std::string& file_name, unsigned int z_or_cube_size, float depth_map_gamma, const std::string& options) const
427 {
428  std::string& last_error = static_cast<const cgv::render::texture_base*>(this)->last_error;
429  if (!is_created()) {
430  last_error = "texture must be created for write_to_file";
431  return false;
432  }
433  frame_buffer fb;
434  if (!fb.create(ctx,int(get_width()), int(get_height()))) {
435  last_error = "could not create frame buffer object for write_to_file";
436  return false;
437  }
438  if (z_or_cube_size != -1) {
439  if (!fb.attach(ctx,*this,z_or_cube_size,0,0)) {
440  last_error = "could not attach texture for write_to_file";
441  return false;
442  }
443  }
444  else {
445  if (!fb.attach(ctx,*this)) {
446  last_error = "could not attach texture for write_to_file";
447  return false;
448  }
449  }
450  bool is_depth = get_standard_component_format() == CF_D;
451  data_format df("uint8[R,G,B]");
452  if (is_depth) {
453  render_buffer rb("[R,G,B]");
454  df = data_format("uint32[D]");
455  rb.create(ctx, int(get_width()), int(get_height()));
456  if (!fb.attach(ctx, rb)) {
457  last_error = "could not attach color buffer for write_to_file";
458  return false;
459  }
460  }
461  if (!fb.is_complete(ctx)) {
462  last_error = "frame buffer object not complete for write_to_file due to\n";
463  last_error += fb.last_error;
464  return false;
465  }
466  fb.enable(ctx);
467  df.set_width(get_width());
468  df.set_height(get_height());
469  data_view dv(&df);
470  ctx.read_frame_buffer(dv, 0, 0, cgv::render::FB_0);
471  fb.disable(ctx);
472 
473  if (is_depth) {
474  cgv::utils::statistics stats;
475  size_t i, n = get_width()*get_height();
476  for (i = 0; i < n; ++i)
477  stats.update(dv.get_ptr<cgv::type::uint32_type>()[i]);
478  cgv::type::uint32_type min_val = cgv::type::uint32_type(stats.get_min());
479  cgv::type::uint32_type max_val = cgv::type::uint32_type(stats.get_max());
480  data_format df2(get_width(), get_height(), cgv::type::info::TI_UINT8, cgv::data::CF_RGB);
481  data_view dv2(&df2);
482  for (i = 0; i < n; ++i) {
483  double mapped_value = pow(double(dv.get_ptr<cgv::type::uint32_type>()[i] - min_val) / (max_val - min_val), depth_map_gamma);
484  cgv::type::uint8_type v = cgv::type::uint8_type(255 * mapped_value);
485  dv2.get_ptr<cgv::math::fvec<cgv::type::uint8_type, 3> >()[i].set(v, v, v);
486  }
487  image_writer w(file_name);
488  if (!w.write_image(dv2)) {
489  last_error = "could not write image file ";
490  last_error += file_name;
491  return false;
492  }
493  return true;
494  }
495  image_writer w(file_name);
496  w.multi_set(options);
497  if (!w.write_image(dv)) {
498  last_error = "could not write image file ";
499  last_error += file_name;
500  return false;
501  }
502  return true;
503 }

565 bool texture::create(const context& ctx, const cgv::data::const_data_view& data, int level, int cube_side, int num_array_layers, const std::vector<data_view>* palettes)
566 {
567  const data_format& f = *data.get_format();
568  TextureType tt = (TextureType)f.get_nr_dimensions();
569 
570  if(cube_side > -1) {
571  if(tt == TT_2D)
572  tt = TT_CUBEMAP;
573  } else if(num_array_layers != 0) {
574  if(num_array_layers < 0) {
575  // automatic inference of layers from texture dimensions
576  unsigned n_dims = f.get_nr_dimensions();
577  if(n_dims == 2)
578  tt = TT_1D_ARRAY;
579  if(n_dims == 3)
580  tt = TT_2D_ARRAY;
581  } else {
582  switch(tt) {
583  case TT_1D: tt = TT_1D_ARRAY; break;
584  case TT_2D: tt = TT_2D_ARRAY; break;
585  case TT_3D: tt = TT_2D_ARRAY; break;
586  }
587  }
588  }
589  // TODO: replace is currently only allowed for non-array type textures. If this changes make sure to modify the replace method accordingly including generating mipmaps.
590  if ((tt == TT_1D || tt == TT_2D || tt == TT_3D) && is_created()) {
591  bool replace_allowed = tt == this->tt;
592  for(unsigned i = 0; replace_allowed && i < get_nr_dimensions(); ++i)
593  if(get_resolution(i) != f.get_resolution(i))
594  replace_allowed = false;
595  if (replace_allowed && level < 1) {
596  switch (tt) {
597  case TT_1D : return replace(ctx, 0, data, level, palettes);
598  case TT_2D : return replace(ctx, 0, 0, data, level, palettes);
599  case TT_3D : return replace(ctx, 0, 0, 0, data, level, palettes);
600  }
601  return false;
602  }
603  destruct(ctx);
604  }
605  if (level < 1) {
606  set_nr_dimensions(data.get_format()->get_nr_dimensions());
607  for(unsigned int i = 0; i < get_nr_dimensions(); ++i)
608  set_resolution(i, data.get_format()->get_resolution(i));
609  if (get_nr_components() != data.get_format()->get_nr_components())
610  static_cast<component_format&>(*this) = *data.get_format();
611  if (level == -1 || !internal_format)
612  find_best_format(ctx, palettes);
613  }
614  return complete_create(ctx, ctx.texture_create(*this, *this, data, level, cube_side, num_array_layers, palettes));
615 }
