278 bool texture::create_from_image(cgv::data::data_format& df, cgv::data::data_view& dv, const context& ctx,
279  const std::string& file_name, unsigned char* clear_color_ptr, int level, int cube_side)
280 {
281  bool ensure_power_of_two = clear_color_ptr != 0;
282  std::string fn = file_name;
283  if (df.empty()) {
284  if (fn.empty()) {
285  cgv::render::render_component::last_error = "attempt to create texture from empty file name";
286  return false;
287  }
288  }
289  std::vector<data_format> palette_formats;
290  std::vector<data_view> palettes;
291  if (!fn.empty()) {
292  image_reader ir(df, &palette_formats);
293  if (!ir.read_image(fn, dv)) {
294  cgv::render::render_component::last_error = "error reading image file ";
295  cgv::render::render_component::last_error += fn;
296  cgv::render::render_component::last_error += ": ";
297  cgv::render::render_component::last_error += ir.get_last_error();
298  return false;
299  }
300  for (unsigned i=0; i<palette_formats.size(); ++i) {
301  palettes.push_back(data_view());
302  if (!ir.read_palette(i, palettes.back())) {
303  cgv::render::render_component::last_error = "error reading palette";
304  return false;
305  }
306  }
307  }
308  if (cube_side < 1)
309  destruct(ctx);
310  size_t w = df.get_width(), h = df.get_height();
311  size_t W = w, H = h;
312  data_format df1(df);
313  if (ensure_power_of_two && (!is_power_of_two(w) || !is_power_of_two(h))) {
314  W = power_of_two_ub(df.get_width());
315  H = power_of_two_ub(df.get_height());
316  df1.set_width(W);
317  df1.set_height(H);
318  }
319  //float ext[2] = { (float)w/W, (float)h/H };
320  data_view dv1(&df1);
321  unsigned entry_size = df.get_entry_size();
322  const unsigned char* src_ptr = dv.get_ptr<unsigned char>();
323  unsigned char* dest_ptr = dv1.get_ptr<unsigned char>();
324  unsigned char* dest_ptr_end = dest_ptr + entry_size * W*H;
325  for (unsigned y = 0; y < h; ++y) {
326  dest_ptr_end -= W * entry_size;
327  memcpy(dest_ptr_end, src_ptr, w*entry_size);
328  if (clear_color_ptr) {
329  for (size_t x = w; x < W; ++x)
330  memcpy(dest_ptr_end + x * entry_size, clear_color_ptr, entry_size);
331  }
332  else
333  std::fill(dest_ptr_end + w * entry_size, dest_ptr_end + W*entry_size, 0);
334  src_ptr += w * entry_size;
335  }
336  if (H > h) {
337  size_t N = (H - h)*W;
338  if (clear_color_ptr) {
339  for (size_t i = 0; i < N; ++i)
340  memcpy(dest_ptr + i * entry_size, clear_color_ptr, entry_size);
341  }
342  else
343  std::fill(dest_ptr, dest_ptr + N*entry_size, 0);
344  }
345  return create(ctx, dv1, level, cube_side, false, &palettes);
346 }

365 bool texture::deduce_file_names(const std::string& file_names, std::vector<std::string>& deduced_names)
366 {
367  if (std::find(file_names.begin(), file_names.end(), '{') != file_names.end()) {
368  std::vector<token> toks;
369  deduced_names.resize(6);
370  tokenizer(file_names).set_ws("").set_sep("{,}").bite_all(toks);
371  int selection = -1;
372  for (unsigned i=0; i<toks.size(); ++i) {
373  if (toks[i] == "{") {
374  if (selection != -1) {
375  std::cerr << "warning: nested {} not allowed in cubemap file names specification " << file_names << std::endl;
376  return false;
377  }
378  selection = 0;
379  }
380  else if (toks[i] == "}") {
381  if (selection == -1) {
382  std::cerr << "warning: } without opening { in cubemap file names specification " << file_names << std::endl;
383  return false;
384  }
385  if (selection != 5) {
386  std::cerr << "warning: no 6 file names specified for creating cubemap from images " << file_names << std::endl;
387  return false;
388  }
389  selection = -1;
390  }
391  else if (toks[i] == ",") {
392  if (selection == -1) {
393  std::cerr << "warning: , arising outside {} in cubemap file names specification " << file_names << std::endl;
394  return false;
395  }
396  ++selection;
397  }
398  else {
399  if (selection == -1) {
400  for (unsigned j=0; j<6; ++j)
401  deduced_names[j] += to_string(toks[i]);
402  }
403  else {
404  if (selection == 6) {
405  std::cerr << "warning: more than 6 files names specified for cubemap creation " << file_names << std::endl;
406  return false;
407  }
408  deduced_names[selection] += to_string(toks[i]);
409  }
410  }
411  }
412  }
413  else {
414  std::string path_prefix = get_path(file_names);
415  if (!path_prefix.empty())
416  path_prefix += '/';
417  void* handle = find_first(file_names);
418  while (handle != 0) {
419  deduced_names.push_back(path_prefix+find_name(handle));
420  handle = find_next(handle);
421  }
422  if (deduced_names.size() != 6) {
423  std::cerr << "warning: not exactly 6 files names specified for cubemap creation " << file_names << std::endl;
424  return false;
425  }
426  }
427  return true;
428 }

430 bool texture::create_from_images(const context& ctx, const std::string& file_names, int level)
431 {
432  std::vector<std::string> deduced_names;
433  if (!deduce_file_names(file_names, deduced_names))
434  return false;
435  bool success = true;
436  for (unsigned i=0; success && i<6; ++i)
437  if (!create_from_image(ctx, deduced_names[i],0,0,0,level,i)) {
438  success = false;
439  std::cerr << "could not create cubemap side " << i << " from image " << deduced_names[i] << std::endl;
440  }

442  destruct(ctx);
443  return success;
444 }

447 bool texture::write_to_file(context& ctx, const std::string& file_name, unsigned int z_or_cube_size, float depth_map_gamma, const std::string& options) const
448 {
449  std::string& last_error = static_cast<const cgv::render::texture_base*>(this)->last_error;
450  if (!is_created()) {
451  last_error = "texture must be created for write_to_file";
452  return false;
453  }
454  frame_buffer fb;
455  if (!fb.create(ctx,int(get_width()), int(get_height()))) {
456  last_error = "could not create frame buffer object for write_to_file";
457  return false;
458  }
459  if (z_or_cube_size != -1) {
460  if (!fb.attach(ctx,*this,z_or_cube_size,0,0)) {
461  last_error = "could not attach texture for write_to_file";
462  return false;
463  }
464  }
465  else {
466  if (!fb.attach(ctx,*this)) {
467  last_error = "could not attach texture for write_to_file";
468  return false;
469  }
470  }
471  bool is_depth = get_standard_component_format() == CF_D;
472  data_format df("uint8[R,G,B]");
473  if (is_depth) {
474  render_buffer rb("[R,G,B]");
475  df = data_format("uint32[D]");
476  rb.create(ctx, int(get_width()), int(get_height()));
477  if (!fb.attach(ctx, rb)) {
478  last_error = "could not attach color buffer for write_to_file";
479  return false;
480  }
481  }
482  if (!fb.is_complete(ctx)) {
483  last_error = "frame buffer object not complete for write_to_file due to\n";
484  last_error += fb.last_error;
485  return false;
486  }
487  fb.enable(ctx);
488  df.set_width(get_width());
489  df.set_height(get_height());
490  data_view dv(&df);
491  ctx.read_frame_buffer(dv, 0, 0, cgv::render::FB_0);
492  fb.disable(ctx);
493 
494  if (is_depth) {
495  cgv::utils::statistics stats;
496  size_t i, n = get_width()*get_height();
497  for (i = 0; i < n; ++i)
498  stats.update(dv.get_ptr<cgv::type::uint32_type>()[i]);
499  cgv::type::uint32_type min_val = cgv::type::uint32_type(stats.get_min());
500  cgv::type::uint32_type max_val = cgv::type::uint32_type(stats.get_max());
501  data_format df2(get_width(), get_height(), cgv::type::info::TI_UINT8, cgv::data::CF_RGB);
502  data_view dv2(&df2);
503  for (i = 0; i < n; ++i) {
504  double mapped_value = pow(double(dv.get_ptr<cgv::type::uint32_type>()[i] - min_val) / (max_val - min_val), depth_map_gamma);
505  cgv::type::uint8_type v = cgv::type::uint8_type(255 * mapped_value);
506  dv2.get_ptr<cgv::math::fvec<cgv::type::uint8_type, 3> >()[i].set(v, v, v);
507  }
508  image_writer w(file_name);
509  if (!w.write_image(dv2)) {
510  last_error = "could not write image file ";
511  last_error += file_name;
512  return false;
513  }
514  return true;
515  }
516  image_writer w(file_name);
517  w.multi_set(options);
518  if (!w.write_image(dv)) {
519  last_error = "could not write image file ";
520  last_error += file_name;
521  return false;
522  }
523  return true;
524 }

586 bool texture::create(const context& ctx, const cgv::data::const_data_view& data, int level, int cube_side, int num_array_layers, const std::vector<data_view>* palettes)
587 {
588  const data_format& f = *data.get_format();
589  TextureType tt = (TextureType)f.get_nr_dimensions();
590 
591  if(cube_side > -1) {
592  if(tt == TT_2D)
593  tt = TT_CUBEMAP;
594  } else if(num_array_layers != 0) {
595  if(num_array_layers < 0) {
596  // automatic inference of layers from texture dimensions
597  unsigned n_dims = f.get_nr_dimensions();
598  if(n_dims == 2)
599  tt = TT_1D_ARRAY;
600  if(n_dims == 3)
601  tt = TT_2D_ARRAY;
602  } else {
603  switch(tt) {
604  case TT_1D: tt = TT_1D_ARRAY; break;
605  case TT_2D: tt = TT_2D_ARRAY; break;
606  case TT_3D: tt = TT_2D_ARRAY; break;
607  }
608  }
609  }
610  // TODO: replace is currently only allowed for non-array type textures. If this changes make sure to modify the replace method accordingly including generating mipmaps.
611  if ((tt == TT_1D || tt == TT_2D || tt == TT_3D) && is_created()) {
612  bool replace_allowed = tt == this->tt;
613  for(unsigned i = 0; replace_allowed && i < get_nr_dimensions(); ++i)
614  if(get_resolution(i) != f.get_resolution(i))
615  replace_allowed = false;
616  if (replace_allowed && level < 1) {
617  switch (tt) {
618  case TT_1D : return replace(ctx, 0, data, level, palettes);
619  case TT_2D : return replace(ctx, 0, 0, data, level, palettes);
620  case TT_3D : return replace(ctx, 0, 0, 0, data, level, palettes);
621  }
622  return false;
623  }
624  destruct(ctx);
625  }
626  if (level < 1) {
627  set_nr_dimensions(data.get_format()->get_nr_dimensions());
628  for(unsigned int i = 0; i < get_nr_dimensions(); ++i)
629  set_resolution(i, data.get_format()->get_resolution(i));
630  if (get_nr_components() != data.get_format()->get_nr_components())
631  static_cast<component_format&>(*this) = *data.get_format();
632  if (level == -1 || !internal_format)
633  find_best_format(ctx, palettes);
634  }
635  return complete_create(ctx, ctx.texture_create(*this, *this, data, level, cube_side, num_array_layers, palettes));
636 }
