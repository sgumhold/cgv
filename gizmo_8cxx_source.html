51bool gizmo::handle(cgv::gui::event& e, context& ctx) {
52
53 if(!view_ptr)
54 return false;
55
56 vec3 view_dir = view_ptr->get_view_dir();
57
58 unsigned et = e.get_kind();
59
60 if(et == cgv::gui::EID_MOUSE) {
61 cgv::gui::mouse_event& me = (cgv::gui::mouse_event&) e;
62 cgv::gui::MouseAction ma = me.get_action();
63
64 ivec2 viewport_size(ctx.get_width(), ctx.get_height());
65 ivec2 mpos(
66 static_cast<int>(me.get_x()),
67 viewport_size.y() - static_cast<int>(me.get_y()) - 1
68 );
69
70 cgv::math::ray3 ray(static_cast<vec2>(mpos), static_cast<vec2>(viewport_size), view_ptr->get_eye(), ctx.get_projection_matrix() * ctx.get_modelview_matrix());
71
72 switch(ma) {
73 case cgv::gui::MA_MOVE:
74 {
75 if(active)
76 break;
77
78 bool last_hover = hover;
79 InteractionFeature last_feature = feature;
80
81 hover = intersect(ray);
82
83 if(last_hover != hover || last_feature != feature) {
84 geometry_out_of_date = true;
85 return true;
86 }
87 } break;
88 case cgv::gui::MA_PRESS:
89 {
90 if(hover && feature != InteractionFeature::kNone) {
91 if(handle_drag(ray, view_dir, true)) {
92 active = true;
93 return true;
94 }
95 }
96 } break;
97 case cgv::gui::MA_RELEASE:
98 if(active) {
99 active = false;
100 feature = InteractionFeature::kNone;
101 geometry_out_of_date = true;
102 return true;
103 }
104 break;
105 case cgv::gui::MA_DRAG:
106 {
107 if(active) {
108 if(handle_drag(ray, view_dir, false))
109 return true;
110 }
111 } break;
112 default: break;
113 }
114 }
115
116 return false;
117}

134void gizmo::draw(context& ctx) {
135
136 if(!view_ptr)
137 return;
138
139 if(geometry_out_of_date)
140 create_geometry();
141
142 if(!active || scale < 0.0f)
143 scale = 0.3f * scale_coefficient * static_cast<float>(view_ptr->get_tan_of_half_of_fovy(true)) * dot(position - view_ptr->get_eye(), view_ptr->get_view_dir());
144
145 if(!active && align_to_view)
146 flip_factors = get_flip_factors();
147 else
148 flip_factors = vec3(1.0f);
149
150 ctx.push_modelview_matrix();
151 ctx.mul_modelview_matrix(cgv::math::translate4(position));
152 ctx.mul_modelview_matrix(cgv::math::scale4(vec3(scale) * flip_factors));
153
154 arrows.render(ctx, renderers.arrow);
155 rectangles.render(ctx, renderers.rectangle);
156 sphere.render(ctx, renderers.sphere);
157
158 ctx.pop_modelview_matrix();
159}

244bool gizmo::intersect_axis_aligned_rectangle(const cgv::math::ray3& r, int axis, const vec3& p, float size, float& t) const {
245
246 vec3 nml(0.0f);
247 nml[axis] = 1.0f;
248
249 float ext = size;
250
251 if(cgv::math::ray_plane_intersection(r, p, nml, t)) {
252 vec3 hp = r.position(t);
253 hp -= p;
254
255 vec2 uv;
256 switch(axis) {
257 case 0:
258 uv[0] = hp[1];
259 uv[1] = hp[2];
260 break;
261 case 1:
262 uv[0] = hp[0];
263 uv[1] = hp[2];
264 break;
265 case 2:
266 uv[0] = hp[0];
267 uv[1] = hp[1];
268 break;
269 default:
270 return false;
271 break;
272 }
273
274 uv += 0.5f*ext;
275
276 if(uv[0] >= 0.0f && uv[0] <= ext && uv[1] >= 0.0f && uv[1] <= ext) {
277 return true;
278 }
279 }
280
281 return false;
282}

284bool gizmo::intersect(const cgv::math::ray3& r) {
285
286 float min_t = std::numeric_limits<float>::max();
287
288 for(unsigned i = 0; i < 3; ++i) {
289 vec3 pa = position;
290 vec3 pb = position;
291 pb[i] += 1.0f * scale * flip_factors[i];
292
293 float t = std::numeric_limits<float>::max();
294 if(cgv::math::ray_cylinder_intersection2(r, pa, pb, 0.05f * scale, t)) {
295 if(t >= 0.0f && t < min_t) {
296 min_t = t;
297 feature = static_cast<InteractionFeature>(i + 1);
298 }
299 }
300 }
301
302 for(unsigned i = 0; i < 3; ++i) {
303 vec3 p = position + 0.25f * scale * flip_factors;
304 p[i] = position[i];
305
306 float t = std::numeric_limits<float>::max();
307 if(intersect_axis_aligned_rectangle(r, i, p, 0.3f * scale, t)) {
308 if(t >= 0.0f && t < min_t) {
309 min_t = t;
310 feature = static_cast<InteractionFeature>(i + 4);
311 }
312 }
313 }
314
315 vec2 ts(std::numeric_limits<float>::max());
316 if(cgv::math::ray_sphere_intersection(r, position, 0.1f * scale, ts)) {
317 if(ts.x() >= 0.0f && ts.x() < min_t) {
318 min_t = ts.x();
319 feature = InteractionFeature::kCenter;
320 }
321 }
322
323 return min_t > 0.0f && min_t < std::numeric_limits<float>::max();
324}

326bool gizmo::handle_drag(const cgv::math::ray3& r, const vec3& view_dir, bool drag_start) {
327
328 int axis_idx = get_axis_idx(feature);
329 vec3 axis = get_axis(feature);
330
331 vec3 plane_origin = drag_start ? position : last_position;
332 vec3 plane_normal = vec3(0.0f);
333
334 if(is_axis(feature)) {
335 vec3 angs;
336 for(unsigned i = 0; i < 3; ++i) {
337 float ang = -1.0f;
338 if(i != axis_idx) {
339 vec3 v(0.0f);
340 v[i] = 1.0f;
341 ang = abs(dot(v, view_dir));
342 }
343 angs[i] = ang;
344 }
345
346 plane_normal[cgv::math::max_index(angs)] = 1.0f;
347 } else if(is_plane(feature)) {
348 plane_normal = axis;
349 } else if(is_center(feature)) {
350 plane_normal = view_dir;
351 } else {
352 return false;
353 }
354
355 float t = -1.0f;
356 if(cgv::math::ray_plane_intersection(r, plane_origin, plane_normal, t)) {
357 if(drag_start) {
358 last_position = position;
359
360 if(is_axis(feature))
361 offset[0] = dot(axis, r.position(t) - position);
362 else
363 offset = r.position(t) - position;
364 } else {
365 vec3 new_position = position;
366
367 if(is_axis(feature)) {
368 new_position[axis_idx] = dot(axis, r.position(t)) - offset[0];
369 } else {
370 new_position = r.position(t) - offset;
371 }
372
373 set_position(new_position, true);
374 }
375
376 return true;
377 }
378
379 return false;
380}
