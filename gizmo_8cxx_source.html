52

54  return false;

71 
72  switch(ma) {
73  case cgv::gui::MA_MOVE:
74  {

80

82 
83  if(last_hover != hover || last_feature != feature) {
84  geometry_out_of_date = true;

86  }
87  } break;
88  case cgv::gui::MA_PRESS:

103  }

105  case cgv::gui::MA_DRAG:
106  {
107  if(active) {
108  if(handle_drag(ray, view_dir, false))
109  return true;
110  }
111  } break;

247 bool gizmo::intersect_axis_aligned_rectangle(const cgv::math::ray3& r, int axis, const vec3& p, float size, float& t) const {
248 
249  vec3 nml(0.0f);
250  nml[axis] = 1.0f;
251 
252  float ext = size;
253 
254  if(cgv::math::ray_plane_intersection(r, p, nml, t)) {
255  vec3 hp = r.position(t);
256  hp -= p;
257 
258  vec2 uv;
259  switch(axis) {
260  case 0:
261  uv[0] = hp[1];
262  uv[1] = hp[2];
263  break;
264  case 1:
265  uv[0] = hp[0];
266  uv[1] = hp[2];
267  break;
268  case 2:
269  uv[0] = hp[0];
270  uv[1] = hp[1];
271  break;
272  default:
273  return false;
274  break;
275  }
276 
277  uv += 0.5f*ext;
278 
279  if(uv[0] >= 0.0f && uv[0] <= ext && uv[1] >= 0.0f && uv[1] <= ext) {
280  return true;
281  }
282  }
283 
284  return false;

287 bool gizmo::intersect(const cgv::math::ray3& r) {
288 
289  float min_t = std::numeric_limits<float>::max();
290 
291  for(unsigned i = 0; i < 3; ++i) {
292  vec3 pa = position;
293  vec3 pb = position;
294  pb[i] += 1.0f * scale * flip_factors[i];
295 
296  float t = std::numeric_limits<float>::max();
297  if(cgv::math::ray_cylinder_intersection2(r, pa, pb, 0.05f * scale, t)) {
298  if(t >= 0.0f && t < min_t) {
299  min_t = t;
300  feature = static_cast<InteractionFeature>(i + 1);
301  }
302  }
303  }
304 
305  for(unsigned i = 0; i < 3; ++i) {
306  vec3 p = position + 0.25f * scale * flip_factors;
307  p[i] = position[i];
308 
309  float t = std::numeric_limits<float>::max();
310  if(intersect_axis_aligned_rectangle(r, i, p, 0.3f * scale, t)) {
311  if(t >= 0.0f && t < min_t) {
312  min_t = t;
313  feature = static_cast<InteractionFeature>(i + 4);
314  }
315  }
316  }
317 
318  vec2 ts(std::numeric_limits<float>::max());
319  if(cgv::math::ray_sphere_intersection(r, position, 0.1f * scale, ts)) {
320  if(ts.x() >= 0.0f && ts.x() < min_t) {
321  min_t = ts.x();
322  feature = InteractionFeature::kCenter;
323  }
324  }
325 
326  return min_t > 0.0f && min_t < std::numeric_limits<float>::max();
327 }

329 bool gizmo::handle_drag(const cgv::math::ray3& r, const vec3& view_dir, bool drag_start) {
330 
331  int axis_idx = get_axis_idx(feature);
332  vec3 axis = get_axis(feature);
333 
334  vec3 plane_origin = drag_start ? position : last_position;
335  vec3 plane_normal = vec3(0.0f);
336 
337  if(is_axis(feature)) {
338  vec3 angs;
339  for(unsigned i = 0; i < 3; ++i) {
340  float ang = -1.0f;
341  if(i != axis_idx) {
342  vec3 v(0.0f);
343  v[i] = 1.0f;
344  ang = abs(dot(v, view_dir));
345  }
346  angs[i] = ang;
347  }
348 
349  plane_normal[cgv::math::max_index(angs)] = 1.0f;
350  } else if(is_plane(feature)) {
351  plane_normal = axis;
352  } else if(is_center(feature)) {
353  plane_normal = view_dir;
354  } else {
355  return false;
356  }
357 
358  float t = -1.0f;
359  if(cgv::math::ray_plane_intersection(r, plane_origin, plane_normal, t)) {
360  if(drag_start) {
361  last_position = position;
362 
363  if(is_axis(feature))
364  offset[0] = dot(axis, r.position(t) - position);
365  else
366  offset = r.position(t) - position;
367  } else {
368  vec3 new_position = position;
369 
370  if(is_axis(feature)) {
371  new_position[axis_idx] = dot(axis, r.position(t)) - offset[0];
372  } else {
373  new_position = r.position(t) - offset;
374  }
375 
376  set_position(new_position, true);
377  }
378 
379  return true;
380  }
381 
382  return false;
383 }
