128 bool shader_program::collect_program(const std::string& file_name, bool use_cache, std::vector<std::string>& file_names)
129 {
130  std::string fn = shader_code::find_file(file_name);
131  if (fn.empty())
132  return false;
133  std::string content;
134  if (!cgv::base::read_data_file(fn, content, true))
135  return false;
136 #if WIN32
137  shader_code::decode_if_base64(content);
138 #endif
139  std::vector<line> lines;
140  split_to_lines(content, lines);
141  bool added_file = false;
142  std::string old_shader_path = get_shader_config()->shader_path;
143  std::string path = file::get_path(file_name);
144  if (!path.empty())
145  get_shader_config()->shader_path = path+";"+get_shader_config()->shader_path;
146 
147  for (auto line : lines) {
148  std::string l = to_string((const token&)line);
149  if (l.substr(0,5) == "file:")
150  added_file = collect_file(l.substr(5), use_cache, file_names) || added_file;
151  else if (l.substr(0,12) == "vertex_file:")
152  added_file = collect_file(l.substr(12), use_cache, file_names) || added_file;
153  else if (l.substr(0, 14) == "geometry_file:")
154  added_file = collect_file(l.substr(14), use_cache, file_names) || added_file;
155  else if (l.substr(0, 26) == "tessellation_control_file:")
156  added_file = collect_file(l.substr(26), use_cache, file_names) || added_file;
157  else if (l.substr(0, 29) == "tessellation_evaluation_file:")
158  added_file = collect_file(l.substr(29), use_cache, file_names) || added_file;
159  else if (l.substr(0,14) == "fragment_file:")
160  added_file = collect_file(l.substr(14), use_cache, file_names) || added_file;
161  else if (l.substr(0,6) == "files:")
162  added_file = collect_files(l.substr(6), use_cache, file_names) || added_file;
163  else if (l.substr(0,4) == "dir:")
164  added_file = collect_dir(l.substr(4), false, file_names) || added_file;
165  else if (l.substr(0,8) == "rec_dir:")
166  added_file = collect_dir(l.substr(8), true, file_names) || added_file;
167  else if (l.substr(0,8) == "program:")
168  added_file = collect_program(l.substr(8), use_cache, file_names) || added_file;
169  }
170 
171  get_shader_config()->shader_path = old_shader_path;
172  return added_file;
173 }

319 std::vector<shader_define_map> shader_program::extract_instances(std::string file_name)
320 {
321  std::string content;
322  std::vector<line> lines;
323  std::vector<shader_define_map> result;
324  if (!open_program_file(file_name, false, content, lines))
325  return result;
326  for (unsigned int i = 0; i < lines.size(); ++i) {
327  token tok = lines[i];
328  while (tok.begin < tok.end && cgv::utils::is_space(*tok.begin))
329  ++tok.begin;
330  std::string l = to_string(tok);
331  if (l.empty() || l[0] == '/')
332  continue;
333  if (l.substr(0, 9) != "instance:")
334  continue;
335  std::string defs=l.substr(9);
336  std::vector<token> toks;
337  split_to_tokens(defs, toks, "", false, "", "", ";");
338  shader_define_map defines;
339  for (const auto& t : toks) {
340  std::vector<token> sides;
341  split_to_tokens(t, sides, "", false, "", "", "=");
342  std::vector<std::string> S;
343  for (auto& s : sides) {
344  while (s.begin < s.end && cgv::utils::is_space(*s.begin))
345  ++s.begin;
346  while (s.begin < s.end && cgv::utils::is_space(s.end[-1]))
347  --s.end;
348  if (s.begin < s.end)
349  S.push_back(to_string(s));
350  }
351  if (S.size() == 2)
352  defines[S[0]] = S[1];
353  }
354  result.push_back(defines);
355  }
356  return result;
357 }

359 bool shader_program::attach_program(const context& ctx, std::string file_name, bool show_error, const shader_define_map& defines)
360 {
361  std::string content;
362  std::vector<line> lines;
363  if (!open_program_file(file_name, ctx.is_shader_file_cache_enabled(), content, lines, &last_error)) {
364  std::cerr << last_error << std::endl;
365  return false;
366  }
367  if (get_shader_config()->show_file_paths)
368  std::cout << "read shader program <" << file_name << ">" << std::endl;
369  std::string old_shader_path = get_shader_config()->shader_path;
370  std::string path = file::get_path(file_name);
371  if (!path.empty())
372  get_shader_config()->shader_path = path+";"+get_shader_config()->shader_path;
373 
374  bool no_error = true;
375  std::string error = "2 : attach command failed";
376  for (unsigned int i=0; i<lines.size(); ++i) {
377  token tok = lines[i];
378  while (tok.begin < tok.end && cgv::utils::is_space(*tok.begin))
379  ++tok.begin;
380  std::string l = to_string(tok);
381 
382  bool success = true;
383  // ignore empty lines
384  if (l.empty())
385  continue;
386  // ignore comments
387  if (l[0] == '/')
388  continue;
389  if (l.substr(0,5) == "file:")
390  success = attach_file(ctx, l.substr(5), ST_DETECT, defines);
391  else if (l.substr(0,12) == "vertex_file:")
392  success = attach_file(ctx, l.substr(12), ST_VERTEX, defines);
393  else if (l.substr(0,14) == "geometry_file:")
394  success = attach_file(ctx, l.substr(14), ST_GEOMETRY, defines);
395  else if (l.substr(0, 26) == "tessellation_control_file:")
396  success = attach_file(ctx, l.substr(26), ST_TESS_CONTROL, defines);
397  else if (l.substr(0, 29) == "tessellation_evaluation_file:")
398  success = attach_file(ctx, l.substr(29), ST_TESS_EVALUATION, defines);
399  else if (l.substr(0,14) == "fragment_file:")
400  success = attach_file(ctx, l.substr(14), ST_FRAGMENT, defines);
401  else if(l.substr(0, 13) == "compute_file:")
402  success = attach_file(ctx, l.substr(13), ST_COMPUTE, defines);
403  else if (l.substr(0,6) == "files:")
404  success = attach_files(ctx, l.substr(6), defines);
405  else if (l.substr(0,4) == "dir:")
406  success = attach_dir(ctx, l.substr(4), false);
407  else if (l.substr(0,8) == "rec_dir:")
408  success = attach_dir(ctx, l.substr(8), true);
409  else if (l.substr(0,8) == "program:")
410  success = attach_program(ctx, l.substr(8));
411  else if (l.substr(0,21) == "geometry_shader_info:") {
412  std::vector<token> toks;
413  std::string l1 = l.substr(21);
414  tokenizer(l1).set_ws(";").bite_all(toks);
415  if (toks.size() == 3) {
416  PrimitiveType i_pt = PT_UNDEF, o_pt = PT_UNDEF;
417  int pi, count = 0;
418  for (pi = PT_UNDEF+1; pi < PT_LAST; ++pi) {
419  PrimitiveType pt = (PrimitiveType)pi;
420  std::string s = to_string(pt);
421  if (s == to_string(toks[0]))
422  i_pt = pt;
423  if (s == to_string(toks[1]))
424  o_pt = pt;
425  }
426  if (i_pt == PT_UNDEF) {
427  error = "4 : unknown input_type for geometry shader <";
428  error += to_string(toks[0])+">";
429  success = false;
430  }
431  else if (o_pt == PT_UNDEF) {
432  error = "5 : unknown ouput_type for geometry shader <";
433  error += to_string(toks[1])+">";
434  success = false;
435  }
436  else if (!is_integer(toks[2].begin,toks[2].end,count)) {
437  error = "6 : max_output_count of geometry shader must be an integer but received <";
438  error += to_string(toks[2])+">";
439  success = false;
440  }
441  else {
442  set_geometry_shader_info(i_pt, o_pt, count);
443  }
444  }
445  else {
446  success = false;
447  error = "3 : geometry_shader_info takes three arguments separated by colons";
448  }
449  }
450  else if (l.substr(0, 9) == "instance:") {
451  }
452  else if (show_error) {
453  std::cerr << file_name.c_str() << " (" << i + 1
454  << "): warning G0001 : syntax error in line '"
455  << l.c_str() << "'" << std::endl;
456  }
457  if (!success) {
458  if (show_error) {
459  std::cerr << file_name.c_str() << " (" << i+1
460  << "): error G000" << error.c_str() << std::endl;
461  }
462  no_error = false;
463  }
464  }
465 
466  get_shader_config()->shader_path = old_shader_path;
467  if (show_error && !no_error)
468  std::cerr << last_error << std::endl;
469  return no_error;
470 }
