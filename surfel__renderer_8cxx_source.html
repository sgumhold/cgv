102  bool surfel_renderer::enable(cgv::render::context& ctx)
103  {
104  const surfel_render_style& srs = get_style<surfel_render_style>();
105 
106  bool res;
107  if (!srs.use_group_color && has_indexed_colors) {
108  bool tmp = has_colors;
109  has_colors = true;
110  res = surface_renderer::enable(ctx);
111  has_colors = tmp;
112  }
113  else
114  res = surface_renderer::enable(ctx);
115 
116  glPointSize(srs.point_size);
117  if (srs.blend_points) {
118  glEnable(GL_BLEND);
119  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
120  }
121  if (ref_prog().is_linked()) {
122  if (!has_normals)
123  ref_prog().set_attribute(ctx, "normal", cgv::vec3(0.0f, 0.0f, 1.0f));
124  if (!has_point_sizes)

126  ref_prog().set_uniform(ctx, "use_color_index", has_indexed_colors);

128  ref_prog().set_uniform(ctx, "reference_point_size", reference_point_size);
129  ref_prog().set_uniform(ctx, "use_group_point_size", srs.use_group_point_size);
130  ref_prog().set_uniform(ctx, "orient_splats", srs.orient_splats);
131  float pixel_extent_per_depth = (float)(2.0*tan(0.5*0.0174532925199*y_view_angle) / ctx.get_height());
132  ref_prog().set_uniform(ctx, "pixel_extent_per_depth", pixel_extent_per_depth);
133  ref_prog().set_uniform(ctx, "blend_width_in_pixel", srs.blend_width_in_pixel);
134  ref_prog().set_uniform(ctx, "percentual_halo_width", 0.01f*srs.percentual_halo_width);

137  ref_prog().set_uniform(ctx, "halo_color_strength", srs.halo_color_strength);
138  ref_prog().set_uniform(ctx, "surface_offset", srs.surface_offset);
139  }
