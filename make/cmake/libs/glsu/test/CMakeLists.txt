cmake_minimum_required (VERSION 2.6)

INCLUDE(FindGLUT)
if (NOT GLUT_FOUND )
	if (WIN32)
		find_path( GLUT_INCLUDE_DIR NAMES GL/glut.h PATHS  $ENV{GLUT_ROOT_PATH}/include )
		find_library( GLUT_glut_LIBRARY NAMES glut glut32 freeglut PATHS $ENV{GLUT_ROOT_PATH}/Release )
		if (GLUT_INCLUDE_DIR AND GLUT_glut_LIBRARY)
		    SET( GLUT_LIBRARIES ${GLUT_glut_LIBRARY} )
		    SET( GLUT_FOUND "YES" )
		    SET (GLUT_LIBRARY ${GLUT_LIBRARIES})
		    SET (GLUT_INCLUDE_PATH ${GLUT_INCLUDE_DIR})
		    MARK_AS_ADVANCED( GLUT_INCLUDE_DIR GLUT_glut_LIBRARY )
    
		endif(GLUT_INCLUDE_DIR AND GLUT_glut_LIBRARY)
	endif (WIN32)
endif (NOT GLUT_FOUND )
if ( GLUT_FOUND )
	include_directories( ${GLUT_INCLUDE_DIR} )
	add_executable(test_glsu_glut test_glsu.cxx test_glsu_glut.cxx)
	target_link_libraries(test_glsu_glut glsu ${GLUT_LIBRARIES})
endif ( GLUT_FOUND )

INCLUDE(FindFLTK2)
if ( FLTK2_FOUND )
	include_directories( ${FLTK2_INCLUDE_DIR} )
	add_executable(test_glsu_fltk test_glsu.cxx test_glsu_fltk.cxx)
	target_link_libraries(test_glsu_fltk glsu ${FLTK2_LIBRARIES} )
endif ( FLTK2_FOUND )
