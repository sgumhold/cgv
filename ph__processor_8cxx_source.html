378  bool ph_processor::process_list(unsigned int i)
379  {
380  variant v0;
381  commands[i].expressions[0].evaluate(v0, this);
382  if (exit_code != 0)
383  return true;
384  if (!check_evaluation_error(i, 0))
385  return false;
386  variant v1;
387  commands[i].expressions[1].evaluate(v1, this);
388  if (exit_code != 0)
389  return true;
390  if (!check_evaluation_error(i, 1))
391  return false;
392  variant v2;
393  commands[i].expressions[2].evaluate(v2, this);
394  if (exit_code != 0)
395  return true;
396  if (!check_evaluation_error(i, 2))
397  return false;
398  if (!generate_output)
399  return true;
400 
401  std::string s0 = v0.get_str();
402  token t0(s0);
403  std::string s1 = v1.get_str();
404  std::string s2 = v2.get_str();
405  token t2(s2);
406  std::vector<token> toks;
407  const char* ptr_base = 0;
408  do {
409  token p0, p2;
410  // std::cout << "tokens = " << to_string(t0).c_str() << " <-> "
411  // << to_string(t2).c_str() << std::endl;
412 
413  bite_common_prefix(t0, t2, p0, p2);
414  if (t0.empty() || !is_digit(*t0.begin) ||
415  t2.empty() || !is_digit(*t2.begin)) {
416  if (!p0.empty() && is_digit(p0.end[-1])) {
417  t0.begin = --p0.end;
418  t2.begin = --p2.end;
419  }
420  else if (!(t0.empty() && t2.empty())) {
421  error(std::string("error in list [") + s0 + ";" + s1 + ";" + s2 + "]\n can only iterate between two numbers", commands[i]);
422  return false;
423  }
424  }
425  toks.push_back(p0);
426  if (t0.empty() || t2.empty()) {
427  if (t0.empty() && t2.empty())
428  break;
429  else {
430  error(std::string("error in list [") + s0 + ";" + s1 + ";" + s2 + "]\n 1st and 3rd expression split into different number of terms", commands[i]);
431  return false;
432  }
433  }
434  // std::cout << "bite off prefix " << p0.str().c_str() << " = "
435  // << p2.str().c_str() << std::endl;
436  int n0, n2;
437  if (!bite_digits(t0, n0) || !bite_digits(t2, n2)) {
438  error(std::string("error in list [") + s0 + ";" + s1 + ";" + s2 + "]\n can only iterate between two numbers (2)", commands[i]);
439  return false;
440  }
441  toks.push_back(token(ptr_base + n0, ptr_base + n2));
442  // std::cout << "bite off numbers " << n0 << " and " << n2 << std::endl;
443  } while (!(t0.empty() && t2.empty()));
444 
445  if (toks.empty())
446  return true;
447 
448  int n = 1;
449  if (toks.size() > 1) {
450  n = (int)((toks[1].end - ptr_base) - (toks[1].begin - ptr_base));
451  n += 1;
452  }
453  unsigned int j;
454  for (j = 3; j < toks.size(); j += 2) {
455  int nn = (int)((toks[j].end - ptr_base) - (toks[j].begin - ptr_base));
456  nn += 1;
457  if (nn != n) {
458  error(std::string("error in list [") + s0 + ";" + s1 + ";" + s2 + "]\n several number ranges of different size not allowed", commands[i]);
459  return false;
460  }
461  }
462  for (int k = 0; k < n; ++k) {
463  if (k > 0)
464  (*os) << s1.c_str();
465  for (j = 0; j < toks.size(); j += 2) {
466  (*os) << to_string(toks[j]).c_str();
467  if (j + 1 < toks.size())
468  (*os) << (int)(toks[j + 1].begin - ptr_base) + k;
469  }
470  }
471  return true;
472  }

513  bool ph_processor::reflect(const token& code, const token& comment)
514  {
515  // split code fragment into tokens
516  tokenizer t(code);
517  t.set_sep("()[];");
518  std::vector<token> toks;
519  t.skip_whitespaces();
520  t.reverse_skip_whitespaces();
521  bite_all(t, toks);
522 
523  // check that at least one token is present
524  if (toks.empty()) {
525  error("no code to be reflected here", code);
526  return false;
527  }
528 
529  // check for closing code parenthesis
530  if (to_string(toks[0]) == "}") {
531 
532  // these are only valid if the parenthesis stack is not empty
533  if (rp_stack.empty()) {
534  error("reflect end of something without starting something", code);
535  return false;
536  }
537 
538  // either close the top most namespace
539  if (rp_stack.back() == RP_NAMESPACE) {
540  if (namespace_stack.empty()) {
541  error("reflect end of namespace without a namespace started", code);
542  return false;
543  }
544  if (debug_reflection)
545  std::cout << "reflect end of namespace " << namespace_stack.back().c_str() << std::endl;
546  namespace_stack.pop_back();
547  }
548  // or the top most compound definition
549  else {
550  if (compound_stack.empty()) {
551  error("reflect end of compound without a compound started", code);
552  return false;
553  }
554  if (debug_reflection)
555  std::cout << "reflect end of compound " << compound_stack.back()->get_element(2).get_str().c_str() << std::endl;
556  compound_stack.pop_back();
557  compound_access_type_stack.pop_back();
558  }
559  rp_stack.pop_back();
560  return true;
561  }
562  // check for access type declaration
563  std::string new_access_type;
564  if (to_string(toks[0]) == "private" || to_string(toks[0]) == "private:")
565  new_access_type = "private";
566  else if (to_string(toks[0]) == "protected" || to_string(toks[0]) == "protected:")
567  new_access_type = "protected";
568  else if (to_string(toks[0]) == "public" || to_string(toks[0]) == "public:")
569  new_access_type = "public";
570  if (!new_access_type.empty()) {
571  if (rp_stack.empty() || rp_stack.back() != RP_COMPOUND) {
572  error("access type definition only valid inside compound declaration", toks[0]);
573  return false;
574  }
575  if (compound_access_type_stack.empty()) {
576  error("compound_access_type_stack empty", toks[0]);
577  return false;
578  }
579  compound_access_type_stack.back() = new_access_type;
580  return true;
581  }
582 
583  // in all other cases there must be at least two tokens
584  if (toks.size() < 2) {
585  error("code splits only into one token and cannot be reflected", code);
586  return false;
587  }
588 
589  if (to_string(toks[0]) == "typedef") {
590  // parse typedef declaration
591  if (toks.size() < 3) {
592  error("typedef with only two tokens cannot be reflected", code);
593  return false;
594  }
595  token name_tok = toks.back();
596  token type_tok(toks[1].begin, toks[toks.size() - 2].end);
597 
598  // construct reflection info
599  variant type_def = variant(variant::map_type());
600  if (reflect_version_2)
601  type_def.ref_element("reflect").set_str("typedef");
602  type_def.ref_element("namespaces") = construct_namespace_list();
603  type_def.ref_element("namespace_prefix") = construct_namespace_sequence();
604  type_def.ref_element("name") = variant(to_string(name_tok));
605  type_def.ref_element("type") = variant(to_string(type_tok));
606  type_def.ref_element("comment") = variant(to_string(comment));
607 
608  // store reflection info
609  ensure_reflect_info();
610  if (reflect_version_2) {
611  if (compound_stack.size() > 0)
612  compound_stack.back()->ref_element("elements").append_to_list(type_def);
613  else
614  reflect_info->ref_element("elements").append_to_list(type_def);
615  }
616  if (reflect_version_1) {
617  if (compound_stack.size() > 0)
618  compound_stack.back()->ref_element("typedefs").append_to_list(type_def);
619  else
620  reflect_info->ref_element("typedefs").append_to_list(type_def);
621  }
622  // generate debug info
623  if (debug_reflection)
624  std::cout << "reflect typedef " << type_def << std::endl;
625  return true;
626  }
627  else if (to_string(toks[0]) == "#include") {
628  // parse include declaration
629  if (toks.size() < 2) {
630  error("cannot reflect include without file name", code);
631  return false;
632  }
633  // store reflection info
634  ensure_reflect_info();
635  reflect_info->ref_element("includes").append_to_list(variant(to_string(toks[1])));
636  // generate debug info
637  if (debug_reflection)
638  std::cout << "reflect include " << to_string(toks[1]).c_str() << std::endl;
639  return true;
640  }
641  else if (to_string(toks[0]) == "enum") {
642  // parse enum declaration and generate list of enum entries
643  variant enum_entries = variant(variant::list_type());
644  // skip CGV_API macro
645  bool contains_cgv_api = false;
646  int idx = 1;
647  if (to_string(toks[1]) == "/*CGV_API*/") {
648  contains_cgv_api = true;
649  idx = 2;
650  }
651  // extract token of class name
652  token name_tok = toks[idx];
653  tokenizer t(token(name_tok.end, commands.back().end));
654  t.set_sep(",={}");
655 
656  int v = 0;
657  bool define_name = false;
658  while (!t.skip_ws_check_empty()) {
659  token tok = t.bite();
660  if (to_string(tok) == "{") {
661  define_name = true;
662  continue;
663  }
664  if (to_string(tok) == "}")
665  break;
666  if (to_string(tok) == ",") {
667  define_name = true;

669  }
670  if (to_string(tok) == "=") {

673  }

677  enum_entry.ref_element("value").set_int(v);
678  enum_entries.append_to_list(enum_entry);
679  ++v;
680  }
681  else {
682  int w;

685  return false;

688  v = w + 1;

693  variant enum_info = variant(variant::map_type());

705  if (reflect_version_1) {

707  compound_stack.back()->ref_element("enums").append_to_list(enum_info);

709  reflect_info->ref_element("enums").append_to_list(enum_info);
710  }
711  if (reflect_version_2) {
712  if (compound_stack.size() > 0)
713  compound_stack.back()->ref_element("elements").append_to_list(enum_info);
714  else
715  reflect_info->ref_element("elements").append_to_list(enum_info);
716  }
717  // generate debug info
718  if (debug_reflection)
719  std::cout << "reflect enum " << enum_info << std::endl;

721  }

730  idx = 2;

1069  void filter_out_empty_blocks(string& text)

1079  size_t n = l.length();
1080  if (l.substr(0, 2) == "IF")
1081  line_types[i] = LT_IF;
1082  else if (l.substr(0, 4) == "ELIF")
1083  line_types[i] = LT_ELIF;
1084  else if (l.substr(0, 4) == "ELSE")
1085  line_types[i] = LT_ELSE;
1086  else if (l.substr(0, 5) == "ENDIF")
1087  line_types[i] = LT_ENDIF;
1088  else if (l.substr(0, 7) == "INCLUDE")
1089  line_types[i] = LT_INCLUDE;
1090  else if (l.substr(0, 6) == "DEFINE")
1091  line_types[i] = LT_DEFINE;
1092  else
1093  line_types[i] = LT_OTHER;
1094  }
1095  j = i = 0;
1096  bool empty = true;
1097  unsigned d = 0;
1098  while (i < line_types.size()) {
1099  switch (line_types[i]) {
1100  case LT_IF:
1101  ++d;
1102  break;
1103  case LT_ELIF:
1104  case LT_ELSE:
1105  break;
1106  case LT_ENDIF:
1107  --d;
1108  break;
1109  case LT_INCLUDE:
1110  case LT_DEFINE:
1111  empty = false;
1112  break;
1113  case LT_OTHER:
1114  break;
1115  }
1116  if (d == 0) {
1117  if (!empty) {
1118  while (j <= i) {
1119  if (line_types[j] != LT_OTHER) {
1120  if (!new_text.empty())
1121  new_text += '\n';
1122  new_text += to_string(lines[j]);
1123  }
1124  ++j;
1125  }
1126  empty = true;
1127  }
1128  else
1129  j = i + 1;
1130  }
1131  ++i;
1132  }
1133  text = new_text;
1134  }

1866  // construct command token
1867  command_token ct;

1869  token tok = toker.bite();

1872  if (*tok.begin != special)
1873  ct = command_token(tok);

1876  // skip rest of current line
1877  while (!toker.empty()) {
1878  ++toker.begin;
1879  if (toker.begin[-1] == '\n')
1880  break;
1881  }
1882  continue;
1883  }
1884  // finally split off command token needs to be split off from next token
1885  else {
1886  // if special character repeats, interpret first as escape character
1887  if (toker.begin < toker.end && *toker.begin == special) {
1888  ++toker.begin;
1889  ct = command_token(tok);
1890  }
1891  else if (!ct.split_off_from(toker)) {
1892  error(ct.get_last_error(), ct.get_last_error_token());
1893  continue;
1894  }
1895  }
1896  // eliminate empty tokens before else and elif commands
1897  if (ct.remove_preceeding_empty_text_token() && !commands.empty() &&
1898  commands.back().is_empty())
1899  commands.pop_back();
