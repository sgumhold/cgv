747 unsigned int levenshtein_distance(const std::string& s1, const std::string& s2)
748 {
749  // create two work vectors of integer distances
750  std::vector<int> v0(s2.length() + 1);
751  std::vector<int> v1(s2.length() + 1);
752 
753  // initialize v0 (the previous row of distances)
754  // this row is A[0][i]: edit distance from an empty s to t;
755  // that distance is the number of characters to append to s to make t.
756  for (size_t i = 0; i <= s2.length(); ++i) {
757  v0[i] = int(i);
758  }
759 
760  for (size_t i = 0; i < s1.length(); ++i) {
761  // calculate v1 (current row distances) from the previous row v0
762 
763  // first element of v1 is A[i + 1][0]
764  // edit distance is delete (i + 1) chars from s to match empty t
765  v1[0] = int(i + 1);
766 
767  // use formula to fill in the rest of the row
768  for (size_t j = 0; j < s2.length(); ++j) {
769  // calculating costs for A[i + 1][j + 1]
770  int deletion_cost = v0[j + 1] + 1;
771  int insertion_cost = v1[j] + 1;
772  int substitution_cost = 0;
773  if (s1[i] == s2[j])
774  substitution_cost = v0[j];
775  else
776  substitution_cost = v0[j] + 1;
777 
778  v1[j + 1] = std::min(std::min(deletion_cost, insertion_cost), substitution_cost);
779  }
780 
781  // copy v1 (current row) to v0 (previous row) for next iteration
782  // since data in v1 is always invalidated, a swap without copy could be more efficient
783  std::swap(v0, v1);
784  }
785  // after the last swap, the results of v1 are now in v0
786  return v0[s2.length()];
787 }
