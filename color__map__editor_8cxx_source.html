75 bool color_map_editor::handle_event(cgv::gui::event& e) {
76 
77  // return true if the event gets handled and stopped here or false if you want to pass it to the next plugin
78  unsigned et = e.get_kind();
79  unsigned char modifiers = e.get_modifiers();
80 
81  if (et == cgv::gui::EID_KEY) {
82  cgv::gui::key_event& ke = (cgv::gui::key_event&)e;
83 
84  if (ke.get_action() == cgv::gui::KA_PRESS) {
85  switch (ke.get_key()) {
86  case cgv::gui::KEY_Left_Ctrl:
87  cursor_label = "+";
88  post_damage();
89  break;
90  case cgv::gui::KEY_Left_Alt:
91  cursor_label = "-";
92  post_damage();
93  break;
94  }
95  }
96  else if (ke.get_action() == cgv::gui::KA_RELEASE) {
97  switch (ke.get_key()) {
98  case cgv::gui::KEY_Left_Ctrl:
99  case cgv::gui::KEY_Left_Alt:
100  cursor_label = "";
101  post_damage();
102  break;
103  }
104  }
105  }
106  else if (et == cgv::gui::EID_MOUSE) {
107  cgv::gui::mouse_event& me = (cgv::gui::mouse_event&)e;
108  cgv::gui::MouseAction ma = me.get_action();
109 
110  switch (ma) {
111  case cgv::gui::MA_ENTER:
112  mouse_is_on_overlay = true;
113  return true;
114  case cgv::gui::MA_LEAVE:
115  mouse_is_on_overlay = false;
116  post_damage();
117  return true;
118  case cgv::gui::MA_MOVE:
119  case cgv::gui::MA_DRAG:
120  if(get_context())
121  cursor_position = ivec2(me.get_x(), get_context()->get_height() - 1 - me.get_y());
122  if(!cursor_label.empty())
123  post_damage();
124  break;
125  }
126 
127  bool request_clear_selection = false;
128 
129  if (me.get_button_state() & cgv::gui::MB_LEFT_BUTTON) {
130  if (ma == cgv::gui::MA_PRESS) {
131  ivec2 mpos = get_local_mouse_pos(ivec2(me.get_x(), me.get_y()));
132 
133  request_clear_selection = is_hit(mpos);
134 
135  switch (modifiers) {
136  case cgv::gui::EM_CTRL:
137  if(!get_hit_point(mpos))
138  add_point(mpos);
139  break;
140  case cgv::gui::EM_ALT:
141  {
142  cgv::g2d::draggable* hit_point = get_hit_point(mpos);
143  if(hit_point)
144  remove_point(hit_point);
145  }
146  break;
147  default: break;
148  }
149  }
150  }
151 
152  if(cmc.color_points.handle(e, get_viewport_size(), get_rectangle()))
153  return true;
154  if(cmc.opacity_points.handle(e, get_viewport_size(), get_rectangle()))
155  return true;
156 
157  if(request_clear_selection) {
158  cmc.color_points.clear_selected();
159  cmc.opacity_points.clear_selected();
160  handle_drag_end();
161  }
162  }
163  return false;
164 }

166 void color_map_editor::handle_member_change(const cgv::utils::pointer_test& m) {
167 
168  if(m.is(layout.total_height)) {
169  ivec2 size = get_rectangle().size;
170  size.y() = layout.total_height;
171  set_size(size);
172  }
173 
174  if(m.is(opacity_scale_exponent)) {
175  opacity_scale_exponent = cgv::math::clamp(opacity_scale_exponent, 1.0f, 5.0f);
176 
177  update_point_positions();
178  sort_points();
179  update_geometry();
180  }
181 
182  if(m.is(resolution)) {
183  cgv::render::context* ctx_ptr = get_context();
184  if(cmc.cm)
185  cmc.cm->set_resolution(resolution);
186  update_color_map(false);
187  }
188 
189  if(m.is(use_interpolation)) {
190  cgv::render::context* ctx_ptr = get_context();
191  if(cmc.cm)
192  cmc.cm->enable_interpolation(use_interpolation);
193  update_color_map(false);
194  }
195 
196  if(m.is(use_linear_filtering)) {
197  cgv::render::context* ctx_ptr = get_context();
198  if(cmc.cm) {
199  cgv::render::gl_color_map* gl_cm = cmc.get_gl_color_map();
200  if(gl_cm)
201  gl_cm->enable_linear_filtering(use_linear_filtering);
202  }
203  update_color_map(false);
204  }
205 
206  for(unsigned i = 0; i < cmc.color_points.size(); ++i) {
207  if(m.is(cmc.color_points[i].col)) {
208  update_color_map(true);
209  break;
210  }
211  }
212 
213  for(unsigned i = 0; i < cmc.opacity_points.size(); ++i) {
214  if(m.is(cmc.opacity_points[i].val[1])) {
215  cmc.opacity_points[i].update_pos(layout, opacity_scale_exponent);
216  update_color_map(true);
217  break;
218  }
219  }
220 
221  if(m.is(supports_opacity)) {
222  layout.total_height = supports_opacity ? 200 : 60;
223  on_set(&layout.total_height);
224 
225  if(supports_opacity) {
226  layout.total_height = 200;
227  } else {
228  layout.total_height = 60;
229  if(cmc.cm) {
230  cmc.cm->clear_opacity_points();
231  cmc.opacity_points.clear();
232 
233  update_point_positions();
234  update_color_map(false);
235  }
236  }
237 
238  post_recreate_layout();
239  post_recreate_gui();
240  }

477 void color_map_editor::set_color_map(cgv::render::color_map* cm) {
478  cmc.reset();
479  cmc.cm = cm;
480 
481  if(cmc.cm) {
482  auto& cm = *cmc.cm;
483  auto& cp = cmc.cm->ref_color_points();
484 
485  for(size_t i = 0; i < cp.size(); ++i) {
486  color_point p;
487  p.val = cgv::math::clamp(cp[i].first, 0.0f, 1.0f);
488  p.col = cp[i].second;
489  cmc.color_points.add(p);
490  }
491 
492  if(supports_opacity) {
493  auto& op = cmc.cm->ref_opacity_points();
494  for(size_t i = 0; i < op.size(); ++i) {
495  opacity_point p;
496  p.val.x() = cgv::math::clamp(op[i].first, 0.0f, 1.0f);
497  p.val.y() = cgv::math::clamp(op[i].second, 0.0f, 1.0f);
498  cmc.opacity_points.add(p);
499  }
500  }
501 
502  use_interpolation = cm.is_interpolation_enabled();
503  use_linear_filtering = true;
504 
505  cgv::render::gl_color_map* gl_cm = cmc.get_gl_color_map();
506  if(gl_cm)
507  use_linear_filtering = gl_cm->is_linear_filtering_enabled();
508 
509  update_point_positions();
510  update_color_map(false);
511 
512  post_recreate_gui();
513  }

516 void color_map_editor::set_histogram_data(const std::vector<unsigned> data) {
517  histogram = data;
518 
519  std::vector<float> float_data(histogram.size(), 0.0f);
520  hist_max = 1;
521  hist_max_non_zero = 1;
522  for(size_t i = 0; i < histogram.size(); ++i) {
523  unsigned count = histogram[i];
524  hist_max = std::max(hist_max, count);
525  if(i > 0)
526  hist_max_non_zero = std::max(hist_max_non_zero, count);
527  float_data[i] = static_cast<float>(count);
528  }
529 
530  if(auto ctx_ptr = get_context()) {
531  auto& ctx = *ctx_ptr;
532  hist_tex.destruct(ctx);
533 
534  cgv::data::data_view dv = cgv::data::data_view(new cgv::data::data_format(unsigned(histogram.size()), TI_FLT32, cgv::data::CF_R), float_data.data());
535  hist_tex = cgv::render::texture("flt32[R]");
536  hist_tex.create(ctx, dv, 0);

538  post_damage();

913 void color_map_editor::update_color_map(bool is_data_change) {
914 
915  cgv::render::context* ctx_ptr = get_context();
916  if(!ctx_ptr || !cmc.cm) return;
917  cgv::render::context& ctx = *ctx_ptr;
918 
919  auto& cm = *cmc.cm;
920  auto& color_points = cmc.color_points;
921  auto& opacity_points = cmc.opacity_points;
922 
923  sort_points();
924 
925  cm.clear();
926 
927  for(unsigned i = 0; i < color_points.size(); ++i) {
928  const color_point& p = color_points[i];
929  cm.add_color_point(p.val, p.col);
930  }
931 
932  if(supports_opacity) {
933  for(unsigned i = 0; i < opacity_points.size(); ++i) {
934  const opacity_point& p = opacity_points[i];
935  cm.add_opacity_point(p.val.x(), p.val.y());
936  }
937  } else {
938  // add one fully opaque point that will be removed later on
939  cm.add_opacity_point(0.0f, 1.0f);
940  }
941 
942  size_t size = static_cast<size_t>(resolution);
943  std::vector<rgba> cs_data = cm.interpolate(size);
944 
945  std::vector<uint8_t> data(4 * 2 * size);
946  for(size_t i = 0; i < size; ++i) {
947  rgba col = cs_data[i];
948  uint8_t r = static_cast<uint8_t>(255.0f * col.R());
949  uint8_t g = static_cast<uint8_t>(255.0f * col.G());
950  uint8_t b = static_cast<uint8_t>(255.0f * col.B());
951  uint8_t a = static_cast<uint8_t>(255.0f * col.alpha());
952 
953  size_t idx = 4 * i;
954  data[idx + 0] = r;
955  data[idx + 1] = g;
956  data[idx + 2] = b;
957  data[idx + 3] = a;
958  idx += 4*size;
959  data[idx + 0] = r;
960  data[idx + 1] = g;
961  data[idx + 2] = b;
962  data[idx + 3] = a;
963  }
964 
965  setup_preview_texture(ctx);
966  cgv::data::data_view dv = cgv::data::data_view(new cgv::data::data_format(unsigned(size), 2, TI_UINT8, cgv::data::CF_RGBA), data.data());
967  preview_tex.create(ctx, dv, 0);
968 
969  if(!supports_opacity)
970  cm.clear_opacity_points();
971 
972  update_geometry();
973 
974  if(on_change_callback)
975  on_change_callback();
976 
977  post_damage();
978 }

980 bool color_map_editor::update_geometry() {
981 
982  cgv::render::context* ctx_ptr = get_context();
983  if(!ctx_ptr || !cmc.cm) return false;
984  cgv::render::context& ctx = *ctx_ptr;
985 
986  auto& cm = *cmc.cm;
987  auto& color_points = cmc.color_points;
988  auto& opacity_points = cmc.opacity_points;
989  auto& color_handles = cmc.color_handles;
990  auto& opacity_handles = cmc.opacity_handles;
991  auto& lines = cmc.lines;
992  auto& triangles = cmc.triangles;
993 
994  color_handles.clear();
995  opacity_handles.clear();
996  lines.clear();
997  triangles.clear();
998 
999  bool success = color_points.size() > 0 && opacity_points.size() > 0;
1000 
1001  // create color handles
1002  vec2 pos_offset = vec2(0.0f, 0.5f * color_point::default_height);
1003 
1004  for(unsigned i = 0; i < color_points.size(); ++i) {
1005  const auto& p = color_points[i];
1006  vec2 pos = p.center();
1007  rgba col = color_points.get_selected() == &p ? highlight_color : handle_color;
1008  color_handles.add(pos - pos_offset, col);
1009  color_handles.add(pos + pos_offset, col);
1010  }
1011 
1012  // create opacity handles
1013  for(unsigned i = 0; i < opacity_points.size(); ++i) {
1014  const auto& p = opacity_points[i];
1015  vec2 pos = p.center();
1016  rgba col = opacity_points.get_selected() == &p ? highlight_color : handle_color;
1017  opacity_handles.add(pos, col);
1018  }
1019 
1020  if(opacity_points.size() > 0) {
1021  const auto& pl = opacity_points[0];
1022 
1023  vec2 tex_coord(0.0f, 0.5f);
1024 
1025  lines.add(vec2(float(layout.opacity_editor_rect.x()), pl.center().y()), tex_coord);
1026 
1027  triangles.add(vec2(float(layout.opacity_editor_rect.x()), pl.center().y()), tex_coord);
1028  triangles.add(layout.opacity_editor_rect.position, tex_coord);
1029 
1030  for(unsigned i = 0; i < opacity_points.size(); ++i) {
1031  const auto& p = opacity_points[i];
1032  vec2 pos = p.center();
1033 
1034  tex_coord.x() = p.val.x();
1035 
1036  lines.add(pos, tex_coord);
1037 
1038  triangles.add(pos, tex_coord);
1039  triangles.add(vec2(pos.x(), (float)layout.opacity_editor_rect.y()), tex_coord);
1040  }
1041 
1042  const auto& pr = opacity_points[opacity_points.size() - 1];
1043  vec2 max_pos = layout.opacity_editor_rect.position + vec2(1.0f, 0.0f) * layout.opacity_editor_rect.size;
1044 
1045  tex_coord.x() = 1.0f;
1046 
1047  lines.add(vec2(max_pos.x(), pr.center().y()), tex_coord);
1048 
1049  triangles.add(vec2(max_pos.x(), pr.center().y()), tex_coord);
1050  triangles.add(max_pos, tex_coord);
1051  } else {
1052  success = false;
1053  }
1054 
1055  return success;
1056 }
