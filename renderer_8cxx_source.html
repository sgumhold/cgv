23 void renderer::manage_singleton(context& ctx, const std::string& renderer_name, int& ref_count, int ref_count_change)
24 {
25 switch (ref_count_change) {
26 case 1:
27 if (ref_count == 0) {
28 if (!init(ctx))
29 ctx.error(std::string("unable to initialize ") + renderer_name + " singleton");
30 }
31 ++ref_count;
32 break;
33 case 0:
34 break;
35 case -1:
36 if (ref_count == 0)
37 ctx.error(std::string("attempt to decrease reference count of ") + renderer_name + " singleton below 0");
38 else {
39 if (--ref_count == 0)
40 clear(ctx);
41 }
42 break;
43 default:
44 ctx.error(std::string("invalid change reference count outside {-1,0,1} for ") + renderer_name + " singleton");
45 }
46 }

205 bool renderer::enable(context& ctx)
206 {
207 if(prog_ptr == &prog) {
208 update_defines(defines);
209 if(defines != last_defines) {
210 if(prog.is_created())
211 prog.destruct(ctx);
212 shader_compile_options options;
213 options.defines = defines;
214 if(!build_shader_program(ctx, prog, options))
215 return false;
216#ifndef _DEBUG
217 }
218#else
219 if(current_prog_render_count < 10)
220 std::cerr << "Performance warning: shader is rebuild! Consider using multiple instances of the renderer." << std::endl;
221 current_prog_render_count = 0;
222 }
223 ++current_prog_render_count;
224#endif
225 last_defines = defines;
226 }
227 bool res = ref_prog().enable(ctx);
228 if (aam_ptr)
229 res = aam_ptr->enable(ctx);
230 return res;

233 bool renderer::disable(context& ctx)
234 {
235 bool res = true;
236 if (has_aam())
237 res = aam_ptr->disable(ctx);
238 else {
239 if (ctx.core_profile) {
240 default_aam.disable(ctx);
241 for (int loc : enabled_attribute_arrays)
242 res = default_aam.aab.disable_array(ctx, loc) && res;
243 }
244 else {
245 for (int loc : enabled_attribute_arrays)
246 res = attribute_array_binding::disable_global_array(ctx, loc) && res;
247 }
248 enabled_attribute_arrays.clear();
249 has_colors = false;
250 has_positions = false;
251 index_count = 0;
252 }
253 res = ref_prog().disable(ctx) && res;
254 prog_ptr = &prog;
255 return res;

257 void renderer::draw_impl(context& ctx, PrimitiveType type, size_t start, size_t count, bool use_strips, bool use_adjacency, uint32_t strip_restart_index)
258 {
259 if (use_strips && has_indices()) {
260 glPrimitiveRestartIndex(strip_restart_index);
261 glEnable(GL_PRIMITIVE_RESTART);
262 }
263 if (type == PT_LINES) {
264 if (use_adjacency)
265 if (use_strips)
266 type = PT_LINE_STRIP_ADJACENCY;
267 else
268 type = PT_LINES_ADJACENCY;
269 else
270 if (use_strips)
271 type = PT_LINE_STRIP;
272 }
273 else if (type == PT_TRIANGLES)
274 if (use_adjacency)
275 if (use_strips)
276 type = PT_TRIANGLE_STRIP_ADJACENCY;
277 else
278 type = PT_TRIANGLES_ADJACENCY;
279 else
280 if (use_strips)
281 type = PT_TRIANGLE_STRIP;
282
283 GLenum pt = gl::map_to_gl(type);
284 if (has_indices()) {
285 bool aam_has_index_buffer = aam_ptr && aam_ptr->has_index_buffer();
286 bool use_index_buffer = index_buffer_ptr && !aam_has_index_buffer;
287 bool use_indices = !index_buffer_ptr && !aam_has_index_buffer;
288 if (use_index_buffer)
289 index_buffer_ptr->bind(ctx, VBT_INDICES);
290 const void* offset = reinterpret_cast<const uint8_t*>(use_indices ? indices : 0)
291 + start * cgv::type::info::get_type_size(index_type);
292 glDrawElements(pt, (GLsizei)count, gl::map_to_gl(index_type), offset);
293 if (use_index_buffer)
294 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
295 }
296 else
297 glDrawArrays(pt, (GLint)start, (GLsizei)count);
298 }
