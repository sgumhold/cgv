23  void renderer::manage_singleton(context& ctx, const std::string& renderer_name, int& ref_count, int ref_count_change)
24  {
25  switch (ref_count_change) {
26  case 1:
27  if (ref_count == 0) {
28  if (!init(ctx))
29  ctx.error(std::string("unable to initialize ") + renderer_name + " singleton");
30  }
31  ++ref_count;
32  break;
33  case 0:
34  break;
35  case -1:
36  if (ref_count == 0)
37  ctx.error(std::string("attempt to decrease reference count of ") + renderer_name + " singleton below 0");
38  else {
39  if (--ref_count == 0)
40  clear(ctx);
41  }
42  break;
43  default:
44  ctx.error(std::string("invalid change reference count outside {-1,0,1} for ") + renderer_name + " singleton");
45  }
46  }

191  bool renderer::enable(context& ctx)
192  {
193  if(prog_ptr == &prog) {
194  update_defines(defines);
195  if(defines != last_defines) {
196  if(prog.is_created())
197  prog.destruct(ctx);
198  if(!build_shader_program(ctx, prog, defines))
199  return false;
200 #ifndef _DEBUG
201  }
202 #else
203  if(current_prog_render_count < 10)
204  std::cerr << "Performance warning: shader is rebuild! Consider using multiple instances of the renderer." << std::endl;
205  current_prog_render_count = 0;
206  }
207  ++current_prog_render_count;
208 #endif
209  last_defines = defines;
210  }
211  bool res = ref_prog().enable(ctx);
212  if (aam_ptr)

217  bool renderer::disable(context& ctx)
218  {
219  bool res = true;
220  if (has_aam())
221  res = aam_ptr->disable(ctx);
222  else {
223  if (ctx.core_profile) {
224  default_aam.disable(ctx);
225  for (int loc : enabled_attribute_arrays)
226  res = default_aam.aab.disable_array(ctx, loc) && res;
227  }
228  else {
229  for (int loc : enabled_attribute_arrays)
230  res = attribute_array_binding::disable_global_array(ctx, loc) && res;
231  }
232  enabled_attribute_arrays.clear();
233  has_colors = false;
234  has_positions = false;
235  index_count = 0;
236  }
237  res = ref_prog().disable(ctx) && res;
238  prog_ptr = &prog;
239  return res;

241  void renderer::draw_impl(context& ctx, PrimitiveType type, size_t start, size_t count, bool use_strips, bool use_adjacency, uint32_t strip_restart_index)
242  {
243  if (use_strips && has_indices()) {
244  glPrimitiveRestartIndex(strip_restart_index);
245  glEnable(GL_PRIMITIVE_RESTART);
246  }
247  if (type == PT_LINES) {
248  if (use_adjacency)
249  if (use_strips)
250  type = PT_LINE_STRIP_ADJACENCY;
251  else
252  type = PT_LINES_ADJACENCY;
253  else
254  if (use_strips)
255  type = PT_LINE_STRIP;
256  }
257  else if (type == PT_TRIANGLES)
258  if (use_adjacency)
259  if (use_strips)
260  type = PT_TRIANGLE_STRIP_ADJACENCY;
261  else
262  type = PT_TRIANGLES_ADJACENCY;
263  else
264  if (use_strips)
265  type = PT_TRIANGLE_STRIP;
266 
267  GLenum pt = gl::map_to_gl(type);
268  if (has_indices()) {
269  bool aam_has_index_buffer = aam_ptr && aam_ptr->has_index_buffer();
270  bool use_index_buffer = index_buffer_ptr && !aam_has_index_buffer;
271  bool use_indices = !index_buffer_ptr && !aam_has_index_buffer;
272  if (use_index_buffer)
273  index_buffer_ptr->bind(ctx, VBT_INDICES);
274  const void* offset = reinterpret_cast<const uint8_t*>(use_indices ? indices : 0)
275  + start * cgv::type::info::get_type_size(index_type);
276  glDrawElements(pt, (GLsizei)count, gl::map_to_gl(index_type), offset);
277  if (use_index_buffer)
278  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
279  }
280  else
281  glDrawArrays(pt, (GLint)start, (GLsizei)count);
282  }
