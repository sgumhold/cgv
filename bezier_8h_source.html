134 std::pair<T, T> compute_bounds(const T& p0, const T& p1, const T& p2, const T& p3) const {
135 T mi = std::min(p0, p3);
136 T ma = std::max(p0, p3);
137
138 T c = T(-1) * p0 + T(1) * p1;
139 T b = T(1) * p0 - T(2) * p1 + T(1) * p2;
140 T a = T(-1) * p0 + T(3) * p1 - T(3) * p2 + T(1) * p3;
141
142 T h = b * b - a * c;
143
144 if(h > T(0)) {
145 h = sqrt(h);
146 T t = (-b - h) / a;
147 if(t > T(0) && t < T(1)) {
148 T s = T(1) - t;
149 T q = s * s * s * p0 + T(3) * s * s * t * p1 + T(3) * s * t * t * p2 + t * t * t * p3;
150 mi = std::min(mi, q);
151 ma = std::max(ma, q);
152 }
153 t = (-b + h) / a;
154 if(t > T(0) && t < T(1)) {
155 T s = T(1) - t;
156 T q = s * s * s * p0 + T(3) * s * s * t * p1 + T(3) * s * t * t * p2 + t * t * t * p3;
157 mi = std::min(mi, q);
158 ma = std::max(ma, q);
159 }
160 }
161
162 return { mi, ma };
163 }
