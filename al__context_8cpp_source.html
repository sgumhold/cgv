45 void OALContext::load_sample(std::string filepath)
46 {
47  std::filesystem::path path(filepath);
48  if (path.empty() || !std::filesystem::exists(path))
49  return;
50 
51  SndfileHandle soundfile(path.string().c_str());
52  if (soundfile.error() != SF_ERR_NO_ERROR) {
53  std::cerr << soundfile.strError() << std::endl;
54  return;
55  }
56 
57  assert(oal_context);
58  assert(oal_device);
59 
60  ALuint buffer;
61  alGenBuffers(1, &buffer);
62  assert(AL_NO_ERROR == alcGetError(oal_device));
63 
64  assert(soundfile.channels() == 1 || soundfile.channels() == 2);
65  const size_t num_bytes{soundfile.frames() * soundfile.channels() * sizeof(std::int16_t)};
66  const auto buf = std::make_unique<std::int16_t[]>(num_bytes);
67  // libsndfile documentation notes that reading from float files into int buffers could require scaling
68  soundfile.command(SFC_SET_SCALE_FLOAT_INT_READ, nullptr, SF_TRUE);
69  soundfile.readf(buf.get(), soundfile.frames());
70  if (soundfile.error() != SF_ERR_NO_ERROR) {
71  std::cerr << soundfile.strError() << std::endl;
72  return;
73  }
74 
75  alBufferData(buffer, soundfile.channels() == 2 ? AL_FORMAT_STEREO16 : AL_FORMAT_MONO16, buf.get(), (ALsizei)num_bytes,
76  (ALsizei)soundfile.samplerate());
77  assert(AL_NO_ERROR == alcGetError(oal_device));
78 
79  assert(path.has_filename());
80  sample_buffers.emplace(path.stem().string(), buffer);
81 }

83 void OALContext::load_sample(std::string symbolic_name, const char* data, size_t data_length)
84 {
85  if (symbolic_name.empty())
86  throw std::invalid_argument("Symbolic file name can not be empty!");
87  if (std::end(sample_buffers) != sample_buffers.find(symbolic_name))
88  throw std::invalid_argument("Symbolic name \"" + symbolic_name + "\" already present");
89  assert(data);
90 
91  // The state describing a virtual file descriptor (as anonymous type)
92  struct VIO_state
93  {
94  const char* const data;
95  size_t data_length;
96  const char* current_byte_ptr;
97  } vio_state{data, data_length, data}; // and a concrete variable with list initialization performed
98 
99  SF_VIRTUAL_IO sfvio;
100  sfvio.get_filelen = [](void* user_data) -> sf_count_t {
101  return static_cast<const VIO_state*>(user_data)->data_length;
102  };
103  sfvio.seek = [](sf_count_t offset, int whence, void* user_data) -> sf_count_t {
104  auto* state = static_cast<VIO_state*>(user_data);
105  switch (whence) {
106  case SEEK_CUR:
107  state->current_byte_ptr += offset;
108  return state->current_byte_ptr - state->data;
109  case SEEK_SET:
110  state->current_byte_ptr = state->data + offset;
111  return offset;
112  case SEEK_END:
113  state->current_byte_ptr = (state->data + state->data_length - 1) - offset;
114  return state->current_byte_ptr - state->data;
115  default:
116  return 0;
117  }
118  };
119  sfvio.read = [](void* ptr, sf_count_t count, void* user_data) -> sf_count_t {
120  auto* state = static_cast<VIO_state*>(user_data);
121  memcpy(ptr, state->current_byte_ptr, count);
122  state->current_byte_ptr += count;
123  return count;
124  };
125  sfvio.write = [](const void* ptr, sf_count_t count, void* user_data) -> sf_count_t { return 0; };
126  sfvio.tell = [](void* user_data) -> sf_count_t {
127  auto* state = static_cast<const VIO_state*>(user_data);
128  return state->current_byte_ptr - state->data;
129  };
130 
131  SndfileHandle soundfile(sfvio, &vio_state);
132  if (soundfile.error() != SF_ERR_NO_ERROR) {
133  std::cerr << soundfile.strError() << std::endl;
134  return;
135  }
136 
137  assert(oal_context);
138  assert(oal_device);
139 
140  ALuint buffer;
141  alGenBuffers(1, &buffer);
142  assert(AL_NO_ERROR == alcGetError(oal_device));
143 
144  assert(soundfile.channels() == 1 || soundfile.channels() == 2);
145  const size_t num_bytes{soundfile.frames() * soundfile.channels() * sizeof(std::int16_t)};
146  const auto buf = std::make_unique<std::int16_t[]>(num_bytes);
147  const std::vector<std::int16_t> data_buffer(num_bytes, 0);
148  // libsndfile documentation notes that reading from float files into int buffers could require scaling
149  soundfile.command(SFC_SET_SCALE_FLOAT_INT_READ, nullptr, SF_TRUE);
150  soundfile.readf(buf.get(), soundfile.frames());
151  if (soundfile.error() != SF_ERR_NO_ERROR) {
152  std::cerr << soundfile.strError() << std::endl;
153  return;
154  }

163 void OALContext::load_samples(std::string folder, bool recursive)
164 {
165  std::filesystem::path f(folder);
166  if (!std::filesystem::is_directory(f)) {
167  std::cerr << __FUNCTION__ << ": no valid folder '" << folder << "'";
168  return;
169  }
170 
171  // TODO code duplication should be removed via static tag dispatching
172  if (recursive) {
173  for (auto const& entry : std::filesystem::recursive_directory_iterator(f)) {
174  if (!entry.is_regular_file())
175  continue;
176  load_sample(static_cast<std::filesystem::path>(entry).string());
177  }
178  }
179  else {
180  for (auto const& entry : std::filesystem::directory_iterator(f)) {
181  if (!entry.is_regular_file())
182  continue;
183  load_sample(static_cast<std::filesystem::path>(entry).string());
184  }

194 std::list<std::string> OALContext::enumerate_devices()
195 {
196  assert(alcIsExtensionPresent(nullptr, "ALC_ENUMERATE_ALL_EXT"));
197  const ALCchar* dev_strings = alcGetString(nullptr, ALC_ALL_DEVICES_SPECIFIER);
198  assert(dev_strings);
199 
200  std::list<std::string> device_names;
201  // alcGetString returns cstring array terminated by two successive null bytes
202  // (the first terminating the last cstring, the second terminating the array)
203  while (strlen(dev_strings) != 0) {
204  device_names.emplace_back(dev_strings);
205  // +1 for the null byte
206  dev_strings += device_names.back().length() + 1;
207  }
208  return device_names;
209 }
