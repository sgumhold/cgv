74 ref_prog().set_uniform(ctx, "has_secondary_colors", has_secondary_colors);
75 const auto& brs = get_style<box_render_style>();
76 if (ref_prog().get_uniform_location(ctx, "default_radius") != -1)
77 ref_prog().set_uniform(ctx, "default_radius", brs.default_radius);
78 if (ref_prog().get_uniform_location(ctx, "relative_anchor") != -1)
79 ref_prog().set_uniform(ctx, "relative_anchor", brs.relative_anchor);
80 if (!has_extents)
81 ref_prog().set_attribute(ctx, "extent", brs.default_extent);
82 return true;
83 }
