537 template <typename T, cgv::type::uint32_type N>
538 const fvec<T, N> slerp(const fvec<T, N> &v0, const fvec<T, N> &v1, T t) {
539  T dotv0v1 = dot(v0, v1);
540  // clamp between [-1,1]
541  if (dotv0v1 < -1)
542  dotv0v1 = -1;
543 
544  if (dotv0v1 > 1)
545  dotv0v1 = 1;
546 
547  T theta = acos(dotv0v1) * t;
548  auto v2 = normalize(v1 - (dotv0v1)*v0);
549  return T(cos(theta)) * v0 + T(sin(theta)) * v2;
550 }
