510template <typename T, cgv::type::uint32_type N>
511const fvec<T, N> slerp(const fvec<T, N>& v0, const fvec<T, N>& v1, T t) {
512 T dotv0v1 = dot(v0, v1);
513 // clamp between [-1,1]
514 if(dotv0v1 < -1)
515 dotv0v1 = -1;
516
517 if(dotv0v1 > 1)
518 dotv0v1 = 1;
519
520 T theta = std::acos(dotv0v1) * t;
521 auto v2 = normalize(v1 - (dotv0v1)*v0);
522 return T(std::cos(theta)) * v0 + T(std::sin(theta)) * v2;
523}
