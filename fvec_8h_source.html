541template <typename T, cgv::type::uint32_type N>
542const fvec<T, N> slerp(const fvec<T, N> &v0, const fvec<T, N> &v1, T t) {
543 T dotv0v1 = dot(v0, v1);
544 // clamp between [-1,1]
545 if (dotv0v1 < -1)
546 dotv0v1 = -1;
547
548 if (dotv0v1 > 1)
549 dotv0v1 = 1;
550
551 T theta = acos(dotv0v1) * t;
552 auto v2 = normalize(v1 - (dotv0v1)*v0);
553 return T(cos(theta)) * v0 + T(sin(theta)) * v2;
554}
