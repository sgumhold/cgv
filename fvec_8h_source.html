504template <typename T, cgv::type::uint32_type N>
505const fvec<T, N> slerp(const fvec<T, N>& v0, const fvec<T, N>& v1, T t) {
506 T dotv0v1 = dot(v0, v1);
507 // clamp between [-1,1]
508 if(dotv0v1 < -1)
509 dotv0v1 = -1;
510
511 if(dotv0v1 > 1)
512 dotv0v1 = 1;
513
514 T theta = std::acos(dotv0v1) * t;
515 auto v2 = normalize(v1 - (dotv0v1)*v0);
516 return T(std::cos(theta)) * v0 + T(std::sin(theta)) * v2;
517}
