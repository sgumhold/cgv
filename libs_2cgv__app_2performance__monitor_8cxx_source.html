267 void performance_monitor::update_plot() {
268 
269  ivec2 plot_size = layout.plot_rect.size;
270 
271  float a = static_cast<float>(1000.0 * monitor.delta_time / 33.333333333);
272  float b = std::min(a, 1.0f);
273  float bar_height = plot_size.y() * b;
274  bar_height = std::max(bar_height, 1.0f);
275 
276  rgb bar_color = a > 1.0f ? rgb(0.7f, 0.0f, 0.0f) : plot_color_map.interpolate_color(b);
277 
278  if(bars.render_count() < plot_size.x()) {
279  for(auto& position : bars.position)
280  position.x() -= 1.0f;
281 
282  int x = layout.plot_rect.x1() - 1;
283  x = std::max(x, 0);
284  float bar_x = static_cast<float>(x);
285  bars.add(vec2(bar_x, static_cast<float>(layout.plot_rect.y())), vec2(1.0f, bar_height), bar_color);
286 
287  } else {
288  for(size_t i = 0; i < bars.position.size() - 1; ++i) {
289  bars.size[i].y() = bars.size[i + 1].y();
290  bars.color[i] = bars.color[i + 1];
291  }
292 
293  bars.size.back() = vec2(1.0f, bar_height);
294  bars.color.back() = bar_color;
295  }
296 
297  bars.set_out_of_date();
298  post_damage();
299 }
