140 void performance_monitor::after_finish(cgv::render::context& ctx) {
141 
142  bool enabled = monitor.enabled;
143  if(monitor.enabled_only_when_visible && !show)
144  enabled = false;
145 
146  if(enabled) {
147  ++monitor.total_frame_count;
148  ++monitor.interval_frame_count;
149 
150  double seconds_since_start = monitor.timer.get_elapsed_time();
151  monitor.delta_time = seconds_since_start - monitor.last_seconds_since_start;
152 
153  monitor.running_time += monitor.delta_time;
154 
155  monitor.last_seconds_since_start = seconds_since_start;
156 
157  if(monitor.running_time >= monitor.interval) {
158  monitor.avg_fps = (double)monitor.interval_frame_count / monitor.running_time;
159  monitor.running_time = 0.0;
160  monitor.interval_frame_count = 0u;
161  }
162  }
163 }

310 void performance_monitor::update_plot() {
311 
312  ivec2 plot_size = layout.plot_rect.size;
313 
314  float a = static_cast<float>(1000.0 * monitor.delta_time / 33.333333333);
315  float b = std::min(a, 1.0f);
316  float bar_height = plot_size.y() * b;
317  bar_height = std::max(bar_height, 1.0f);
318 
319  rgb bar_color = a > 1.0f ? rgb(0.7f, 0.0f, 0.0f) : plot_color_map.interpolate_color(b);
320 
321  if(bars.render_count() < plot_size.x()) {
322  for(auto& position : bars.position)
323  position.x() -= 1.0f;
324 
325  int x = layout.plot_rect.x1() - 1;
326  x = std::max(x, 0);
327  float bar_x = static_cast<float>(x);
328  bars.add(vec2(bar_x, static_cast<float>(layout.plot_rect.y())), vec2(1.0f, bar_height), bar_color);
329 
330  } else {
331  for(size_t i = 0; i < bars.position.size() - 1; ++i) {
332  bars.size[i].y() = bars.size[i + 1].y();
333  bars.color[i] = bars.color[i + 1];
334  }
335 
336  bars.size.back() = vec2(1.0f, bar_height);
337  bars.color.back() = bar_color;
338  }
339 
340  bars.set_out_of_date();
341  post_damage();
342 }
