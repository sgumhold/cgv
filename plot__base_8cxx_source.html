335 void plot_base::prepare_extents()
336 {
337  extent = vec3(0.0f);
338  unsigned max_ai = std::min(unsigned(get_domain_config_ptr()->axis_configs.size()), get_dim());
339  unsigned F_count = 0;
340  float F;
341  for (unsigned ai = 0; ai < max_ai; ++ai) {
342  const auto& ac = get_domain_config_ptr()->axis_configs[ai];
343  extent[ai] = ac.extent;
344  if (ac.extent_scaling == 0.0f)
345  continue;
346  // (E1 * ES1) / A1 = (E2 * ES2) / A2 = F
347  // F = E * ES / A;
348  // E = F * A / ES;
349  float f = ac.extent * ac.extent_scaling / ac.get_attribute_extent();
350  if (F_count)
351  F = std::min(F, f);
352  else
353  F = f;
354  ++F_count;
355  }
356  if (F_count > 1) {
357  for (unsigned ai = 0; ai < max_ai; ++ai) {
358  const auto& ac = get_domain_config_ptr()->axis_configs[ai];
359  if (ac.extent_scaling == 0.0f)
360  continue;
361  extent[ai] = F * ac.get_attribute_extent() / ac.extent_scaling;
362  }
363  }
364 }

449 size_t plot_base::enable_attributes(cgv::render::context& ctx, int i, const sample_access& sa)
450 {
451  auto& asa = attribute_source_arrays[i];
452  auto& ass = asa.attribute_sources;
453  // first check whether we need to update vbo
454  unsigned ai = 0, j;
455  bool update_vbo = false;
456  size_t vbo_nr_floats = 0;
457  for (ai = 0; ai < ass.size(); ++ai) {
458  auto& as = ass[ai];
459  switch (as.source) {
460  case AS_SAMPLE_CONTAINER:
461  j = as.sub_plot_index == -1 ? i : as.sub_plot_index;
462  if (attribute_source_arrays[j].samples_out_of_date) {
463  update_vbo = true;
464  as.count = sa.size(j);
465  vbo_nr_floats += as.count;
466  }
467  break;
468  case AS_POINTER:
469  if (asa.samples_out_of_date) {
470  update_vbo = true;
471  vbo_nr_floats += as.count;
472  }
473  break;
474  default:
475  break;
476  }
477  }
478  // update vbo if necessary
479  if (update_vbo) {
480  // copy data into CPU buffer
481  std::vector<float> buffer(vbo_nr_floats);
482  float* dest = buffer.data();
483  for (ai = 0; ai < ass.size(); ++ai) {
484  const auto& as = ass[ai];
485  switch (as.source) {
486  case AS_SAMPLE_CONTAINER:
487  j = as.sub_plot_index == -1 ? i : as.sub_plot_index;
488  if (attribute_source_arrays[j].samples_out_of_date) {
489  for (int k = 0; k < as.count; ++k)
490  *dest++ = sa(j,k,(unsigned)as.offset);
491  }
492  break;
493  case AS_POINTER:
494  if (asa.samples_out_of_date) {
495  const float* src = as.pointer;
496  unsigned stride = unsigned(as.stride/sizeof(float));
497  for (int k = 0; k < as.count; ++k, src += stride)
498  *dest++ = *src;
499  }
500  break;
501  default:
502  break;
503  }
504  }
505  if (asa.vbo.get_size_in_bytes() != vbo_nr_floats * sizeof(float)) {
506  asa.vbo.destruct(ctx);
507  asa.vbo.create(ctx, buffer);
508  }
509  else
510  asa.vbo.replace(ctx, 0, buffer.data(), buffer.size());
511  asa.sources_out_of_date = true;
512  }
513  // update attribute bindings
514  if (!asa.aab.is_created()) {
515  asa.aab.create(ctx);
516  asa.sources_out_of_date = true;
517  }
518  if (asa.sources_out_of_date) {
519  size_t count = -1;
520  size_t vbo_offset = 0;
521  float f;
522  for (ai = 0; ai < ass.size(); ++ai) {
523  const auto& as = ass[ai];
524  switch (as.source) {
525  case AS_SAMPLE_CONTAINER: {
526  j = as.sub_plot_index == -1 ? i : as.sub_plot_index;
527  asa.aab.set_attribute_array(ctx, ai, cgv::render::get_element_type(f), asa.vbo, vbo_offset, sa.size(j), 0);
528  count = std::min(sa.size(j), count);
529  vbo_offset += count * sizeof(float);
530  break;
531  }
532  case AS_POINTER:
533  asa.aab.set_attribute_array(ctx, ai, cgv::render::get_element_type(f), asa.vbo, vbo_offset, as.count, 0);
534  count = std::min(as.count, count);
535  vbo_offset += count * sizeof(float);
536  break;
537  case AS_VBO:
538  asa.aab.set_attribute_array(ctx, ai, cgv::render::get_element_type(f), *as.vbo_ptr, as.count, as.offset, (unsigned)as.stride);
539  count = std::min(as.count, count);
540  break;
541  }
542  }
543  asa.count = count;
544  asa.sources_out_of_date = false;
545  }
546  // enable and return count
547  asa.aab.enable(ctx);
548  return asa.count;

614 bool plot_base::extract_tick_rectangles_and_tick_labels(
615  std::vector<box2>& R, std::vector<rgb>& C, std::vector<float>& D,
616  std::vector<label_info>& tick_labels, int ai, int ci, int ti, float he,
617  float z_plot, float plot_scale, vec2 plot_offset, float d, bool multi_axis)
618 {
619  axis_config& ac = get_domain_config_ptr()->axis_configs[ai];
620  float acw = 1.5f * ac.line_width * get_domain_config_ptr()->reference_size;
621  tick_config& tc = ti == 0 ? ac.primary_ticks : ac.secondary_ticks;
622  if (tc.type == TT_NONE)
623  return false;
624  float min_tick = ac.tick_space_from_attribute_space(ac.get_attribute_min());
625  float max_tick = ac.tick_space_from_attribute_space(ac.get_attribute_max());
626  int min_i = (int)ceil(min_tick / tc.step - std::numeric_limits<float>::epsilon());
627  int max_i = (int)((max_tick - fmod(max_tick, tc.step)) / tc.step);
628  // ignore secondary ticks on domain boundary
629  if (multi_axis) {
630  if (ti == 1 && min_i * tc.step - min_tick < std::numeric_limits<float>::epsilon())
631  ++min_i;
632  if (ti == 1 && max_i * tc.step - max_tick > -std::numeric_limits<float>::epsilon())
633  --max_i;
634  }
635  float lw = 0.5f * get_domain_config_ptr()->reference_size * tc.line_width;
636  float dl = 0.5f * get_domain_config_ptr()->reference_size * tc.length;
637  for (int i = min_i; i <= max_i; ++i) {
638  float c_tick = (float)(i * tc.step);
639  float c_attr = ac.attribute_space_from_tick_space(c_tick);
640  // ignore ticks on axes
641  if (multi_axis && (fabs(c_attr) < std::numeric_limits<float>::epsilon()))
642  continue;
643  // ignore secondary ticks on primary ticks
644  if (ti == 1 && fabs(fmod(c_tick, ac.primary_ticks.step)) < 0.00001f)
645  continue;
646  std::string label_str;
647  if (tc.label)
648  label_str = cgv::utils::to_string(c_attr);
649  float c_plot = plot_scale*ac.plot_space_from_window_space(ac.window_space_from_tick_space(c_tick))+plot_offset(ci);
650  vec2 mn, mx;
651  mn[ci] = c_plot - lw;
652  mx[ci] = c_plot + lw;
653  switch (tc.type) {
654  case TT_DASH:
655  mn[1 - ci] = -he + plot_offset(1-ci);
656  mx[1 - ci] = -he + dl + plot_offset(1 - ci);
657  R.push_back(box2(mn, mx));
658  C.push_back(ac.color);
659  D.push_back(d);
660  if (!label_str.empty()) {
661  mx[1 - ci] += acw;
662  tick_labels.push_back(label_info(mx.to_vec(), label_str, ci == 0 ? cgv::render::TA_BOTTOM : cgv::render::TA_LEFT));
663  if (ti == 1)
664  tick_labels.back().scale = 0.75f;
665  }
666  if (multi_axis) {
667  mn[1 - ci] = he - dl + plot_offset(1 - ci);
668  mx[1 - ci] = he + plot_offset(1 - ci);
669  R.push_back(box2(mn, mx));
670  C.push_back(ac.color);
671  D.push_back(d);
672  if (!label_str.empty()) {
673  mn[1 - ci] -= acw;
674  tick_labels.push_back(label_info(mn.to_vec(), label_str, ci == 0 ? cgv::render::TA_TOP : cgv::render::TA_RIGHT));
675  if (ti == 1)
676  tick_labels.back().scale = 0.75f;
677  }
678  if (z_plot != std::numeric_limits<float>::quiet_NaN()) {
679  mn[1 - ci] = z_plot - dl + plot_offset(1 - ci);
680  mx[1 - ci] = z_plot + dl + plot_offset(1 - ci);
681  R.push_back(box2(mn, mx));
682  C.push_back(ac.color);
683  D.push_back(d);
684  }
685  }
686  break;
687  case TT_LINE:
688  case TT_PLANE:
689  mn[1 - ci] = -he + plot_offset(1-ci);
690  mx[1 - ci] = he + plot_offset(1-ci);
691  R.push_back(box2(mn, mx));
692  C.push_back(ac.color);
693  D.push_back(d);
694  if (!label_str.empty()) {
695  mn[1 - ci] += acw;
696  mx[1 - ci] -= acw;
697  mn[ci] += 2.5f * lw;
698  mx[ci] += 2.5f * lw;
699  if (multi_axis) {
700  tick_labels.push_back(label_info(mx.to_vec(), label_str, cgv::render::TextAlignment(ci == 0 ? cgv::render::TA_TOP + cgv::render::TA_LEFT : cgv::render::TA_RIGHT + cgv::render::TA_BOTTOM)));
701  if (ti == 1)
702  tick_labels.back().scale = 0.75f;
703  }
704  tick_labels.push_back(label_info(mn.to_vec(), label_str, cgv::render::TextAlignment(ci == 0 ? cgv::render::TA_BOTTOM + cgv::render::TA_LEFT : cgv::render::TA_LEFT + cgv::render::TA_BOTTOM)));
705  if (ti == 1)
706  tick_labels.back().scale = 0.75f;
707 
708  }
709  break;
710  }
711  }
712  return true;

1039 bool plot_base::determine_axis_extent_from_subplot(unsigned ai, unsigned i, float& samples_min, float& samples_max)
1040 {
1041  if (attribute_source_arrays.size() <= i)
1042  return false;
1043  if (attribute_source_arrays[i].attribute_sources.size() <= ai)
1044  return false;
1045  const attribute_source& as = attribute_source_arrays[i].attribute_sources[ai];
1046  if (as.source == AS_NONE)

1050  switch (as.source) {
1051  case AS_SAMPLE_CONTAINER:
1052  {
1053  int j = as.sub_plot_index == -1 ? i : as.sub_plot_index;
1054  new_found_sample = compute_sample_coordinate_interval(j, (int)as.offset, new_samples_min, new_samples_max);
1055  break;

1057  case AS_POINTER:

1059  const float* ptr = as.pointer;
1060  for (unsigned j = 0; j < as.count; ++j) {
1061  if (new_found_sample) {
1062  new_samples_min = std::min(new_samples_min, *ptr);
1063  new_samples_max = std::max(new_samples_max, *ptr);
1064  }
1065  else {
1066  new_samples_min = new_samples_max = *ptr;
1067  new_found_sample = true;
1068  }
1069  reinterpret_cast<const char*&>(ptr) += as.stride;
1070  }
1071  break;
1072  }
1073  case AS_VBO:
1074  return false;
1075  }
1076  if (!new_found_sample)
1077  return false;
1078 
1079  samples_min = new_samples_min;
1080  samples_max = new_samples_max;
1081  return true;
