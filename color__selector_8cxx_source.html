35 bool color_selector::handle_event(cgv::gui::event& e) {
36 
37  // return true if the event gets handled and stopped here or false if you want to pass it to the next plugin
38  unsigned et = e.get_kind();
39  unsigned char modifiers = e.get_modifiers();
40 
41  if (et == cgv::gui::EID_MOUSE) {
42  cgv::gui::mouse_event& me = (cgv::gui::mouse_event&)e;
43  cgv::gui::MouseAction ma = me.get_action();
44 
45  if (me.get_button_state() & cgv::gui::MB_LEFT_BUTTON) {
46  if (ma == cgv::gui::MA_PRESS) {
47  ivec2 mpos = get_local_mouse_pos(ivec2(me.get_x(), me.get_y()));
48 
49  int hit_index = -1;
50  cgv::g2d::irect hit_rect;
51 
52  if(layout.color_rect.contains(mpos)) {
53  hit_index = 0;
54  hit_rect = layout.color_rect;
55  }
56 
57  if(layout.hue_rect.contains(mpos)) {
58  hit_index = 1;
59  hit_rect = layout.hue_rect;
60  }
61 
62  if(layout.opacity_rect.contains(mpos)) {
63  hit_index = 2;
64  hit_rect = layout.opacity_rect;
65  }
66 
67  if(hit_index > -1 && hit_index < 4) {
68  vec2 local_mpos = static_cast<vec2>(mpos - hit_rect.position);
69  vec2 val = local_mpos / static_cast<vec2>(hit_rect.size);
70  if(hit_index > 0)
71  val.x() = 0.0f;
72  selector_handles[hit_index].val = val;
73  selector_handles[hit_index].update_pos();
74 
75  update_color();
76 
77  if(hit_index == 1)
78  update_color_texture();
79 
80  post_damage();
81  }
82  }
83  }
84 
85  if(selector_handles.handle(e, get_viewport_size(), get_rectangle()))
86  return true;
87  }
88  return false;

422 void color_selector::update_color() {
423 
424  const auto& cp = selector_handles[0];
425  const auto& hp = selector_handles[1];
426  const auto& op = selector_handles[2];
427  rgb_color = cgv::media::color<float, cgv::media::HLS>(hp.val.y(), 0.5f, 1.0f);
428 
429  float s = cp.val.x(); // saturation
430  float v = cp.val.y(); // value
431 
432  rgb_color = v * rgb_color;
433  rgb_color = (1.0f - s)*rgb(v) + s * rgb_color;
434 
435  rgba_color = rgba(rgb_color, op.val.y());
436 
437  update_texts();
438 
439  if(has_opacity) {
440  if(on_change_rgba_callback)
441  on_change_rgba_callback(rgba_color);
442  } else {
443  if(on_change_rgb_callback)
444  on_change_rgb_callback(rgb_color);
445  }
446 
447  update_member(&rgb_color);
448 }

481 void color_selector::set_color(rgba color, bool opacity, bool init) {
482 
483  this->rgb_color.R() = color.R();
484  this->rgb_color.G() = color.G();
485  this->rgb_color.B() = color.B();
486  this->rgba_color = color;
487 
488  float h = color.H();
489  float v = v = color.S() * std::min(color.L(), 1.0f - color.L()) + color.L();
490  float s = v ? 2.0f - 2.0f * color.L() / v : 0.0f;
491 
492  selector_handles[0].val = vec2(s, v);
493  selector_handles[1].val.y() = h;
494  selector_handles[2].val.y() = color.alpha();
495 
496  selector_handles[0].update_pos();
497  selector_handles[1].update_pos();
498  selector_handles[2].update_pos();
499 
500  if(has_opacity != opacity) {
501  // need to recreate the layout and gui in case the opacity usage changed
502  post_recreate_layout();
503  post_recreate_gui();
504  has_opacity = opacity;
505  }
506 
507  update_color_texture();
508  update_texts();
509 
510  if(!init && auto_show)
511  set_visibility(true);
512 
513  post_damage();
