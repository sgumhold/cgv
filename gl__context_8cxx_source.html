355 bool gl_context::configure_gl()
356 {
357  if (!ensure_glew_initialized()) {
358  error("gl_context::configure_gl could not initialize glew");
359  return false;
360  }
361  const GLubyte* version_string = glGetString(GL_VERSION);
362  version_major = version_string[0] - '0';
363  version_minor = version_string[2] - '0';
364  if (version_major >= 3) {
365  GLint context_flags;
366  glGetIntegerv(GL_CONTEXT_FLAGS, &context_flags);
367 #ifdef WIN32
368  // weird behavior under windows or just nvidia or just my laptop (Stefan)??
369  debug = (context_flags & WGL_CONTEXT_FORWARD_COMPATIBLE_BIT_ARB) != 0;
370  forward_compatible = (context_flags & WGL_CONTEXT_DEBUG_BIT_ARB) != 0;
371 #else
372  debug = (context_flags & GLX_CONTEXT_DEBUG_BIT_ARB) != 0;
373  forward_compatible = (context_flags & GLX_CONTEXT_FORWARD_COMPATIBLE_BIT_ARB) != 0;
374 #endif
375  }
376  else {
377  debug = false;
378  forward_compatible = false;
379  }
380  int version = 10 * version_major + version_minor;
381  if (version >= 32) {
382  GLint context_profile;
383  glGetIntegerv(GL_CONTEXT_PROFILE_MASK, &context_profile);
384 #ifdef WIN32
385  core_profile = (context_profile & WGL_CONTEXT_CORE_PROFILE_BIT_ARB) != 0;
386 #else
387  core_profile = (context_profile & GLX_CONTEXT_CORE_PROFILE_BIT_ARB) != 0;
388 #endif
389  }
390  else
391  core_profile = false;
392 
393  const GLubyte* vendor_c_string = glGetString(GL_VENDOR);
394  std::string vendor_string(reinterpret_cast<const char*>(vendor_c_string));
395  vendor_string = cgv::utils::to_upper(vendor_string);
396 
397  if (vendor_string.find("NVIDIA"))
398  gpu_vendor = GPU_VENDOR_NVIDIA;
399  else if (vendor_string.find("INTEL"))
400  gpu_vendor = GPU_VENDOR_INTEL;
401  else if (vendor_string.find("AMD") || vendor_string.find("ATI"))
402  gpu_vendor = GPU_VENDOR_AMD;
403 
404 #ifdef _DEBUG
405  std::cout << "OpenGL version " << version_major << "." << version_minor << (core_profile?" (core)":"") << (debug?" (debug)":"") << (forward_compatible?" (forward_compatible)":"") << std::endl;
406  const GLubyte* renderer_c_string = glGetString(GL_RENDERER);
407  const GLubyte* glslversion_c_string = glGetString(GL_SHADING_LANGUAGE_VERSION);
408  if (vendor_c_string)
409  std::cout << " vendor : " << vendor_c_string << std::endl;
410  if (renderer_c_string)
411  std::cout << " renderer : " << renderer_c_string << std::endl;
412  if (glslversion_c_string)
413  std::cout << " glslversion: " << glslversion_c_string << std::endl;
414 #endif
415  if (debug) {
416  glEnable(GL_DEBUG_OUTPUT);
417  if (version >= 43) {
418  glEnable(GL_DEBUG_OUTPUT_SYNCHRONOUS);
419  if (!check_gl_error("gl_context::configure() debug output"))
420  glDebugMessageCallback(debug_callback, this);
421  }
422  }
423  //enable_font_face(info_font_face, info_font_size);
424  // use the eye location to compute the specular lighting
425  if (!core_profile) {
426  glLightModelf(GL_LIGHT_MODEL_LOCAL_VIEWER, 0);
427  // this makes opengl normalize all surface normals before lighting calculations,
428  // which is essential when using scaling to deform tesselated primities
429  glEnable(GL_NORMALIZE);
430 
431  // should be initialized by the driver, but better be safe than risk errors later
432  glMatrixMode(GL_MODELVIEW);
433  }
434  set_viewport(ivec4(0, 0, get_width(), get_height()));
435 // if (check_gl_error("gl_context::configure_gl before init of children"))
436 // return false;
437 
438  group_ptr grp(dynamic_cast<group*>(this));
439  single_method_action<cgv::render::drawable, bool, cgv::render::context&> sma(*this, &drawable::init, false, false);
440  for (unsigned i = 0; i<grp->get_nr_children(); ++i)
441  traverser(sma, "nc").traverse(grp->get_child(i));
442 
443 // if (check_gl_error("gl_context::configure_gl after init of children."))
444 // return false;
445 
446  return true;
447 }

509 void gl_context::init_render_pass()
510 {
511  if (info_font_face.empty()) {
512  font_ptr info_font = default_font(true);
513  if (!info_font.empty()) {
514  info_font_face = info_font->get_font_face(FFA_REGULAR);
515  info_font_face->enable(this, info_font_size);
516  if (current_font_face.empty())
517  enable_font_face(info_font_face, info_font_size);
518  }
519  }
520 #ifdef WIN32
521  wglSwapIntervalEXT(enable_vsync ? 1 : 0);
522 #else
523  glXSwapIntervalEXT(glXGetCurrentDisplay(), glXGetCurrentDrawable(), enable_vsync ? 1 : 0);
524 #endif
525  if (sRGB_framebuffer)
526  glEnable(GL_FRAMEBUFFER_SRGB);
527  else
528  glDisable(GL_FRAMEBUFFER_SRGB);
529 
530  static cgv::render::RenderPassFlags last_render_pass_flags = get_default_render_pass_flags();
531  cgv::render::RenderPassFlags current_render_pass_flags = get_render_pass_flags();
532  if (current_render_pass_flags & RPF_SET_LIGHTS) {
533  for (unsigned i = 0; i < nr_default_light_sources; ++i)
534  set_light_source(default_light_source_handles[i], default_light_source[i], false);
535 
536  for (unsigned i = 0; i < nr_default_light_sources; ++i)
537  if (current_render_pass_flags & RPF_SET_LIGHTS_ON)
538  enable_light_source(default_light_source_handles[i]);
539  else
540  disable_light_source(default_light_source_handles[i]);
541  }
542  else if ((last_render_pass_flags & RPF_SET_LIGHTS) == 0) {
543  for (unsigned i = 0; i < nr_default_light_sources; ++i)
544  if (is_light_source_enabled(default_light_source_handles[i]))
545  disable_light_source(default_light_source_handles[i]);
546  }
547  last_render_pass_flags = current_render_pass_flags;
548 
549  if (get_render_pass_flags()&RPF_SET_MATERIAL) {
550  set_material(default_material);
551  }
552  if ((get_render_pass_flags()&RPF_ENABLE_MATERIAL) && !core_profile) {
553  // this mode allows to define the ambient and diffuse color of the surface material
554  // via the glColor commands
555  glEnable(GL_COLOR_MATERIAL);
556  }
557  if (get_render_pass_flags()&RPF_SET_STATE_FLAGS) {
558  // set some default settings
559  enable_depth_test();
560  set_cull_state(CM_BACKFACE);
561  if (!core_profile)
562  glEnable(GL_NORMALIZE);
563  }
564  if ((get_render_pass_flags()&RPF_SET_PROJECTION) != 0)
565  set_projection_matrix(cgv::math::perspective4<double>(45.0, (double)get_width()/get_height(),0.001,1000.0));
566 
567  if ((get_render_pass_flags()&RPF_SET_MODELVIEW) != 0)
568  set_modelview_matrix(cgv::math::look_at4<double>(vec3(0,0,10), vec3(0,0,0), vec3(0,1,0)));
569 
570  if (check_gl_error("gl_context::init_render_pass before init_frame"))
571  return;
572 
573  group* grp = dynamic_cast<group*>(this);
574  if (grp && (get_render_pass_flags()&RPF_DRAWABLES_INIT_FRAME)) {
575  single_method_action<drawable,void,cgv::render::context&> sma(*this, &drawable::init_frame, true, true);
576  traverser(sma).traverse(group_ptr(grp));
577  }
578 
579  if (check_gl_error("gl_context::init_render_pass after init_frame"))
580  return;
581  // this defines the background color to which the frame buffer is set by glClear
582  if(get_render_pass_flags() & RPF_SET_CLEAR_COLOR)
583  set_bg_color(get_bg_color());
584  // this defines the background depth buffer value set by glClear
585  if(get_render_pass_flags() & RPF_SET_CLEAR_DEPTH)
586  set_bg_depth(get_bg_depth());
587  // this defines the background depth buffer value set by glClear
588  if(get_render_pass_flags() & RPF_SET_CLEAR_STENCIL)
589  set_bg_stencil(get_bg_stencil());
590  // this defines the background color to which the accum buffer is set by glClear
591  if(get_render_pass_flags() & RPF_SET_CLEAR_ACCUM && !core_profile)
592  set_bg_accum_color(get_bg_accum_color());
593  clear_background(
594  get_render_pass_flags() & RPF_CLEAR_COLOR,
595  get_render_pass_flags() & RPF_CLEAR_DEPTH,
596  get_render_pass_flags() & RPF_CLEAR_STENCIL,
597  get_render_pass_flags() & RPF_CLEAR_ACCUM
598  );
599 }

1055 bool gl_context::read_frame_buffer(data::data_view& dv,
1056  unsigned int x, unsigned int y, FrameBufferType buffer_type,
1057  TypeId type, data::ComponentFormat cf, int w, int h)
1058 {
1059  const cgv::data::data_format* df = dv.get_format();
1060  if (df) {
1061  w = int(df->get_width());
1062  h = int(df->get_height());
1063  type = df->get_component_type();
1064  cf = df->get_standard_component_format();
1065  if (w < 1 || h < 1) {
1066  error(std::string("read_frame_buffer: received invalid dimensions (") + cgv::utils::to_string(w) + "," + cgv::utils::to_string(h) + ")");
1067  return false;
1068  }
1069  }
1070  else {
1071  if (w < 1 || h < 1) {
1072  GLint vp[4];
1073  glGetIntegerv(GL_VIEWPORT,vp);
1074  w = w < 1 ? vp[2] : w;
1075  h = h < 1 ? vp[3] : h;
1076  }
1077  }
1078  GLuint gl_type = map_to_gl(type);
1079  if (gl_type == 0) {
1080  error(std::string("read_frame_buffer: could not make component type ")+cgv::type::info::get_type_name(df->get_component_type())+" to gl type");
1081  return false;
1082  }
1083  GLuint gl_format = GL_DEPTH_COMPONENT;
1084  if (cf != cgv::data::CF_D) {
1085  gl_format = GL_STENCIL_INDEX;
1086  if (cf != cgv::data::CF_S) {
1087  gl_format = map_to_gl(cf);
1088  if (gl_format == GL_RGB && cf != cgv::data::CF_RGB) {
1089  error(std::string("read_frame_buffer: could not match component format ") + cgv::utils::to_string(df->get_component_format()));
1090  return false;
1091  }
1092  }
1093  }
1094  GLenum gl_buffer = GL_BACK;
1095  if (buffer_type < FB_BACK)
1096  gl_buffer = GL_COLOR_ATTACHMENT0+buffer_type;
1097  else {
1098  switch (buffer_type) {
1099  case FB_FRONT : gl_buffer = GL_FRONT; break;
1100  case FB_BACK : gl_buffer = GL_BACK; break;
1101  case FB_FRONT_LEFT : gl_buffer = GL_FRONT_LEFT; break;
1102  case FB_FRONT_RIGHT : gl_buffer = GL_FRONT_RIGHT; break;
1103  case FB_BACK_LEFT : gl_buffer = GL_BACK_LEFT; break;
1104  case FB_BACK_RIGHT : gl_buffer = GL_BACK_RIGHT; break;
1105  default:
1106  error(std::string("invalid buffer type ")+cgv::utils::to_string(buffer_type));
1107  return false;
1108  }
1109  }
1110  // after all necessary information could be extracted, ensure that format
1111  // and data view are created
1112  if (!df) {
1113  df = new cgv::data::data_format(w,h,type,cf);
1114  dv.~data_view();
1115  new(&dv) data::data_view(df);
1116  dv.manage_format(true);
1117  }
1118  GLint old_buffer, old_pack;
1119  glGetIntegerv(GL_READ_BUFFER, &old_buffer);
1120  glGetIntegerv(GL_PACK_ALIGNMENT, &old_pack);
1121  glReadBuffer(gl_buffer);
1122  glPixelStorei(GL_PACK_ALIGNMENT, df->get_component_alignment());
1123  glReadPixels(x,y,w,h,gl_format,gl_type,dv.get_ptr<void>());
1124  glReadBuffer(old_buffer);
1125  glPixelStorei(GL_PACK_ALIGNMENT, old_pack);
1126  dv.reflect_horizontally();
1127  return true;
1128 }
