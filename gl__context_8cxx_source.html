204  version_major = version_string[0] - '0';
205  version_minor = version_string[2] - '0';
206  if (version_major >= 3) {
207  GLint context_flags;
208  glGetIntegerv(GL_CONTEXT_FLAGS, &context_flags);
209 #ifdef WIN32
210  // weird behavior under windows or just nvidia or just my laptop (Stefan)??
211  debug = (context_flags & WGL_CONTEXT_FORWARD_COMPATIBLE_BIT_ARB) != 0;
212  forward_compatible = (context_flags & WGL_CONTEXT_DEBUG_BIT_ARB) != 0;
213 #else
214  debug = (context_flags & GLX_CONTEXT_DEBUG_BIT_ARB) != 0;
215  forward_compatible = (context_flags & GLX_CONTEXT_FORWARD_COMPATIBLE_BIT_ARB) != 0;
216 #endif
217  }
218  else {
219  debug = false;
220  forward_compatible = false;
221  }
222  int version = 10 * version_major * version_minor;
223  if (version >= 32) {
224  GLint context_profile;
225  glGetIntegerv(GL_CONTEXT_PROFILE_MASK, &context_profile);
226 #ifdef WIN32
227  core_profile = (context_profile & WGL_CONTEXT_CORE_PROFILE_BIT_ARB) != 0;
228 #else
229  core_profile = (context_profile & GLX_CONTEXT_CORE_PROFILE_BIT_ARB) != 0;
230 #endif
231  }
232  else
233  core_profile = false;
234 
235  const GLubyte* vendor_c_string = glGetString(GL_VENDOR);
236  std::string vendor_string(reinterpret_cast<const char*>(vendor_c_string));
237  vendor_string = cgv::utils::to_upper(vendor_string);
238

240  gpu_vendor = GPU_VENDOR_NVIDIA;
241  else if (vendor_string.find("INTEL"))
242  gpu_vendor = GPU_VENDOR_INTEL;
243  else if (vendor_string.find("AMD") || vendor_string.find("ATI"))
244  gpu_vendor = GPU_VENDOR_AMD;
245 
246 #ifdef _DEBUG
247  std::cout << "OpenGL version " << version_major << "." << version_minor << (core_profile?" (core)":"") << (debug?" (debug)":"") << (forward_compatible?" (forward_compatible)":"") << std::endl;
248  const GLubyte* renderer_c_string = glGetString(GL_RENDERER);
249  const GLubyte* glslversion_c_string = glGetString(GL_SHADING_LANGUAGE_VERSION);
250  if (vendor_c_string)
251  std::cout << " vendor : " << vendor_c_string << std::endl;
252  if (renderer_c_string)
253  std::cout << " renderer : " << renderer_c_string << std::endl;
254  if (glslversion_c_string)
255  std::cout << " glslversion: " << glslversion_c_string << std::endl;
256 #endif
257  if (debug) {
258  glEnable(GL_DEBUG_OUTPUT);
259  if (version >= 43) {
260  glEnable(GL_DEBUG_OUTPUT_SYNCHRONOUS);
261  if (!check_gl_error("gl_context::configure() debug output"))
262  glDebugMessageCallback(debug_callback, this);
263  }
264  }
265  //enable_font_face(info_font_face, info_font_size);
266  // use the eye location to compute the specular lighting
267  if (!core_profile) {
268  glLightModelf(GL_LIGHT_MODEL_LOCAL_VIEWER, 0);
269  // this makes opengl normalize all surface normals before lighting calculations,
270  // which is essential when using scaling to deform tesselated primities
271  glEnable(GL_NORMALIZE);
272  }

274 // if (check_gl_error("gl_context::configure_gl before init of children"))
275 // return false;
276

279  for (unsigned i = 0; i<grp->get_nr_children(); ++i)
280  traverser(sma, "nc").traverse(grp->get_child(i));
281 
282 // if (check_gl_error("gl_context::configure_gl after init of children."))
283 // return false;
284 
285  return true;

313 void gl_context::init_render_pass()
314 {
315  if (info_font_face.empty()) {
316  font_ptr info_font = default_font(true);
317  if (!info_font.empty()) {
318  info_font_face = info_font->get_font_face(FFA_REGULAR);
319  info_font_face->enable(this, info_font_size);
320  if (current_font_face.empty())
321  enable_font_face(info_font_face, info_font_size);
322  }
323  }
324 #ifdef WIN32
325  wglSwapIntervalEXT(enable_vsync ? 1 : 0);
326 #else
327  glXSwapIntervalEXT(glXGetCurrentDisplay(), glXGetCurrentDrawable(), enable_vsync ? 1 : 0);
328 #endif
329  if (sRGB_framebuffer)
330  glEnable(GL_FRAMEBUFFER_SRGB);
331  else
332  glDisable(GL_FRAMEBUFFER_SRGB);
333 
334 // glPushAttrib(GL_ALL_ATTRIB_BITS);
335  static cgv::render::RenderPassFlags last_render_pass_flags = get_default_render_pass_flags();
336  cgv::render::RenderPassFlags current_render_pass_flags = get_render_pass_flags();
337  if (current_render_pass_flags & RPF_SET_LIGHTS) {
338  for (unsigned i = 0; i < nr_default_light_sources; ++i)
339  set_light_source(default_light_source_handles[i], default_light_source[i], false);
340 
341  for (unsigned i = 0; i < nr_default_light_sources; ++i)
342  if (current_render_pass_flags & RPF_SET_LIGHTS_ON)
343  enable_light_source(default_light_source_handles[i]);
344  else
345  disable_light_source(default_light_source_handles[i]);
346  }
347  else if ((last_render_pass_flags & RPF_SET_LIGHTS) == 0) {
348  for (unsigned i = 0; i < nr_default_light_sources; ++i)
349  if (is_light_source_enabled(default_light_source_handles[i]))
350  disable_light_source(default_light_source_handles[i]);
351  }
352  last_render_pass_flags = current_render_pass_flags;
353 
354  if (get_render_pass_flags()&RPF_SET_MATERIAL) {
355  set_material(default_material);
356  }
357  if ((get_render_pass_flags()&RPF_ENABLE_MATERIAL) && !core_profile) {
358  // this mode allows to define the ambient and diffuse color of the surface material
359  // via the glColor commands
360  glEnable(GL_COLOR_MATERIAL);
361  }
362  if (get_render_pass_flags()&RPF_SET_STATE_FLAGS) {
363  // set some default settings
364  glEnable(GL_DEPTH_TEST);
365  glCullFace(GL_BACK);
366  glEnable(GL_CULL_FACE);
367  if (!core_profile)
368  glEnable(GL_NORMALIZE);
369  }
370  if ((get_render_pass_flags()&RPF_SET_PROJECTION) != 0) {
371  set_projection_matrix(cgv::math::perspective4<double>(45.0, (double)get_width()/get_height(),0.001,1000.0));
372  }
373  //glMatrixMode(GL_MODELVIEW);
374  if ((get_render_pass_flags()&RPF_SET_MODELVIEW) != 0)
375  set_modelview_matrix(cgv::math::look_at4<double>(vec3(0,0,10), vec3(0,0,0), vec3(0,1,0)));
376 
377  if (check_gl_error("gl_context::init_render_pass before init_frame"))
378  return;
379 
380  group* grp = dynamic_cast<group*>(this);
381  if (grp && (get_render_pass_flags()&RPF_DRAWABLES_INIT_FRAME)) {
382  single_method_action<drawable,void,cgv::render::context&> sma(*this, &drawable::init_frame, true, true);
383  traverser(sma).traverse(group_ptr(grp));
384  }
385 
386  if (check_gl_error("gl_context::init_render_pass after init_frame"))
387  return;
388  // this defines the background color to which the frame buffer is set by glClear
389  if (get_render_pass_flags()&RPF_SET_CLEAR_COLOR)
390  glClearColor(bg_r,bg_g,bg_b,bg_a);
391  // this defines the background color to which the accum buffer is set by glClear
392  if (get_render_pass_flags()&RPF_SET_CLEAR_ACCUM)
393  glClearAccum(bg_accum_r,bg_accum_g,bg_accum_b,bg_accum_a);
394  // this defines the background depth buffer value set by glClear
395  if (get_render_pass_flags()&RPF_SET_CLEAR_DEPTH)
396  glClearDepth(bg_d);
397  // this defines the background depth buffer value set by glClear
398  if (get_render_pass_flags()&RPF_SET_CLEAR_STENCIL)
399  glClearStencil(bg_s);
400  // clear necessary buffers
401  GLenum bits = 0;
402  if (get_render_pass_flags()&RPF_CLEAR_COLOR)
403  bits |= GL_COLOR_BUFFER_BIT;
404  if (get_render_pass_flags()&RPF_CLEAR_DEPTH)
405  bits |= GL_DEPTH_BUFFER_BIT;
406  if (get_render_pass_flags()&RPF_CLEAR_STENCIL)
407  bits |= GL_STENCIL_BUFFER_BIT;
408  if (get_render_pass_flags()&RPF_CLEAR_ACCUM)
409  bits |= GL_ACCUM_BUFFER_BIT;
410  if (bits)
411  glClear(bits);
412 }

758 void gl_context::draw_light_source(const light_source& l, float i, float light_scale)
759 {
760  set_color(i*l.get_emission());
761  push_modelview_matrix();
762  switch (l.get_type()) {
763  case LT_DIRECTIONAL :
764  mul_modelview_matrix(cgv::math::scale4<double>(light_scale, light_scale, light_scale));
765  tesselate_arrow(vec3(0.0f), l.get_position(), 0.1f,2.0f,0.5f);

868 bool gl_context::read_frame_buffer(data::data_view& dv,
869  unsigned int x, unsigned int y, FrameBufferType buffer_type,
870  TypeId type, data::ComponentFormat cf, int w, int h)
871 {
872  const cgv::data::data_format* df = dv.get_format();
873  if (df) {
874  w = df->get_width();
875  h = df->get_height();
876  type = df->get_component_type();
877  cf = df->get_standard_component_format();
878  if (w < 1 || h < 1) {
879  error(std::string("read_frame_buffer: received invalid dimensions (") + cgv::utils::to_string(w) + "," + cgv::utils::to_string(h) + ")");
880  return false;
881  }
882  }
883  else {
884  if (w < 1 || h < 1) {
885  GLint vp[4];
886  glGetIntegerv(GL_VIEWPORT,vp);
887  w = w < 1 ? vp[2] : w;
888  h = h < 1 ? vp[3] : h;
889  }
890  }
891  GLuint gl_type = map_to_gl(type);
892  if (gl_type == 0) {
893  error(std::string("read_frame_buffer: could not make component type ")+cgv::type::info::get_type_name(df->get_component_type())+" to gl type");
894  return false;
895  }
896  GLuint gl_format = GL_DEPTH_COMPONENT;
897  if (cf != cgv::data::CF_D) {
898  gl_format = GL_STENCIL_INDEX;
899  if (cf != cgv::data::CF_S) {
900  gl_format = map_to_gl(cf);
901  if (gl_format == GL_RGB && cf != cgv::data::CF_RGB) {
902  error(std::string("read_frame_buffer: could not match component format ") + cgv::utils::to_string(df->get_component_format()));
903  return false;
904  }
905  }
906  }
907  GLenum gl_buffer = GL_BACK;
908  if (buffer_type < FB_BACK)
909  gl_buffer = GL_COLOR_ATTACHMENT0+buffer_type;
910  else {
911  switch (buffer_type) {
912  case FB_FRONT : gl_buffer = GL_FRONT; break;
913  case FB_BACK : gl_buffer = GL_BACK; break;
914  case FB_FRONT_LEFT : gl_buffer = GL_FRONT_LEFT; break;
915  case FB_FRONT_RIGHT : gl_buffer = GL_FRONT_RIGHT; break;
916  case FB_BACK_LEFT : gl_buffer = GL_BACK_LEFT; break;
917  case FB_BACK_RIGHT : gl_buffer = GL_BACK_RIGHT; break;
918  default:
919  error(std::string("invalid buffer type ")+cgv::utils::to_string(buffer_type));
920  return false;
921  }
922  }
923  // after all necessary information could be extracted, ensure that format
924  // and data view are created
925  if (!df) {
926  df = new cgv::data::data_format(w,h,type,cf);
927  dv.~data_view();
928  new(&dv) data::data_view(df);
929  dv.manage_format(true);
930  }
931  GLint old_buffer, old_pack;
932  glGetIntegerv(GL_READ_BUFFER, &old_buffer);
933  glGetIntegerv(GL_PACK_ALIGNMENT, &old_pack);
934  glReadBuffer(gl_buffer);
935  glPixelStorei(GL_PACK_ALIGNMENT, df->get_component_alignment());
936  glReadPixels(x,y,w,h,gl_format,gl_type,dv.get_ptr<void>());
937  glReadBuffer(old_buffer);
938  glPixelStorei(GL_PACK_ALIGNMENT, old_pack);
939  dv.reflect_horizontally();
940  return true;
941 }

1003  const int* vertex_indices, const int* normal_indices, const int* tex_coord_indices, bool flip_normals) const
1004 {
1005  unsigned i;
1006  shader_program* prog_ptr = static_cast<shader_program*>(get_current_program());
1007  if (!prog_ptr || prog_ptr->get_position_index() == -1)
1008  return false;

1011  P[i] = *reinterpret_cast<const vec3*>(vertices + 3 * vertex_indices[i]);

1016  N[i] = *reinterpret_cast<const vec3*>(normals + 3 * normal_indices[i]);
1017  if (flip_normals)
1018  N[i] = -N[i];
1019  }
1020  }
1021  if (prog_ptr->get_texcoord_index() != -1 && tex_coords && tex_coord_indices) {
1022  T.resize(nr_vertices);
1023  for (i = 0; i < nr_vertices; ++i)
1024  T[i] = *reinterpret_cast<const vec2*>(tex_coords + 2 * tex_coord_indices[i]);

1028  if (core_profile && !aab_ptr) {
1029  aab_ptr = new attribute_array_binding();
1030  aab_ptr->create(*this);
1031  }
1032  if (!aab_ptr) {
1033  attribute_array_binding::set_global_attribute_array<>(*this, prog_ptr->get_position_index(), P);
1034  attribute_array_binding::enable_global_array(*this, prog_ptr->get_position_index());
1035  if (prog_ptr->get_normal_index() != -1) {
1036  if (normals && normal_indices) {
1037  attribute_array_binding::set_global_attribute_array<>(*this, prog_ptr->get_normal_index(), N);
1038  attribute_array_binding::enable_global_array(*this, prog_ptr->get_normal_index());
1039  }
1040  else
1041  attribute_array_binding::disable_global_array(*this, prog_ptr->get_normal_index());
1042  }
1043  if (prog_ptr->get_texcoord_index() != -1) {
1044  if (tex_coords && tex_coord_indices) {
1045  attribute_array_binding::set_global_attribute_array<>(*this, prog_ptr->get_texcoord_index(), T);
1046  attribute_array_binding::enable_global_array(*this, prog_ptr->get_texcoord_index());
1047  }
1048  else
1049  attribute_array_binding::disable_global_array(*this, prog_ptr->get_texcoord_index());
1050  }
1051  }
1052  else {
1053  vertex_buffer*& vbo_ptr = get_vbo_ptr();
1054  if (!vbo_ptr)
1055  vbo_ptr = new vertex_buffer();
1056  else
1057  vbo_ptr->destruct(*this);
1058  vbo_ptr->create(*this, P.size() * sizeof(vec3) + N.size() * sizeof(vec3) + T.size() * sizeof(vec2));

1063  vbo_ptr->replace(const_cast<gl_context&>(*this), nml_off, &N.front(), N.size());
1064  tex_off += N.size() * sizeof(vec2);
1065  }

1067  vbo_ptr->replace(const_cast<gl_context&>(*this), tex_off, &T.front(), T.size());

1070  aab_ptr->set_attribute_array(*this, prog_ptr->get_position_index(), td3, *vbo_ptr, 0, P.size());
