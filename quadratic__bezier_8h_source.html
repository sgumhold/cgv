33 }

127template<typename T>
128static std::pair<T, T> quadratic_bezier_signed_distance(const quadratic_bezier_curve<fvec<T, 3>>& curve, const fvec<T, 3>& pos) {
129 using vec_type = fvec<T, 3>;
130 vec_type a = curve.p1 - curve.p0;
131 vec_type b = curve.p0 - T(2) * curve.p1 + curve.p2;
132 vec_type c = a * T(2);
133 vec_type d = curve.p0 - pos;
134
135 T kk = T(1) / dot(b, b);
136 T kx = kk * dot(a, b);
137 T ky = kk * (T(2) * dot(a, a) + dot(d, b)) / T(3);
138 T kz = kk * dot(d, a);
139
140 T dist = T(0);
141 T t = T(0);
142
143 T p = ky - kx * kx;
144 T p3 = p * p * p;
145 T q = kx * (T(2) * kx * kx - T(3) * ky) + kz;
146 T h = q * q + T(4) * p3;
147
148 if(h >= T(0)) {
149 h = std::sqrt(h);
150 fvec<T, 2> x = (vec2(h, -h) - q) / T(2);
151 fvec<T, 2> uv = sign(x) * pow(abs(x), vec2(T(1) / T(3)));
152 t = clamp(uv.x() + uv.y() - kx, T(0), T(1));
153
154 // 1 root
155 dist = sqr_length(d + (c + b * t) * t);
156 } else {
157 T z = std::sqrt(-p);
158 T v = std::acos(q / (p * z * T(2))) / T(3);
159 T m = std::cos(v);
160 T n = std::sin(v) * T(1.732050808);
161 vec_type ts = clamp(vec_type(m + m, -n - m, n - m) * z - kx, T(0), T(1));
162
163 // 3 roots, but only need two
164 T dt = sqr_length(d + (c + b * ts.x()) * ts.x());
165 dist = dt;
166 t = ts.x();
167
168 dt = sqr_length(d + (c + b * ts.y()) * ts.y());
169 if(dt < dist) {
170 dist = dt;
171 t = ts.y();
172 }
173 }
174
175 dist = sqrt(dist);
176
177 return { dist, t };
178}
