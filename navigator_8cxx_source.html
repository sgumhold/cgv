85 bool navigator::handle_event(cgv::gui::event& e) {
86 
87  // return true if the event gets handled and stopped here or false if you want to pass it to the next plugin
88  unsigned et = e.get_kind();
89  unsigned char modifiers = e.get_modifiers();
90 
91  if(et == cgv::gui::EID_MOUSE) {
92  cgv::gui::mouse_event& me = (cgv::gui::mouse_event&) e;
93  cgv::gui::MouseAction ma = me.get_action();
94 
95  int last_hit_axis = hit_axis;
96  hit_axis = 0;
97 
98  if(get_context()) {
99  context& ctx = *get_context();
100 
101  if(ma == cgv::gui::MA_LEAVE) {
102  hit_axis = 0;
103  } else {
104  ivec2 mpos(static_cast<int>(me.get_x()), static_cast<int>(me.get_y()));
105 
106  mpos = get_local_mouse_pos(mpos);
107  vec2 window_coord = vec2(mpos) * vec2(2.0f) / get_rectangle().size - vec2(1.0f);
108 
109  vec3 origin = vec3(window_coord, navigator_eye_pos.z());
110  vec3 direction = vec3(0.0f, 0.0f, -1.0f);
111 
112  if(use_perspective) {
113  mat4 MVP = get_projection_matrix() * get_view_matrix(ctx);
114 
115  vec4 world_coord(window_coord.x(), window_coord.y(), 1.0f, 1.0f);
116  world_coord = inv(MVP) * world_coord;
117  world_coord /= world_coord.w();
118 
119  vec3 origin = navigator_eye_pos;
120  vec3 direction = normalize(vec3(world_coord) - origin);
121  }
122 
123  mat4 IM = inv(get_model_matrix(ctx));
124 
125  origin = vec3(IM * vec4(origin, 1.0f));
126  direction = vec3(IM * vec4(direction, 0.0f));
127 
128  float t = std::numeric_limits<float>::max();
129  if(intersect_box(origin, direction, t)) {
130  vec3 hit_pos = origin + t * direction;
131 
132  unsigned mi = cgv::math::max_index(cgv::math::abs(hit_pos));
133 
134  hit_axis = static_cast<int>(mi) + 1;
135  if(hit_pos[mi] < 0.0f)
136  hit_axis = -hit_axis;
137  }
138  }
139 
140  if(last_hit_axis != hit_axis)
141  post_redraw();
142  }
143 
144  switch(ma) {
145  case cgv::gui::MA_PRESS:
146  if(me.get_button() == cgv::gui::MB_LEFT_BUTTON) {
147  check_for_click = me.get_time();
148  return true;
149  }
150  break;
151  case cgv::gui::MA_RELEASE:
152  if(check_for_click != -1) {
153  double dt = me.get_time() - check_for_click;
154  if(dt < 0.2) {
155  if(hit_axis != 0) {
156  int axis_idx = abs(hit_axis) - 1;
157 
158  vec3 view_dir(0.0f);
159  view_dir[axis_idx] = hit_axis < 0.0f ? -1.0f : 1.0f;
160 
161  if(view_ptr) {
162  vec3 focus = view_ptr->get_focus();
163  float dist = (focus - view_ptr->get_eye()).length();
164 
165  vec3 view_up_dir(0.0f, 1.0f, 0.0f);
166  if(axis_idx == 1)
167  view_up_dir = vec3(0.0f, 0.0f, hit_axis < 0 ? 1.0f : -1.0f);
168 
169  dvec3 axis;
170  double angle;
171  view_ptr->compute_axis_and_angle(-view_dir, view_up_dir, axis, angle);
172 
173  cgv::gui::animate_with_axis_rotation(view_ptr->ref_view_dir(), axis, angle, 0.5)->set_base_ptr(this);
174  cgv::gui::animate_with_axis_rotation(view_ptr->ref_view_up_dir(), axis, angle, 0.5)->set_base_ptr(this);
175 
176  post_redraw();
177  return true;
178  }
179  }
180  }
181  }
182  return true;
183  break;
184  }
185 
186  return false;
187  } else {
188  return false;
189  }

379 bool navigator::intersect_box(const vec3 &origin, const vec3& direction, float& t) const {
380 
381  vec3 min(-0.5f);
382  vec3 max(+0.5f);
383 
384  float t_min, t_max;
385 
386  vec3 inv_dir = vec3(1.0f) / direction;
387 
388  float t1 = (min.x() - origin.x())*inv_dir.x();
389  float t2 = (max.x() - origin.x())*inv_dir.x();
390 
391  t_min = std::min(t1, t2);
392  t_max = std::max(t1, t2);
393 
394  t1 = (min.y() - origin.y())*inv_dir.y();
395  t2 = (max.y() - origin.y())*inv_dir.y();
396 
397  t_min = std::max(t_min, std::min(t1, t2));
398  t_max = std::min(t_max, std::max(t1, t2));
399 
400  t1 = (min.z() - origin.z())*inv_dir.z();
401  t2 = (max.z() - origin.z())*inv_dir.z();
402 
403  t_min = std::max(t_min, std::min(t1, t2));
404  t_max = std::min(t_max, std::max(t1, t2));
405 
406  t = t_min;
407  return t_max > std::max(t_min, 0.0f);
408 }
