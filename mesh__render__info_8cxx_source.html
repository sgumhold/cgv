84 void mesh_render_info::construct_draw_calls(cgv::render::context& ctx)
85 {
86  // construct aa with new aab for surface rendering
87  int aa_index = (int)ref_aas().size();
88  ref_aas().push_back(attribute_array());
89  ref_aas().push_back(attribute_array());
90 
91  auto& aa = ref_aas()[aa_index];
92  auto& wire_aa = ref_aas()[aa_index + 1];
93 
94  aa.aab_ptr = new cgv::render::attribute_array_binding();
95  aa.aab_ptr->create(ctx);
96 
97  wire_aa.aab_ptr = new cgv::render::attribute_array_binding();
98  wire_aa.aab_ptr->create(ctx);
99 
100  // determine stride
101  uint32_t stride = 3;
102  if (include_tex_coords)
103  stride += 2;
104  if (include_normals)
105  stride += 3;
106  if(include_tangents)
107  stride += 3;
108  stride += uint32_t(color_increment);
109  if (stride == 3)
110  stride = 0;
111  else
112  stride *= uint32_t(element_size);
113 
114  // set element and attribute pointers
115  aa.aab_ptr->set_element_array(ctx, *ref_vbos()[1]);
116  aa.add_attribute(position_descr, 0, 0, nr_vertices, stride, cgv::render::VA_POSITION);
117  wire_aa.aab_ptr->set_element_array(ctx, *ref_vbos()[1]);
118  wire_aa.add_attribute(position_descr, 0, 0, nr_vertices, stride, cgv::render::VA_POSITION);
119  size_t offset = 3 * element_size;
120  if (include_tex_coords) {
121  aa.add_attribute(tex_coords_descr, 0, offset, nr_vertices, stride, cgv::render::VA_TEXCOORD);
122  wire_aa.add_attribute(tex_coords_descr, 0, offset, nr_vertices, stride, cgv::render::VA_TEXCOORD);
123  offset += 2 * element_size;
124  }
125  if (include_normals) {
126  aa.add_attribute(position_descr, 0, offset, nr_vertices, stride, cgv::render::VA_NORMAL);
127  wire_aa.add_attribute(position_descr, 0, offset, nr_vertices, stride, cgv::render::VA_NORMAL);
128  offset += 3 * element_size;
129  }
130  if(include_tangents) {
131  aa.add_attribute(position_descr, 0, offset, nr_vertices, stride, cgv::render::VA_BY_NAME, "tangent");
132  wire_aa.add_attribute(position_descr, 0, offset, nr_vertices, stride, cgv::render::VA_BY_NAME, "tangent");
133  offset += 3 * element_size;
134  }
135  if (color_increment > 0) {
136  static int nr_comps[] = { 4,4,3,4 };
137  static cgv::type::info::TypeId type_ids[] = { cgv::type::info::TI_UINT8,cgv::type::info::TI_UINT8,cgv::type::info::TI_FLT32,cgv::type::info::TI_FLT32 };
138  aa.add_attribute(cgv::render::type_descriptor(type_ids[ct], nr_comps[ct], true),
139  0, offset, nr_vertices, stride, cgv::render::VA_COLOR);
140  wire_aa.add_attribute(cgv::render::type_descriptor(type_ids[ct], nr_comps[ct], true),
141  0, offset, nr_vertices, stride, cgv::render::VA_COLOR);
142  offset += color_increment * element_size;
143  }
144 
145  // construct draw call structure which is independent of mesh fragment
146  cgv::render::draw_call dc;
147  dc.aa_index = aa_index;
148  dc.primitive_type = PT_TRIANGLES;
149  dc.vertex_offset = 0;
150  dc.draw_call_type = cgv::render::RCT_INDEXED;
151  dc.index_type = cgv::type::info::TI_UINT32;
152  dc.instance_count = 1;
153  dc.prog = 0;
154  dc.material_index = -1;
155  dc.alpha_mode = AM_OPAQUE;
156 
157  // in case of no fragments, generate a single draw call
158  if (material_primitive_start.empty()) {
159  dc.count = uint32_t(nr_triangle_elements);
160  dc.indices = 0;
161  ref_draw_calls().push_back(dc);
162  }
163  // otherwise, for each mesh fragment construct a draw call
164  else {
165  size_t fi;
166  for (fi = 0; fi < material_primitive_start.size(); ++fi) {
167  const auto& mps = material_primitive_start[fi];
168  size_t next_start = nr_triangle_elements;
169  if (fi + 1 < material_primitive_start.size())
170  next_start = material_primitive_start[fi + 1][2];
171  dc.count = uint32_t(next_start - mps[2]);
172  dc.indices = (void*)(3 * sizeof(uint32_t)*mps[2]);
173  dc.material_index = mps[0];
174  if (dc.material_index != -1) {
175  const auto& mat = *ref_materials()[dc.material_index];
176  if ((mat.get_transparency_index() != -1) ||
177  (mat.get_transparency() > 0.01f)) {
178  dc.alpha_mode = AM_MASK_AND_BLEND;
179  dc.alpha_cutoff = 0.02f;
180  }
181  else
182  dc.alpha_mode = AM_OPAQUE;
183  }
184  ref_draw_calls().push_back(dc);
185  }
186  }
187  // set draw call for wire frame rendering
188  wire_draw_call.aa_index = aa_index+1;
189  wire_draw_call.primitive_type = PT_LINES;
190  wire_draw_call.vertex_offset = 0;
191  wire_draw_call.draw_call_type = cgv::render::RCT_INDEXED;
192  wire_draw_call.count = uint32_t(nr_edge_elements);
193  wire_draw_call.index_type = cgv::type::info::TI_UINT32;
194  wire_draw_call.indices = (void*)(sizeof(uint32_t)*nr_triangle_elements);
195  wire_draw_call.material_index = -1;
196  wire_draw_call.alpha_mode = AM_OPAQUE;
197  wire_draw_call.instance_count = 1;
198  wire_draw_call.prog = 0;
199 }

229 void mesh_render_info::draw_primitive(cgv::render::context& ctx, size_t primitive_index, bool skip_opaque, bool skip_blended, bool use_materials)
230 {
231  // extract indices of to be drawn calls
232  std::vector<size_t> dcis;
233  size_t dci = 0;
234  while (dci < draw_calls.size()) {
235  while (dci < draw_calls.size() && material_primitive_start[dci][1] != primitive_index)
236  ++dci;
237  if (skip_opaque)
238  while (dci < draw_calls.size() && (draw_calls[dci].alpha_mode & AM_BLEND) == 0)
239  ++dci;
240  if (skip_blended)
241  while (dci < draw_calls.size() && (draw_calls[dci].alpha_mode & AM_BLEND) != 0)
242  ++dci;
243  if (dci >= draw_calls.size())
244  break;
245  dcis.push_back(dci);
246  ++dci;
247  }
248  // draw extracted calls
249  draw_call* prev_dc = 0;
250  for (size_t i = 0; i < dcis.size(); ++i) {
251  draw_call* next_dc = i + 1 < dcis.size() ? &draw_calls[dcis[i + 1]] : 0;
252  draw(ctx, draw_calls[dcis[i]], prev_dc, next_dc, use_materials);
253  prev_dc = &draw_calls[dcis[i]];
254  }
