135  void compute_edge_point(const T& _v_1, const T& _v_2, int e, const pnt_type& p, const vec_type& d,
136  pnt_type& q, vec_type& n)
137  {
138  X de = d(e);
139  T v_1 = _v_1;
140  T v_2 = _v_2;
141  pnt_type p_end = p;
142  q = p;
143  unsigned int nr_iters = 0;
144  bool finished;
145  X ref = consistency_threshold*fabs(v_2-v_1);
146  do {
147  finished = false;
148  // compute point on edge
149  X alpha;
150  if (fabs(v_2-v_1) > epsilon) {
151  finished = true;
152  alpha = (X)(iso_value-v_1)/(v_2-v_1);
153  }
154  else
155  alpha = (X) 0.5;
156 
157  q(e) = p_end(e) - (1-alpha)*de;
158  // compute normal at point
159  n = vec_type(func.evaluate_gradient(q.to_vec()));
160 
161  // stop if maximum number of iterations reached
162  if (++nr_iters >= max_nr_iters)
163  break;
164  // check if gradient is in accordance with value difference
165  if (finished) {
166  if (alpha < 0 || alpha > 1) {
167  std::cout << "invalid alpha = " << alpha << std::endl;
168  }
169  X f_e = (v_2 - v_1) / de;
170  if (fabs(f_e - n(e)) > consistency_threshold*f_e) {
171  X v = func.evaluate(q.to_vec());
172  if (fabs(v-iso_value) > ref) {
173  if ((v > iso_value) == (v_2 > iso_value)) {
174  de *= alpha;
175  v_2 = v;
176  p_end(e) = q(e);
177  }
178  else {
179  v_1 = v;
180  de *= 1-alpha;
181  }
182  finished = false;
183  }
184  }
185  }
186  } while (!finished);
187  if (nr_iters > 15)
188  std::cout << "not converged after " << nr_iters << " iterations" << std::endl;
189  if (n.length() < 1e-6)
190  std::cout << "degenerate gradient" << n << std::endl;
191  else
192  n.normalize();
193  }

284  void extract(const T& _iso_value,
285  const axis_aligned_box<X,3>& box,
286  unsigned int _resx, unsigned int _resy, unsigned int _resz,
287  bool show_progress = false)
288  {
289  // prepare private members
290  resx = _resx; resy = _resy; resz = _resz;
291  minp = p = box.get_min_pnt();
292  d = box.get_extent();
293  d(0) /= (resx-1); d(1) /= (resy-1); d(2) /= (resz-1);
294  iso_value = _iso_value;
295 
296  // prepare progression
297  cgv::utils::progression prog;
298  if (show_progress) prog.init("extraction", resz, 10);
299 
300  // construct three slice infos
301  dc_slice_info<T> slice_info_1(resx,resy), slice_info_2(resx,resy), slice_info_3(resx,resy);
302  dc_slice_info<T> *slice_info_ptrs[3] = { &slice_info_1, &slice_info_2, &slice_info_3 };
303 
304  // iterate through all slices
305  unsigned int nr_vertices[4] = { 0, 0, 0, 0 };
306  unsigned int k, n;
307 
308  process_slice(0, slice_info_ptrs[0]);
309  p(2) += d(2);
310  process_slice(slice_info_ptrs[0], slice_info_ptrs[1]);
311  process_slab(slice_info_ptrs[0], slice_info_ptrs[1]);
312  p(2) += d(2);
313  // show progression
314  if (show_progress) {
315  prog.step();
316  prog.step();
317  }
318  for (k=2; k<resz; ++k, p(2) += d(2)) {
319  n = base_type::get_nr_vertices();
320  // evaluate function on next slice and construct slice interior vertices
321  dc_slice_info<T> *info_ptr_0 = slice_info_ptrs[(k-2)%3];
322  dc_slice_info<T> *info_ptr_1 = slice_info_ptrs[(k-1)%3];
323  dc_slice_info<T> *info_ptr_2 = slice_info_ptrs[k%3];
324  process_slice(info_ptr_1, info_ptr_2);
325  process_slab(info_ptr_1, info_ptr_2);
326  generate_slice_quads(info_ptr_0, info_ptr_1);
327 
328  n = base_type::get_nr_vertices()-n;
329  nr_vertices[k%4] = n;
330  n = nr_vertices[(k+3)%4];
331  if (n > 0)
332  base_type::drop_vertices(n);
333  // show progression
334  if (show_progress)
335  prog.step();
336  }
337  }
