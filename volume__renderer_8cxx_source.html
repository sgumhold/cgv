77  void volume_renderer::init_noise_texture(context& ctx)
78  {
79  if(noise_texture.is_created())
80  noise_texture.destruct(ctx);
81 
82  unsigned size = 64;
83  std::vector<uint8_t> noise_data(size*size);
84 
85  std::default_random_engine rng(42);
86  std::uniform_int_distribution<unsigned> dist(0u, 255u);
87 
88  for(size_t i = 0; i < noise_data.size(); ++i)
89  noise_data[i] = static_cast<uint8_t>(dist(rng));
90 
91  cgv::data::data_view dv = cgv::data::data_view(new cgv::data::data_format(size, size, TI_UINT8, cgv::data::CF_R), noise_data.data());
92  noise_texture.create(ctx, dv, 0);
93  }

202  bool volume_renderer::enable(context& ctx) {
203  const volume_render_style& vrs = get_style<volume_render_style>();
204  if(!renderer::enable(ctx))
205  return false;
206  int vp[4];
207  glGetIntegerv(GL_VIEWPORT, vp);
208  ref_prog().set_uniform(ctx, "viewport_dims", vec2(float(vp[2] - vp[0]), float(vp[3] - vp[1])));
209  ref_prog().set_uniform(ctx, "noise_offset", noise_offset);
210 
211  ref_prog().set_uniform(ctx, "scale_adjustment_factor", vrs.scale_adjustment_factor);
212 
213  ref_prog().set_uniform(ctx, "light_local_to_eye", vrs.light_local_to_eye);
214  ref_prog().set_uniform(ctx, "light_direction", normalize(vrs.light_direction));
215  float specular_exponent = cgv::math::lerp(128.0f, 1.0f, cgv::math::clamp(vrs.roughness, 0.0f, 1.0f));
216  ref_prog().set_uniform(ctx, "ambient_strength", vrs.ambient_strength);
217  ref_prog().set_uniform(ctx, "diffuse_strength", vrs.diffuse_strength);
218  ref_prog().set_uniform(ctx, "specular_strength", vrs.specular_strength);
219  ref_prog().set_uniform(ctx, "specular_exponent", specular_exponent);
220  ref_prog().set_uniform(ctx, "specular_color_mix", vrs.specular_color_mix);
221 
222  ref_prog().set_uniform(ctx, "gradient_lambda", std::max(vrs.gradient_lambda, 0.001f));
223 
224  ref_prog().set_uniform(ctx, "isovalue", vrs.isovalue);
225  ref_prog().set_uniform(ctx, "isosurface_color", vrs.isosurface_color);
226 
227  ref_prog().set_uniform(ctx, "slice_axis", vrs.slice_axis);
228  ref_prog().set_uniform(ctx, "slice_coordinate", vrs.slice_coordinate);
229  ref_prog().set_uniform(ctx, "slice_opacity", vrs.slice_opacity);
230 
231  ref_prog().set_uniform(ctx, "clip_box_min", vrs.clip_box.get_min_pnt());
232 
233  ref_prog().set_uniform(ctx, "clip_box_max", vrs.clip_box.get_max_pnt());
234 
235  ctx.push_depth_test_state();
236  ctx.disable_depth_test();
237  ctx.push_blend_state();
238  ctx.enable_blending();
239  ctx.set_blend_func_back_to_front();
240  ctx.push_cull_state();
241  ctx.set_cull_state(CM_FRONTFACE);
242 
243  if(volume_texture) volume_texture->enable(ctx, 0);
244  if(transfer_function_texture) transfer_function_texture->enable(ctx, 1);
245  noise_texture.enable(ctx, 2);
246  if(gradient_texture) gradient_texture->enable(ctx, 3);
247  if(depth_texture) depth_texture->enable(ctx, 4);
248  return true;
249  }
