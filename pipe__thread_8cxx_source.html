13 void queued_output_thread::run()
14 {
15  if (!connect_to_child_process())
16  return;
17  m.lock();
18  connected = true;
19  m.unlock();
20  while (!have_stop_request()) {
21  // check for queued block
22  std::pair<char*, size_t> block = { 0,0 };
23  m.lock();
24  if (!blocks.empty()) {
25  block = blocks.front();
26  blocks.pop_front();
27  }
28  m.unlock();
29  // if this is available,
30  if (block.first) {
31  // send it to the pipe and delete block
32  write_block_to_pipe(block.first, block.second);
33  delete[] block.first;
34  }
35  else {
36  // otherwise check for termination
37  bool all_sent;
38  m.lock();
39  all_sent = all_data_sent;
40  m.unlock();
41  if (all_sent)
42  break;
43  wait(20);
44  }
45  }
46  // in case of stop request or done, close pipe
47  close();
48 }

57 bool queued_output_thread::send_block(const char* data, size_t count)
58 {
59  bool all_sent;
60  m.lock();
61  all_sent = all_data_sent;
62  m.unlock();
63  if (all_sent)
64  return false;
65  bool out_of_memory = false;
66  char* block;
67  try {
68  block = new char[count];
69  }
70  catch (const std::bad_alloc&) {
71  out_of_memory = true;
72  }
73  if (out_of_memory)
74  return false;
75  std::copy(data, data + count, block);
76  m.lock();
77  blocks.push_back({ block, count });
78  m.unlock();
79  return true;
80 }

166 void queued_input_thread::run()
167 {
168  if (!connect_to_source())
169  return;
170 
171  while (!have_stop_request()) {
172  // Allocate buffer for reading
173  char* buffer = new char[block_buffer_size];
174  size_t bytes_read = read_block_from_pipe(buffer);
175 
176  if (bytes_read > 0) {
177  // Add block to queue if data was read
178  m.lock();
179  blocks.push_back({buffer});
180  m.unlock();
181  }
182  else {
183  // No data, check for termination
184  bool done;
185  m.lock();
186  done = all_data_received;
187  m.unlock();
188  delete[] buffer;
189  if (done)
190  break;
191  wait(ms_to_wait);
192  }
193  }
194 
195  // Clean up and close source
196  close();
197 }
