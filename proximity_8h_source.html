121 {
122 T l = dot(cd, cd);
123 T a = dot(p - co, cd) / l;
124 l = std::sqrt(l);
125 fvec<T, 3> q0 = co + a * cd;
126 T r0 = (p - q0).length();
127 if (a < 0) {
128 if (r0 > cr - a * l)
129 n = (1 / r0) * (p - q0);
130 else
131 n = -cd / l;
132 if (r0 < cr)
133 q = co + p - q0;
134 else
135 q = co + cr / r0 * (p - q0);
136 }
137 else if (a > 1) {
138 if (r0 > cr + (a - 1) * l)
139 n = (1 / r0) * (p - q0);

143 q = co + cd + p - q0;
