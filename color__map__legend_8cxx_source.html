88 void color_map_legend::draw_content(cgv::render::context& ctx) {
89 
90  begin_content(ctx);
91 
92  // draw inner border
93  content_canvas.enable_shader(ctx, "rectangle");
94  content_canvas.set_style(ctx, border_style);
95  content_canvas.draw_shape(ctx, layout.color_map_rect.position - 1, layout.color_map_rect.size + 2);
96 
97  // draw background grid as contrast for transparent color maps or indicator that no color map is set
98  content_canvas.enable_shader(ctx, "grid");
99  content_canvas.set_style(ctx, background_style);
100  content_canvas.draw_shape(ctx, layout.color_map_rect);
101 
102  if(tex.is_created()) {
103  // draw the color map texture
104  content_canvas.push_modelview_matrix();
105  ivec2 pos = layout.color_map_rect.position;
106  ivec2 size = layout.color_map_rect.size;
107  float angle = 0.0f;
108 
109  if(layout.orientation == OO_VERTICAL) {
110  pos.x() += layout.color_map_rect.size.x();
111  std::swap(size.x(), size.y());
112  angle = 90.0f;
113  }
114 
115  content_canvas.mul_modelview_matrix(ctx, cgv::math::translate2h(pos));
116  content_canvas.mul_modelview_matrix(ctx, cgv::math::rotate2h(angle));
117 
118  // draw color scale texture
119  color_map_style.use_texture_alpha = show_opacity;
120 
121  color_map_style.texcoord_offset.x() = display_range[0];
122  color_map_style.texcoord_scaling.x() = display_range[1] - display_range[0];
123 
124  if(flip_texture)
125  color_map_style.texcoord_scaling.x() *= -1.0f;
126 
127  content_canvas.enable_shader(ctx, "rectangle");
128  content_canvas.set_style(ctx, color_map_style);
129  tex.enable(ctx, 0);
130  content_canvas.draw_shape(ctx, ivec2(0), size);
131  tex.disable(ctx);
132 
133  content_canvas.pop_modelview_matrix(ctx);
134  }
135  content_canvas.disable_current_shader(ctx);
136 
137  // draw tick marks
138  tick_renderer.render(ctx, content_canvas, cgv::render::PT_POINTS, ticks, tick_style);
139 
140  // draw tick labels
141  auto& font_renderer = cgv::g2d::ref_msdf_gl_canvas_font_renderer(ctx);
142  if(font_renderer.enable(ctx, content_canvas, labels, text_style)) {
143  font_renderer.draw(ctx, labels, 0, int(labels.size()) - 1);
144 
145  content_canvas.push_modelview_matrix();
146  content_canvas.mul_modelview_matrix(ctx, cgv::math::translate2h(layout.title_position));
147  content_canvas.mul_modelview_matrix(ctx, cgv::math::rotate2h(layout.title_angle));
148 
149  font_renderer.draw(ctx, content_canvas, labels, labels.size() - 1, 1);
150 
151  content_canvas.pop_modelview_matrix(ctx);

179 void color_map_legend::set_color_map(cgv::render::context& ctx, const cgv::render::color_map& cm) {
180 
181  cgv::render::TextureFilter filter = cgv::render::TF_LINEAR;
182  if(cm.has_texture_support()) {
183  const cgv::render::gl_color_map* gl_cm_ptr = dynamic_cast<const cgv::render::gl_color_map*>(&cm);
184  filter = gl_cm_ptr->is_linear_filtering_enabled() ? cgv::render::TF_LINEAR : cgv::render::TF_NEAREST;
185  }
186 
187  unsigned resolution = cm.get_resolution();
188  std::vector<rgb> color_data = cm.interpolate_color(static_cast<size_t>(resolution));
189  std::vector<float> opacity_data(static_cast<size_t>(resolution), 1.0f);
190 
191  if(!cm.ref_opacity_points().empty())
192  opacity_data = cm.interpolate_opacity(static_cast<size_t>(resolution));
193 
194  std::vector<uint8_t> data_8(2 * 4 * color_data.size());
195  for(unsigned i = 0; i < color_data.size(); ++i) {
196  rgba col = color_data[i];
197  data_8[4 * i + 0] = static_cast<uint8_t>(255.0f * col.R());
198  data_8[4 * i + 1] = static_cast<uint8_t>(255.0f * col.G());
199  data_8[4 * i + 2] = static_cast<uint8_t>(255.0f * col.B());
200  data_8[4 * i + 3] = static_cast<uint8_t>(255.0f * opacity_data[i]);
201  }
202 
203  std::copy(data_8.begin(), data_8.begin() + 4 * resolution, data_8.begin() + 4 * resolution);
204 
205  cgv::data::data_view dv = cgv::data::data_view(new cgv::data::data_format(resolution, 2u, TI_UINT8, cgv::data::CF_RGBA), data_8.data());
206 
207  unsigned width = (unsigned)tex.get_width();
208 
209  bool replaced = false;
210  if(tex.is_created() && width == resolution && tex.get_nr_components() == 4) {
211  tex.set_min_filter(filter);
212  tex.set_mag_filter(filter);
213  replaced = tex.replace(ctx, 0, 0, dv);
214  }
215 
216  if(!replaced) {
217  tex.destruct(ctx);
218  tex = cgv::render::texture("uint8[R,G,B,A]", filter, filter);
219  tex.create(ctx, dv, 0);
220  }
221 
222  post_damage();
223 }

323 void color_map_legend::create_labels() {
324 
325  labels.clear();
326 
327  if(layout.label_alignment == AO_FREE)
328  return;
329 
330  unsigned precision = label_format.precision;
331 
332  if(label_format.auto_precision) {
333  precision = 0;
334  const float delta = std::abs(value_range[1] - value_range[0]);
335  const unsigned max_precision = 7;
336 
337  if(delta > 5.0f) {
338  precision = 1;
339  } else {
340  float limit = 1.0f;
341  for(unsigned i = 2; i <= max_precision; ++i) {
342  if(delta > limit || i == max_precision) {
343  precision = i;
344  break;
345  }
346  limit /= 2.0f;
347  }
348  }
349  }
350 
351  float max_length = -1.0f;
352 
353  for(size_t i = 0; i < num_ticks; ++i) {
354  float fi = static_cast<float>(i);
355  float t = fi / static_cast<float>(num_ticks - 1);
356  float val = cgv::math::lerp(value_range.x(), value_range.y(), t);
357 
358  std::string str;
359 
360  if(label_format.integers)
361  str = std::to_string(static_cast<int>(round(val)));
362  else {
363  str = cgv::utils::to_string(val, -1, precision, true);
364  if(!label_format.trailing_zeros && str.length() > 1)
365  cgv::utils::rtrim(cgv::utils::rtrim(str, "0"), ".");
366  }
367 
368  labels.add_text(str, ivec2(0), cgv::render::TextAlignment::TA_NONE);
369  max_length = std::max(max_length, labels.ref_texts().back().size.x());
370  }
371 
372  if(labels.size() > 1) {
373  if(layout.orientation == OO_HORIZONTAL)
374  layout.x_label_size = static_cast<int>(std::max(labels.ref_texts().front().size.x(), labels.ref_texts().back().size.x()) * text_style.font_size);
375  else
376  layout.x_label_size = static_cast<int>(max_length * text_style.font_size);
377  } else {
378  layout.x_label_size = 0;
379  }
380 
381  labels.add_text(title, ivec2(0), cgv::render::TextAlignment::TA_BOTTOM_LEFT);
382 }

384 void color_map_legend::create_ticks() {
385 
386  ticks.clear();
387 
388  if(layout.label_alignment == AO_FREE)
389  return;
390 
391  ivec2 tick_size(1, 6);
392 
393  int axis = 0;
394  int label_offset = 4;
395  AlignmentOption label_alignment = layout.label_alignment;
396  cgv::render::TextAlignment text_h_start = cgv::render::TextAlignment::TA_LEFT;
397  cgv::render::TextAlignment text_h_end = cgv::render::TextAlignment::TA_RIGHT;
398  cgv::render::TextAlignment text_v_start = cgv::render::TextAlignment::TA_TOP;
399  cgv::render::TextAlignment text_v_end = cgv::render::TextAlignment::TA_BOTTOM;
400  cgv::render::TextAlignment title_alignment_1 = cgv::render::TextAlignment::TA_TOP;
401  cgv::render::TextAlignment title_alignment_2 = cgv::render::TextAlignment::TA_BOTTOM;
402 
403  layout.title_angle = 0.0f;
404 
405  if(layout.orientation == OO_VERTICAL) {
406  axis = 1;
407  label_offset = 6;
408 
409  std::swap(tick_size.x(), tick_size.y());
410 
411  if(label_alignment == AO_START) label_alignment = AO_END;
412  else if(label_alignment == AO_END) label_alignment = AO_START;
413 
414  text_h_start = cgv::render::TextAlignment::TA_BOTTOM;
415  text_h_end = cgv::render::TextAlignment::TA_TOP;
416  text_v_start = cgv::render::TextAlignment::TA_RIGHT;
417  text_v_end = cgv::render::TextAlignment::TA_LEFT;
418  std::swap(title_alignment_1, title_alignment_2);
419 
420  layout.title_angle = 90.0f;
421  }
422 
423  ivec2 color_rect_pos = layout.color_map_rect.position;
424  ivec2 color_rect_size = layout.color_map_rect.size;
425 
426  int length = color_rect_size[axis];
427  float step = static_cast<float>(length + 1) / static_cast<float>(num_ticks - 1);
428 
429  ivec2 title_pos = color_rect_pos;
430  ivec2 tick_start = color_rect_pos;
431 
432  cgv::render::TextAlignment title_alignment, text_alignment;
433  title_alignment = title_alignment_1;
434  text_alignment = text_v_end;
435 
436  bool inside = false;
437  switch(label_alignment) {
438  case AO_START:
439  title_pos[1 - axis] -= 4;
440  tick_start[1 - axis] += color_rect_size[1 - axis] + 3;
441  break;
442  case AO_CENTER:
443  title_pos[axis] += 2;
444  title_pos[1 - axis] += color_rect_size[1 - axis] - (axis ? 3 : 1);
445  tick_start[1 - axis] += 3;
446  inside = true;
447  break;
448  case AO_END:
449  title_alignment = title_alignment_2;
450  title_pos[1 - axis] += color_rect_size[1 - axis] + 4;
451  tick_start[1 - axis] -= 3;
452  label_offset = -label_offset;
453  text_alignment = text_v_start;
454  break;
455  default: break;
456  }
457 
458  title_alignment = static_cast<cgv::render::TextAlignment>(title_alignment + cgv::render::TextAlignment::TA_LEFT);
459 
460  for(size_t i = 0; i < num_ticks; ++i) {
461  float fi = static_cast<float>(i);
462  int offset = static_cast<int>(round(fi * step));
463 
464  ivec2 tick_pos = tick_start;
465  tick_pos[axis] += offset;
466 
467  ivec2 label_pos = tick_start;
468  label_pos[1 - axis] += label_offset;
469  label_pos[axis] += offset;
470 
471  cgv::render::TextAlignment alignment = text_alignment;
472 
473  if(inside) {
474  if(i == 0) {
475  label_pos[axis] += 3;
476  alignment = static_cast<cgv::render::TextAlignment>(alignment | text_h_start);
477  } else if(i == num_ticks - 1) {
478  label_pos[axis] -= 3;
479  alignment = static_cast<cgv::render::TextAlignment>(alignment | text_h_end);
480  }
481  }
482 
483  ticks.add(tick_pos, tick_size);
484 
485  labels.set_position(int(i), label_pos);
486  labels.set_alignment(int(i), alignment);
487  }
488 
489  layout.title_position = title_pos;
490 
491  if(labels.size() > 0)
492  labels.set_alignment(unsigned(labels.ref_texts().size() - 1), title_alignment);
493 }
494 
495 }
