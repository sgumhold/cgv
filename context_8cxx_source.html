30std::ostream& operator << (std::ostream& os, const type_descriptor& td)
31{
32 std::string prefix, postfix, type;
33 switch (td.coordinate_type) {
34 case cgv::type::info::TI_BOOL: type = "bool"; prefix = "b"; break;
35 case cgv::type::info::TI_FLT32: type = "float"; break;
36 case cgv::type::info::TI_FLT64: type = "double"; prefix = "d"; break;
37 case cgv::type::info::TI_INT32: type = "int"; prefix = "i"; break;
38 case cgv::type::info::TI_UINT32: type = "uint"; prefix = "u"; break;
39 }
40 switch (td.element_type) {
41 case cgv::render::ElementType::ET_VALUE:
42 prefix = "";
43 break;
44 case cgv::render::ElementType::ET_VECTOR:
45 type = "vec";
46 postfix = "0";
47 postfix[0] += td.nr_rows;
48 break;
49 case cgv::render::ElementType::ET_MATRIX: type = "mat"; break;
50 type = "mat";
51 postfix = "0";
52 postfix[0] += td.nr_rows;
53 if (td.nr_columns != td.nr_rows) {
54 postfix += "x0";
55 postfix[2] += td.nr_columns;
56 }
57 break;
58 }
59 return os << prefix << type << postfix;

62void program_variable_info::compute_sizes(size_t& cnt, size_t& s, size_t& S) const
63{
64 cnt = 1;
65 if (type_descr.element_type == cgv::render::ET_VECTOR)
66 cnt = type_descr.nr_rows;
67 if (type_descr.element_type == cgv::render::ET_MATRIX)
68 cnt = type_descr.nr_rows * type_descr.nr_columns;
69 unsigned ctype_size = cgv::type::info::get_type_size(type_descr.coordinate_type);
70 if (type_descr.coordinate_type == cgv::type::info::TI_BOOL)
71 ctype_size = 4;
72 s = ctype_size * cnt;
73 S = s * array_size;
74}

77std::ostream& operator << (std::ostream& os, const program_variable_info& V)
78{
79 os << V.type_descr << " " << V.name;
80 if (V.array_size > 1)
81 os << "[" << V.array_size << "]";
82 if (V.program_location != -1)
83 os << " @" << V.program_location;
84 if (V.current_value.empty())

86

88 size_t cnt, s, S;
89 V.compute_sizes(cnt, s, S);
90 if (V.array_size > 1)
91 os << "{";
92 for (size_t j = 0; j < V.array_size; ++j) {
93 if (j > 0)
94 os << ", ";
95 if (cnt > 1)
96 os << "[";
97 for (size_t k = 0; k < cnt; ++k) {
98 if (k > 0)
99 os << ", ";
100 const void* value_ptr = V.current_value.data() + j * s;
101 switch (V.type_descr.coordinate_type) {
102 case cgv::type::info::TI_BOOL: os << (*(const int32_t*)(value_ptr) != 0 ? "true" : "false"); break;
103 case cgv::type::info::TI_INT32: os << *(const int32_t*)(value_ptr); break;
104 case cgv::type::info::TI_UINT32: os << *(const uint32_t*)(value_ptr); break;

106 case cgv::type::info::TI_FLT64: os << *(const double*)(value_ptr); break;
107 }

601void* context::add_light_source(const cgv::media::illum::light_source& light, bool enabled, bool place_now)
602{
603 // construct new light source handle
604 ++light_source_handle;
605 void* handle = reinterpret_cast<void*>(light_source_handle);
606 // determine light source position
607 vec3 Le = get_light_eye_position(light, place_now);

609 //

614 }
615 // store new light source in map
616 light_sources[handle] = std::pair<cgv::media::illum::light_source, light_source_status>(

625bool context::remove_light_source(void* handle)
626{
627 // find handle in map
628 auto iter = light_sources.find(handle);
629 if (iter == light_sources.end())
630 return false;
631 // check if light source was enabled
632 if (iter->second.second.enabled) {
633 // then remove from list of enabled light sources
634 enabled_light_source_handles.erase(enabled_light_source_handles.begin() + iter->second.second.light_source_index);
635 // and correct indices of moved light sources
636 for (int i = iter->second.second.light_source_index; i < (int)enabled_light_source_handles.size(); ++i)
637 light_sources[enabled_light_source_handles[i]].second.light_source_index = i;
638 on_lights_changed();
639 }
640 // remove from map
641 light_sources.erase(iter);
642 return true;
643}

901void context::render_pass(RenderPass rp, RenderPassFlags rpf, void* user_data, int rp_idx)
902{
903 // ensure that default light sources are created
904 if (default_light_source_handles[0] == 0) {
905 for (unsigned i=0; i<nr_default_light_sources; ++i)
906 default_light_source_handles[i] = add_light_source(default_light_source[i], false, false);
907 }
908 render_info ri;
909 ri.pass = rp;
910 ri.flags = rpf;
911 ri.user_data = user_data;
912 ri.pass_index = rp_idx;
913 render_pass_stack.push(ri);
914 render_pass_debug_output(ri, "init");
915 init_render_pass();
916 if (get_render_pass_flags()&RPF_SET_LIGHTS) {
917 for (unsigned i = 0; i < nr_default_light_sources; ++i)
918 place_light_source(default_light_source_handles[i]);
919 }
920
921 group* grp = dynamic_cast<group*>(this);
922 if (grp && (rpf&RPF_DRAWABLES_DRAW)) {
923 render_pass_debug_output(ri, "draw+finish_draw");
924 matched_method_action<drawable,void,void,context&>
925 mma(*this, &drawable::draw, &drawable::finish_draw, true, true);
926 traverser(mma).traverse(group_ptr(grp));
927 }
928 if (rpf & RPF_DRAW_TEXTUAL_INFO) {
929 render_pass_debug_output(ri, "textual_info");
930 draw_textual_info();
931 }
932 if (grp && (rpf&RPF_DRAWABLES_FINISH_FRAME)) {
933 render_pass_debug_output(ri, "finish_frame");
934 single_method_action<drawable,void,context&>
935 sma(*this, &drawable::finish_frame, true, true);
936 traverser(sma).traverse(group_ptr(grp));
937 }
938 if (grp && (rpf&RPF_DRAWABLES_AFTER_FINISH)) {
939 render_pass_debug_output(ri, "after_finish");
940 single_method_action<drawable,void,context&>
941 sma(*this, &drawable::after_finish, true, true);
942 traverser(sma).traverse(group_ptr(grp));

944 if ((rpf&RPF_HANDLE_SCREEN_SHOT) && do_screen_shot) {
945 render_pass_debug_output(ri, "screenshot");
946 perform_screen_shot();
947 do_screen_shot = false;

949 render_pass_debug_output(ri, "finish render pass");

955void context::process_text(const std::string& text)
956{
957 push_pixel_coords();
958 unsigned int i, j = 0;
959 for (i = 0; i<text.size(); ++i) {
960 int n = i-j;
961 switch (text[i]) {
962 case '\a' :
963 draw_text(text.substr(j,n));
964 ++nr_identations;
965 if (at_line_begin)
966 cursor_x += (int)(tab_size*current_font_size);
967 j = i+1;
968 break;
969 case '\b' :
970 draw_text(text.substr(j,n));
971 if (nr_identations > 0) {
972 --nr_identations;
973 if (at_line_begin)
974 cursor_x -= (int)(tab_size*current_font_size);
975 }
976 j = i+1;
977 break;
978 case '\t' :
979 draw_text(text.substr(j,n));
980 cursor_x = (((cursor_x-x_offset)/(int)(tab_size*current_font_size))+1)*(int)(tab_size*current_font_size)+x_offset;
981 at_line_begin = false;
982 j = i+1;
983 break;
984 case '\n' :
985 draw_text(text.substr(j,n));
986 cursor_x = x_offset+(int)(nr_identations*tab_size*current_font_size);
987 cursor_y -= (int)(1.2f*current_font_size);
988 at_line_begin = true;
989 j = i+1;
990 break;
991 default:
992 at_line_begin = false;
993 }
994 }
995 draw_text(text.substr(j,i-j));
996 pop_pixel_coords();
997}

1134void compute_face_normals(const float* vertices, float* normals, const int* vertex_indices, int* normal_indices, int nr_faces, int face_degree)
1135{
1136 for (int i = 0; i < nr_faces; ++i) {
1137 vec3& normal = reinterpret_cast<vec3&>(normals[3 * i]);
1138 normal.zeros();
1139 vec3 reference_pnt = *reinterpret_cast<const vec3*>(vertices + 3 * vertex_indices[face_degree*i + face_degree - 1]);
1140 vec3 last_difference;
1141 last_difference.zeros();
1142 for (int j = 0; j < face_degree; ++j) {
1143 vec3 new_difference = *reinterpret_cast<const vec3*>(vertices + 3 * vertex_indices[face_degree*i + j]) - reference_pnt;
1144 normal += cross(last_difference, new_difference);
1145 last_difference = new_difference;
1146 }
1147 normal.normalize();
1148 for (int j = 0; j<face_degree; ++j)
1149 normal_indices[face_degree*i+j] = i;
1150 }
1151}
