160  bool rectangle_renderer::enable(cgv::render::context& ctx)
161  {
162  if(!surface_renderer::enable(ctx))
163  return false;
164  const rectangle_render_style& rrs = get_style<rectangle_render_style>();
165  if (!has_depth_offsets)
166  ref_prog().set_attribute(ctx, "depth_offset", rrs.default_depth_offset);
167  int sci = ref_prog().get_attribute_location(ctx, "secondary_color");
168  if(!has_secondary_colors && sci != -1)
169  ref_prog().set_attribute(ctx, "secondary_color", rrs.default_secondary_color);
170  if (!has_border_colors)
171  ref_prog().set_attribute(ctx, "border_color", rrs.default_border_color);
172  if (!has_border_infos)
173  ref_prog().set_attribute(ctx, "border_info", vec3(rrs.border_width_in_pixel, rrs.percentual_border_width, float(rrs.border_mode)));
174  // configure opengl
175  if (rrs.blend_rectangles) {
176  rrs.is_blend = glIsEnabled(GL_BLEND);
177  glGetIntegerv(GL_BLEND_DST, reinterpret_cast<GLint*>(&rrs.blend_dst));
178  glGetIntegerv(GL_BLEND_SRC, reinterpret_cast<GLint*>(&rrs.blend_src));
179  glEnable(GL_BLEND);
180  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
181  }
182  ref_prog().set_uniform(ctx, "has_rotations", has_rotations);
183  ref_prog().set_uniform(ctx, "has_translations", has_translations);
184  ref_prog().set_uniform(ctx, "position_is_center", get_style<rectangle_render_style>().position_is_center);
185  ref_prog().set_uniform(ctx, "use_texture", has_texcoords);
186  ref_prog().set_uniform(ctx, "viewport_height", (float)ctx.get_height());
187  ref_prog().set_uniform(ctx, "texture_mode", (const int&)rrs.texture_mode);
188  ref_prog().set_uniform(ctx, "pixel_blend", rrs.pixel_blend);
189  ref_prog().set_uniform(ctx, "border_width_in_pixel", rrs.border_width_in_pixel);
190  ref_prog().set_uniform(ctx, "percentual_border_width", rrs.percentual_border_width);
191  ref_prog().set_uniform(ctx, "border_mode", (const int&)rrs.border_mode);
192  return true;
193  }
194  bool rectangle_renderer::disable(cgv::render::context& ctx)
195  {
196  if(!attributes_persist()) {
197  has_extents = false;
198  has_secondary_colors = false;
199  has_border_colors = false;
200  has_border_infos = false;
201  has_rotations = false;
202  has_translations = false;
203  has_depth_offsets = false;
204  }
205  const rectangle_render_style& rrs = get_style<rectangle_render_style>();
206  if (rrs.blend_rectangles) {
207  if (!rrs.is_blend)
208  glDisable(GL_BLEND);
209  glBlendFunc(rrs.blend_src, rrs.blend_dst);
210  }
211  return surface_renderer::disable(ctx);
212  }
