47 bool line_renderer::init(context& ctx)
48 {
49 bool res = renderer::init(ctx);
50 // FIXME: Normally we could use the shader_program::set_attribute(const context& ctx, const std::string& name, const T& value) overload
51 // to write this more cleanly but this will print an error if the attribute is not found. Under normal circumstances this would never happen,
52 // but some derived classes like the box_wire_renderer and the normal renderer do not use all the lne_renderere attributes and so the error
53 // would appear.
54 int li;
55 li = get_prog_attribute_location(ctx, "normal", false);
56 if(li != -1)
57 ref_prog().set_attribute(ctx, li, vec3(0.0f, 0.0f, 1.0f));
58 li = get_prog_attribute_location(ctx, "depth_offset", false);
59 if(li != -1)
60 ref_prog().set_attribute(ctx, li, 0.0f);
61 li = get_prog_attribute_location(ctx, "line_width", false);
62 if(li != -1)
63 ref_prog().set_attribute(ctx, li, 1.0f);
64 return res;

67 bool line_renderer::enable(context& ctx)
68 {
69 const line_render_style& lrs = get_style<line_render_style>();
70 if (!group_renderer::enable(ctx))
71 return false;
72 // set program attributes
73 if (!has_colors)
74 ctx.set_color(lrs.default_color);
75 if(!has_normals) {
76 int li = get_prog_attribute_location(ctx, "normal", false);
77 if(li != -1)
78 ref_prog().set_attribute(ctx, li, lrs.default_normal);
79 }
80 if(!has_depth_offsets) {
81 int li = get_prog_attribute_location(ctx, "depth_offset", false);
82 if(li != -1)
83 ref_prog().set_attribute(ctx, li, lrs.default_depth_offset);
84 }
85 if(!has_line_widths) {
86 int li = get_prog_attribute_location(ctx, "line_width", false);
87 if(li != -1)
88 ref_prog().set_attribute(ctx, li, lrs.default_line_width);
89 }
90 // configure blend state
91 if (lrs.blend_lines) {
92 ctx.push_blend_state();
93 ctx.get_blend_state();
94 ctx.enable_blending();
95 ctx.set_blend_func(BF_SRC_ALPHA, BF_ONE_MINUS_SRC_ALPHA);
96 }
97
98 // set program uniforms
99 ref_prog().set_uniform(ctx, "halo_color", lrs.halo_color);
100 ref_prog().set_uniform(ctx, "halo_width_in_pixel", lrs.halo_width_in_pixel);
101 ref_prog().set_uniform(ctx, "percentual_halo_width", 0.01f * lrs.percentual_halo_width);
102 ref_prog().set_uniform(ctx, "screen_aligned", lrs.screen_aligned);
103 ref_prog().set_uniform(ctx, "measure_line_width_in_pixel", lrs.measure_line_width_in_pixel);
104 ref_prog().set_uniform(ctx, "reference_line_width", lrs.reference_line_width);
105 ref_prog().set_uniform(ctx, "blend_width_in_pixel", lrs.blend_width_in_pixel);
106 ref_prog().set_uniform(ctx, "viewport_height", (float)ctx.get_height());
107 ref_prog().set_uniform(ctx, "halo_color_strength", lrs.halo_color_strength);
108 return true;
109 }
