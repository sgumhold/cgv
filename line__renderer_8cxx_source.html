84  bool line_renderer::enable(context& ctx)
85  {
86  const line_render_style& lrs = get_style<line_render_style>();
87  if (!group_renderer::enable(ctx))
88  return false;
89  // set program attributes
90  if (!has_colors)
91  ctx.set_color(lrs.default_color);
92  if (!has_normals) {
93  int li = get_prog_attribute_location(ctx, "normal", false);
94  if (li != -1)
95  ref_prog().set_attribute(ctx, li, lrs.default_normal);
96  }
97  if (!has_depth_offsets) {
98  int li = get_prog_attribute_location(ctx, "depth_offset", false);
99  if (li != -1)
100  ref_prog().set_attribute(ctx, li, lrs.default_depth_offset);
101  }
102  if (!has_line_widths) {
103  int li = get_prog_attribute_location(ctx, "line_width", false);
104  if (li != -1)
105  ref_prog().set_attribute(ctx, li, lrs.default_line_width);
106  }
107  // configure opengl
108  if (lrs.blend_lines) {
109  lrs.is_blend = glIsEnabled(GL_BLEND);
110  glGetIntegerv(GL_BLEND_DST, reinterpret_cast<GLint*>(&lrs.blend_dst));
111  glGetIntegerv(GL_BLEND_SRC, reinterpret_cast<GLint*>(&lrs.blend_src));
112  glEnable(GL_BLEND);
113  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
114  }
115 
116  // set program uniforms
117  ref_prog().set_uniform(ctx, "halo_color", lrs.halo_color);
118  ref_prog().set_uniform(ctx, "halo_width_in_pixel", lrs.halo_width_in_pixel);
119  ref_prog().set_uniform(ctx, "percentual_halo_width", 0.01f * lrs.percentual_halo_width);
120  ref_prog().set_uniform(ctx, "screen_aligned", lrs.screen_aligned);
121  ref_prog().set_uniform(ctx, "measure_line_width_in_pixel", lrs.measure_line_width_in_pixel);
122  ref_prog().set_uniform(ctx, "reference_line_width", lrs.reference_line_width);
123  ref_prog().set_uniform(ctx, "blend_width_in_pixel", lrs.blend_width_in_pixel);
124  ref_prog().set_uniform(ctx, "viewport_height", (float)ctx.get_height());
125  ref_prog().set_uniform(ctx, "halo_color_strength", lrs.halo_color_strength);
126  return true;
127  }
