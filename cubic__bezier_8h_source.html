83 template<typename T>
84 std::pair<T, T> compute_bounds(const T& p0, const T& p1, const T& p2, const T& p3) const{
85 T mi = std::min(p0, p3);
86 T ma = std::max(p0, p3);
87
88 T c = T(-1) * p0 + T(1) * p1;
89 T b = T(1) * p0 - T(2) * p1 + T(1) * p2;
90 T a = T(-1) * p0 + T(3) * p1 - T(3) * p2 + T(1) * p3;
91
92 T h = b * b - a * c;
93
94 if(h > T(0)) {
95 h = sqrt(h);
96 T t = (-b - h) / a;
97 if(t > T(0) && t < T(1)) {
98 T s = T(1) - t;
99 T q = s * s * s * p0 + T(3) * s * s * t * p1 + T(3) * s * t * t * p2 + t * t * t * p3;
100 mi = std::min(mi, q);
101 ma = std::max(ma, q);
102 }
103 t = (-b + h) / a;
104 if(t > T(0) && t < T(1)) {
105 T s = T(1) - t;
106 T q = s * s * s * p0 + T(3) * s * s * t * p1 + T(3) * s * t * t * p2 + t * t * t * p3;
107 mi = std::min(mi, q);
108 ma = std::max(ma, q);
109 }
110 }
111
112 return { mi, ma };
113 }
