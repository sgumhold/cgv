13  T ray_cylinder_intersection(const fvec<T, 3>& ro, const fvec<T, 3>& rd, const fvec<T, 3>& pa, const fvec<T, 3>& ca, T ra, fvec<T, 3>& n)

15  fvec<T, 3> oc = ro - pa;

17  T card = dot(ca, rd);
18  T caoc = dot(ca, oc);
19  T a = caca - card * card;
20  T b = caca * dot(oc, rd) - caoc * card;
21  T c = caca * dot(oc, oc) - caoc * caoc - ra * ra * caca;
22  T h = b * b - a * c;
23  if (h < 0.0)
24  return std::numeric_limits<T>::max();
25  h = sqrt(h);
26  T t = (-b - h) / a;
27  // body
28  T y = caoc + t * card;
29  if (y > 0.0f && y < caca) {
30  n = (oc + t * rd - ca * y / caca) / ra;
31  return t;
32  }
33  // caps
34  t = (((y < 0.0f) ? 0.0f : caca) - caoc) / card;
35  if (std::abs(b + a * t) < h) {
36  n = ca * sign(y) / caca;
37  return t;
38  }
39  return std::numeric_limits<T>::max();
40  }

42  T ray_torus_intersection(const fvec<T, 3>& ro, const fvec<T, 3>& rd, const fvec<T, 2> tor, fvec<T, 3>& n0)
43  {
44  //T torIntersect(in fvec<T,3> ro, in fvec<T,3> rd, in fvec<T,2> tor)
45  T po = 1.0;
46  T Ra2 = tor.x() * tor.x();
47  T ra2 = tor.y() * tor.y();
48  T m = dot(ro, ro);
49  T n = dot(ro, rd);
50  T k = (m + Ra2 - ra2) / 2.0f;
51  T k3 = n;
52  const fvec<T, 2>& ro_xy = reinterpret_cast<const fvec<T, 2>&>(ro);
53  const fvec<T, 2>& rd_xy = reinterpret_cast<const fvec<T, 2>&>(rd);
54  T k2 = n * n - Ra2 * dot(rd_xy, rd_xy) + k;
55  T k1 = n * k - Ra2 * dot(rd_xy, ro_xy);
56  T k0 = k * k - Ra2 * dot(ro_xy, ro_xy);
57 
58  if (std::abs(k3 * (k3 * k3 - k2) + k1) < 0.01f)
59  {
60  po = -1.0f;
61  T tmp = k1; k1 = k3; k3 = tmp;
62  k0 = 1.0f / k0;
63  k1 = k1 * k0;
64  k2 = k2 * k0;
65  k3 = k3 * k0;
66  }
67 
68  T c2 = k2 * 2.0f - 3.0f * k3 * k3;
69  T c1 = k3 * (k3 * k3 - k2) + k1;
70  T c0 = k3 * (k3 * (c2 + 2.0f * k2) - 8.0f * k1) + 4.0f * k0;
71  c2 /= 3.0f;
72  c1 *= 2.0f;
73  c0 /= 3.0f;
74  T Q = c2 * c2 + c0;
75  T R = c2 * c2 * c2 - 3.0f * c2 * c0 + c1 * c1;
76  T h = R * R - Q * Q * Q;
77 
78  if (h >= 0.0)
79  {
80  h = sqrt(h);
81  T v = sign(R + h) * std::pow(std::abs(R + h), 1.0f / 3.0f); // cube root
82  T u = sign(R - h) * std::pow(std::abs(R - h), 1.0f / 3.0f); // cube root
83  fvec<T, 2> s = fvec<T, 2>((v + u) + 4.0f * c2, (v - u) * sqrt(3.0f));
84  T y = sqrt(0.5f * (length(s) + s.x()));
85  T x = 0.5f * s.y() / y;
86  T r = 2.0f * c1 / (x * x + y * y);
87  T t1 = x - r - k3; t1 = (po < 0.0f) ? 2.0f / t1 : t1;
88  T t2 = -x - r - k3; t2 = (po < 0.0f) ? 2.0f / t2 : t2;
89  T t = 1e20f;
90  if (t1 > 0.0f) t = t1;
91  if (t2 > 0.0f) t = std::min(t, t2);
92  fvec<T, 3> pos = ro + t * rd;
93  n0 = normalize(pos * ((dot(pos, pos) - ra2) * fvec<T, 3>(1, 1, 1) - Ra2 * fvec<T, 3>(1, 1, -1)));
94  return t;
95  }
96 
97  T sQ = sqrt(Q);
98  T w = sQ * cos(acos(-R / (sQ * Q)) / 3.0f);
99  T d2 = -(w + c2);
100  if (d2 < 0.0f)
101  return std::numeric_limits<T>::max();
102  T d1 = sqrt(d2);
103  T h1 = sqrt(w - 2.0f * c2 + c1 / d1);
104  T h2 = sqrt(w - 2.0f * c2 - c1 / d1);
105  T t1 = -d1 - h1 - k3; t1 = (po < 0.0f) ? 2.0f / t1 : t1;
106  T t2 = -d1 + h1 - k3; t2 = (po < 0.0f) ? 2.0f / t2 : t2;
107  T t3 = d1 - h2 - k3; t3 = (po < 0.0f) ? 2.0f / t3 : t3;
108  T t4 = d1 + h2 - k3; t4 = (po < 0.0f) ? 2.0f / t4 : t4;
109  T t = 1e20f;
110  if (t1 > 0.0f) t = t1;
111  if (t2 > 0.0f) t = std::min(t, t2);
112  if (t3 > 0.0f) t = std::min(t, t3);
113  if (t4 > 0.0f) t = std::min(t, t4);
114  fvec<T, 3> pos = ro + t * rd;
115  n0 = normalize(pos * ((dot(pos, pos) - ra2) * fvec<T, 3>(1, 1, 1) - Ra2 * fvec<T, 3>(1, 1, -1)));
116  return t;
117  }
