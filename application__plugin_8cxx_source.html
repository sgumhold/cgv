16 bool application_plugin_base::handle(cgv::gui::event& e)
17 {
18  if(e.get_kind() == cgv::gui::EID_MOUSE) {
19  // Perform special handling for mouse events.
20  cgv::gui::mouse_event& me = (cgv::gui::mouse_event&) e;
21  cgv::gui::MouseAction ma = me.get_action();
22 
23  // Drag & Drop events are never handled by overlays.
24  if(me.get_flags() & cgv::gui::EF_DND)
25  return handle_event(e);
26 
27  // Only mouse MA_PRESS, MA_MOVE or MA_WHEEL events can trigger an overlay to capture an event.
28  if(ma == cgv::gui::MA_PRESS || ma == cgv::gui::MA_MOVE || ma == cgv::gui::MA_WHEEL) {
29  ivec2 mpos(me.get_x(), me.get_y());
30 
31  // Test for all visible overlays in reverse registration order if they are hit by the current
32  // mouse pointer position and store the first hit overlay as the active blocking overlay.
33  blocking_overlay_ptr = nullptr;
34  for(auto it = overlays.rbegin(); it != overlays.rend(); ++it) {
35  overlay_ptr op = (*it);
36  if(op->is_visible() && op->is_hit(mpos)) {
37  blocking_overlay_ptr = op;
38  break;
39  }
40  }
41 
42  // Test if the mouse pointer enters or leaves the active overlay, which may issue additional event handling.
43  if(ma == cgv::gui::MA_MOVE && blocking_overlay_ptr != last_blocking_overlay_ptr) {
44  if(last_blocking_overlay_ptr) {
45  // The mouse pointer leaves the active overlay. Handle it as an additional MA_LEAVE event.
46  me.set_action(cgv::gui::MA_LEAVE);
47  last_blocking_overlay_ptr->handle_event(e);
48  }
49  if(blocking_overlay_ptr) {
50  // The mouse pointer enters the active overlay. Handle it as an additional MA_ENTER event.
51  me.set_action(cgv::gui::MA_ENTER);
52  if(!blocking_overlay_ptr->handle_event(e))
53  blocking_overlay_ptr = nullptr;
54  }
55  // Revert the event action to MA_MOVE.
56  me.set_action(cgv::gui::MA_MOVE);
57  }
58  }
59 
60  bool was_handled = false;
61  bool was_blocked = false;
62  bool result = false;
63 
64  // Call handle_event on the active overlay, if found, and record the result and whether the overlay
65  // will block the event from further processing.
66  if(blocking_overlay_ptr) {
67  was_handled = true;
68  was_blocked = blocking_overlay_ptr->blocks_events();
69  result = blocking_overlay_ptr->handle_event(e);
70  }
71 
72  // An MA_RELASE action will reset the active overlay.
73  if(ma == cgv::gui::MA_RELEASE)
74  blocking_overlay_ptr = nullptr;
75 
76  // Store the current active overlay for reference to the next event.
77  last_blocking_overlay_ptr = blocking_overlay_ptr;
78 
79  // Return the result of handle_event from the overlay or true if it was blocked, indicating that the event will not be passed on to the next plugin.
80  // If the event was not handled, pass it to the application plugin itself.
81  if(was_handled) {
82  if(was_blocked)
83  return true;
84  else
85  return result;
86  } else {
87  return handle_event(e);
88  }
89  } else {
90  // TODO: make the overlay have a handles keys flag?
91  // TODO: have a flag that enables blocking the event from further processing when returning true or false?
92  //
93  // Handle all non-mouse type events on all visible overlays in reverse registration order.
94  for(auto it = overlays.rbegin(); it != overlays.rend(); ++it) {
95  overlay_ptr op = (*it);
96  if(op->is_visible())
97  op->handle_event(e);
98  }
99  return handle_event(e);
100  }
