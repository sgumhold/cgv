90  bool surface_renderer::enable(context& ctx)
91  {
92  bool res = group_renderer::enable(ctx);
93  const surface_render_style& srs = get_style<surface_render_style>();
94  if (cull_per_primitive) {
95  if (srs.culling_mode == CM_OFF) {
96  glDisable(GL_CULL_FACE);
97  }
98  else {
99  glCullFace(srs.culling_mode == CM_FRONTFACE ? GL_FRONT : GL_BACK);
100  glEnable(GL_CULL_FACE);
101  }
102  }
103  if (ref_prog().is_linked()) {
104  if (!has_colors)
105  ctx.set_color(srs.surface_color, srs.surface_opacity);
106  ctx.set_material(srs.material);
107  ref_prog().set_uniform(ctx, "map_color_to_material", int(srs.map_color_to_material));
108  ref_prog().set_uniform(ctx, "culling_mode", int(srs.culling_mode));
109  ref_prog().set_uniform(ctx, "illumination_mode", int(srs.illumination_mode));
110  }
111  else
112  res = false;
113  return res;
114  }
