58 {
59 const point_render_style& prs = get_style<point_render_style>();
60 bool res;
61 if (!prs.use_group_color) {

82 {
83 const point_render_style& prs = get_style<point_render_style>();
84
85 bool res;
86 if (!prs.use_group_color && has_indexed_colors) {

106 ref_prog().set_uniform(ctx, "use_color_index", has_indexed_colors);
107 ref_prog().set_uniform(ctx, "measure_point_size_in_pixel", prs.measure_point_size_in_pixel);
108 ref_prog().set_uniform(ctx, "screen_aligned", prs.screen_aligned);
109 ref_prog().set_uniform(ctx, "reference_point_size", reference_point_size);
110 ref_prog().set_uniform(ctx, "use_group_point_size", prs.use_group_point_size);
111 ref_prog().set_uniform(ctx, "viewport_height", (float)ctx.get_height());
112 ref_prog().set_uniform(ctx, "blend_width_in_pixel", prs.blend_width_in_pixel);
113 ref_prog().set_uniform(ctx, "percentual_halo_width", 0.01f*prs.percentual_halo_width);
