82  bool res;

103  bool point_renderer::enable(cgv::render::context& ctx)
104  {
105  const point_render_style& prs = get_style<point_render_style>();
106 
107  bool res;
108  if (!prs.use_group_color && has_indexed_colors) {
109  bool tmp = has_colors;
110  has_colors = true;
111  res = group_renderer::enable(ctx);
112  has_colors = tmp;
113  }
114  else
115  res = group_renderer::enable(ctx);
116 
117  glPointSize(prs.point_size);
118  if (prs.blend_points) {
119  glEnable(GL_BLEND);
120  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
121  }
122  if (ref_prog().is_linked()) {
123  if (!has_point_sizes)
124  ref_prog().set_attribute(ctx, "point_size", prs.point_size);
125  if (!has_depth_offsets)
126  ref_prog().set_attribute(ctx, "depth_offset", prs.default_depth_offset);
127  ref_prog().set_uniform(ctx, "use_color_index", has_indexed_colors);
128  ref_prog().set_uniform(ctx, "measure_point_size_in_pixel", prs.measure_point_size_in_pixel);
129  ref_prog().set_uniform(ctx, "screen_aligned", prs.screen_aligned);
130  ref_prog().set_uniform(ctx, "reference_point_size", reference_point_size);
131  ref_prog().set_uniform(ctx, "use_group_point_size", prs.use_group_point_size);
132  ref_prog().set_uniform(ctx, "viewport_height", (float)ctx.get_height());
133  ref_prog().set_uniform(ctx, "blend_width_in_pixel", prs.blend_width_in_pixel);
134  ref_prog().set_uniform(ctx, "percentual_halo_width", 0.01f*prs.percentual_halo_width);
135  ref_prog().set_uniform(ctx, "halo_width_in_pixel", prs.halo_width_in_pixel);
