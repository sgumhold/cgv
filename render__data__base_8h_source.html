222  virtual bool transfer(context& ctx, renderer& r) {
223  state_out_of_date = false;
224 
225  // The positions array determines the amount of stored vertices.
226  // All other arrays except indices must match its size.
227  if(!positions.empty()) {
228  r.set_position_array(ctx, positions);
229  if(colors.size() == size())
230  r.set_color_array(ctx, colors);
231  else if(colors.empty())
232  r.remove_color_array(ctx);
233 
234  //if(colors.size() == size())
235  // r.set_color_array(ctx, colors);
236  if(!indices.empty())
237  r.set_indices(ctx, indices);
238  else
239  r.remove_indices(ctx);
240  return true;
241  } else {
242  // If no positions are set the render data is assumed to be empty. In this case the
243  // attribute array manager is destructed and reinitialized to clear the GPU buffers.
244  if(attribute_array.is_created()) {
245  attribute_array.destruct(ctx);
246  attribute_array.init(ctx);
247  }
248  // Return false since the attribute array manager holds no data anymore.
249  return false;
250  }
251  }
