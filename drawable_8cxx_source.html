58 bool drawable::get_world_location(int x, int y, const cgv::render::view& V, cgv::math::fvec<double, 3>& world_location, double* window_z_ptr) const
59 {
60  if (!get_context())
61  return false;
62  // analyze the mouse location
63  cgv::render::context& ctx = *get_context();
64  const dmat4* DPV_ptr, *DPV_other_ptr;
65  int x_other, y_other, vp_col_idx, vp_row_idx, vp_width, vp_height;
66  int eye_panel = V.get_modelview_projection_window_matrices(x, y, ctx.get_width(), ctx.get_height(), &DPV_ptr, &DPV_other_ptr, &x_other, &y_other, &vp_col_idx, &vp_row_idx, &vp_width, &vp_height);
67 
68  // get the possibly two (if stereo is enabled) different device z-values
69  double z = ctx.get_window_z(x, y);
70  double z_other = ctx.get_window_z(x_other, y_other);
71  // unproject to world coordinates with smaller (closer to eye) z-value one
72  if (z <= z_other) {
73  if (DPV_ptr->ncols() != 4)
74  return false;
75  // use conversion to (double*) operator to map cgv::math::vec<double> to cgv::math::fvec<float,3>
76  world_location = ctx.get_model_point(x, y, z, *DPV_ptr);
77  if (window_z_ptr)
78  *window_z_ptr = z;
79  }
80  else {
81  if (DPV_other_ptr->ncols() != 4)
82  return false;
83  world_location = ctx.get_model_point(x_other, y_other, z_other, *DPV_other_ptr);
84  if (window_z_ptr)
85  *window_z_ptr = z_other;
86  }
87  return true;
88 }
