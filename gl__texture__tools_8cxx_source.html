124 bool generate_mipmaps(unsigned int dim, bool is_cubemap, bool is_array, std::string* last_error)
125 {
126  if (dim == 0 || dim > 3) {
127  if (last_error)
128  *last_error = "wrong dimension of texture";
129  return false;
130  }
131  if(is_array && (dim < 2 || dim > 3)) {
132  if(last_error)
133  *last_error = "wrong dimension for array texture";
134  return false;
135  }
136  if(is_cubemap && dim != 2) {
137  if(last_error)
138  *last_error = "wrong dimension for cubemap texture";
139  return false;
140  }

142  if (last_error)

144  return false;

146  if(is_array)
147  dim += 2;
148  if(is_cubemap)
149  dim = 6;

511 unsigned find_best_texture_format(const cgv::data::component_format& _cf, cgv::data::component_format* best_cf, const std::vector<data_view>* palettes)
512 {
513  cgv::data::component_format cf = _cf;
514  if (cf.get_nr_components() == 1 && (cf.get_component_name(0) == "L" || cf.get_component_name(0) == "I"))
515  cf.set_component_names("R");
516  cgv::data::component_format best_cf_;
517  if (!best_cf)
518  best_cf = &best_cf_;
519  if (palettes && palettes->size() > 0 && cf.get_nr_components() == 1 && cf.get_component_name(0) == "0")
520  cf = *palettes->at(0).get_format();
521  unsigned int i = find_best_match(cf, std_texture_formats);
522  if (best_cf)
523  *best_cf = cgv::data::component_format(std_texture_formats[i]);
524  unsigned gl_format = gl_std_texture_format_ids[i];
525 
526  if (ensure_glew_initialized() && GLEW_ARB_depth_texture) {
527  i = find_best_match(cf,depth_formats, best_cf);
528  if (i != -1) {
529  if (best_cf)
530  *best_cf = cgv::data::component_format(depth_formats[i]);
531  gl_format = gl_depth_format_ids[i];
532  }
533  }
534  if (true){//ensure_glew_initialized() && GLEW_EXT_texture_snorm) {
535  i = find_best_match(cf, snorm_texture_formats, best_cf);
536  if (i != -1) {
537  if (best_cf)
538  *best_cf = cgv::data::component_format(snorm_texture_formats[i]);
539  gl_format = gl_snorm_texture_format_ids[i];
540  }
541  }
542  if (ensure_glew_initialized() && GLEW_EXT_texture_integer) {
543  i = find_best_match(cf,int_texture_formats, best_cf);
544  if (i != -1) {
545  if (best_cf)
546  *best_cf = cgv::data::component_format(int_texture_formats[i]);
547  gl_format = gl_int_texture_format_ids[i];
548  }
549  }
550  if (ensure_glew_initialized() && GLEW_ARB_texture_float) {
551  i = find_best_match(cf, float_texture_formats, best_cf);
552  if (i != -1) {
553  if (best_cf)
554  *best_cf = cgv::data::component_format(float_texture_formats[i]);
555  gl_format = gl_float_texture_format_ids[i];
556  }
557  }
558  if (ensure_glew_initialized() && GLEW_ARB_texture_rg) {
559  i = find_best_match(cf, rg_texture_formats, best_cf);
560  if (i != -1) {
561  if (best_cf)
562  *best_cf = cgv::data::component_format(rg_texture_formats[i]);
563  gl_format = gl_rg_texture_format_ids[i];
564  }
565  }
566  return gl_format;
567 }

569 // return nr components
570 unsigned configure_src_format(const cgv::data::const_data_view& data, GLuint& src_type, GLuint& src_fmt, const std::vector<data_view>* palettes)
571 {
572  unsigned nr_comp = data.get_format()->get_nr_components();
573  // configure pixel transfer
574  // glPixelStorei(GL_PACK_SWAP_BYTES, swap_bytes ? GL_TRUE : GL_FALSE);
575  // glPixelStorei(GL_PACK_LSB_FIRST, lsb_first ? GL_TRUE : GL_FALSE);
576  // glPixelStorei(GL_PACK_ROW_LENGTH, (GLint) row_length);
577  glPixelStorei(GL_UNPACK_ALIGNMENT, (GLint) data.get_format()->get_alignment(0));
578  src_type = map_to_gl(data.get_format()->get_component_type());
579  ComponentFormat cf = data.get_format()->get_standard_component_format();
580  if (cf != CF_UNDEF)
581  src_fmt = map_to_gl(cf, data.get_format()->get_integer_interpretation());
582  else {
583  if (palettes && palettes->size() > 0 && nr_comp == 1 && data.get_format()->get_component_name(0) == "0") {
584  src_fmt = GL_COLOR_INDEX;
585  const data_view& pv(palettes->at(0));
586  const data_format& pf(*pv.get_format());
587  nr_comp = pf.get_nr_components();
588  unsigned comp_size = pf.get_entry_size() / pf.get_nr_components();
589  std::vector<float> tmp;
590  tmp.resize(pf.get_width());
591  for (unsigned ci=0; ci<nr_comp; ++ci) {
592  glPixelTransferf(GL_RED_BIAS, 0.000001f);
593  for (unsigned i=0; i<pf.get_width(); ++i)
594  tmp[i] = pf.get<unsigned char>(ci, pv.step_i(pv.get_ptr<unsigned char>(), i))/255.0f;
595  switch (pf.get_component_name(ci)[0]) {
596  case 'R' : glPixelMapfv(GL_PIXEL_MAP_I_TO_R, pf.get_width(), &tmp.front()); break;
597  case 'G' : glPixelMapfv(GL_PIXEL_MAP_I_TO_G, pf.get_width(), &tmp.front()); break;
598  case 'B' : glPixelMapfv(GL_PIXEL_MAP_I_TO_B, pf.get_width(), &tmp.front()); break;
599  case 'A' : glPixelMapfv(GL_PIXEL_MAP_I_TO_A, pf.get_width(), &tmp.front()); break;
600  case 'L' :
601  glPixelMapfv(GL_PIXEL_MAP_I_TO_R, pf.get_width(), &tmp.front());
602  glPixelMapfv(GL_PIXEL_MAP_I_TO_G, pf.get_width(), &tmp.front());
603  glPixelMapfv(GL_PIXEL_MAP_I_TO_B, pf.get_width(), &tmp.front());
604  break;
605  }
606  }
607  }
608  }
609  return nr_comp;
610 }

612 bool load_texture(const cgv::data::const_data_view& data, unsigned gl_tex_format, unsigned level, unsigned cube_side, int num_array_layers, const std::vector<data_view>* palettes)
613 {
614  unsigned nr_dim = data.get_format()->get_nr_dimensions();
615  const unsigned char* data_ptr = data.get_ptr<unsigned char>();
616  unsigned w = data.get_format()->get_width();
617  bool cube_map = (nr_dim == 2) && (cube_side != -1);
618  bool texture_array = (nr_dim > 0) && (nr_dim < 4) && !cube_map && num_array_layers != 0;
619 
620  if(!(ensure_glew_initialized() && GLEW_EXT_texture_array))
621  texture_array = false;
622 
623  GLuint src_type, src_fmt;
624  unsigned nr_comp = configure_src_format(data, src_type, src_fmt, palettes);
625 
626  bool gen_mipmap = level == -1;
627  if (gen_mipmap)
628  level = 0;
629  switch(nr_dim) {
630  case 1:
631  if(texture_array) {
632  glTexImage2D(GL_TEXTURE_1D_ARRAY, level, gl_tex_format, w, 1, 0, src_fmt, src_type, data_ptr);
633  } else {
634  glTexImage1D(GL_TEXTURE_1D, level, gl_tex_format, w, 0, src_fmt, src_type, data_ptr);
635  }
636  break;
637  case 2:
638  if(texture_array) {
639  if(num_array_layers < 0) {

905 bool render_to_texture3D(context& ctx, shader_program& prog, TextureSampling texture_sampling, texture& target_tex, texture* target_tex2, texture* target_tex3, texture* target_tex4)
906 {
907  // extract texture resolution
908  unsigned tex_res[3] = { target_tex.get_width(), target_tex.get_height(), target_tex.get_depth() };
909  // check consistency of all texture resolutions
910  if (target_tex2) {
911  if (target_tex2->get_width() != tex_res[0] || target_tex2->get_height() != tex_res[1] || target_tex2->get_depth() != tex_res[2]) {
912  std::cerr << "ERROR in cgv:render::gl::render_to_texture3D: texture resolution of target_tex2 does not match resolution of target_tex" << std::endl;
913  return false;
914  }
915  }
916  if (target_tex3) {
917  if (target_tex3->get_width() != tex_res[0] || target_tex3->get_height() != tex_res[1] || target_tex3->get_depth() != tex_res[2]) {
918  std::cerr << "ERROR in cgv:render::gl::render_to_texture3D: texture resolution of target_tex3 does not match resolution of target_tex" << std::endl;
919  return false;
920  }
921  }
922  if (target_tex4) {
923  if (target_tex4->get_width() != tex_res[0] || target_tex4->get_height() != tex_res[1] || target_tex4->get_depth() != tex_res[2]) {
924  std::cerr << "ERROR in cgv:render::gl::render_to_texture3D: texture resolution of target_tex4 does not match resolution of target_tex" << std::endl;
925  return false;
926  }
927  }
928  // create fbo with resolution of slices
929  cgv::render::frame_buffer fbo;
930  fbo.create(ctx, tex_res[0], tex_res[1]);
931  fbo.attach(ctx, target_tex, 0, 0, 0);
932  if (!fbo.is_complete(ctx)) {
933  std::cerr << "fbo to update volume gradient not complete" << std::endl;
934  return false;
935  }
936  static float V[4 * 3] = {
937  -1, -1, 0, +1, -1, 0,
938  +1, +1, 0, -1, +1, 0
939  };
940  static int F[1 * 4] = {
941  0, 1, 2, 3
942  };
943  float T[4 * 2] = {
944  0, 0, 1, 0,
945  1, 1, 0, 1
946  };
947  if (texture_sampling == TS_VERTEX) {
948  T[0] = T[6] = float(-0.5 / tex_res[0]);
949  T[2] = T[4] = float(1.0 + 0.5 / tex_res[0]);
950  T[1] = T[3] = float(-0.5 / tex_res[1]);
951  T[5] = T[7] = float(1.0 + 0.5 / tex_res[1]);
952  }
953  ctx.push_window_transformation_array();
954  ctx.set_viewport(ivec4(0, 0, tex_res[0], tex_res[1]));
955  int slice_coord_loc = prog.get_uniform_location(ctx, "slice_coord");
956  // go through slices
957  for (int i = 0; i < (int) tex_res[2]; i++) {
958 
959  if (slice_coord_loc != -1) {
960  float slice_coord = (texture_sampling == TS_CELL) ? (i + 0.5f) / tex_res[2] : (float)i / (tex_res[2] - 1);
961  prog.set_uniform(ctx, slice_coord_loc, slice_coord);
962  }
963  // attach textures to fbo
964  fbo.attach(ctx, target_tex, i, 0, 0);
965  if (target_tex2)
966  fbo.attach(ctx, *target_tex2, i, 0, 1);
967  if (target_tex3)
968  fbo.attach(ctx, *target_tex3, i, 0, 2);
969  if (target_tex4)
970  fbo.attach(ctx, *target_tex4, i, 0, 3);
971  fbo.enable(ctx, 0);
972  ctx.draw_faces(V, 0, T, F, 0, F, 1, 4);
973  fbo.disable(ctx);
974  }
975  ctx.pop_window_transformation_array();
976  return true;
977 }
