109T arc_length_even_subdivision(const parametric_curve<CurveT<fvec<T, N>>>& curve, T t = T(1), size_t num_segments = 128) {
110 T step = T(1) / static_cast<T>(num_segments);
111 T result = T(0);
112
113 fvec<T, N> prev_point = curve.evaluate(T(0));
114 for(size_t i = 0; i < num_segments; ++i) {
115 ;
116 T t_ = static_cast<T>(i + 1) * step;
117 if(t_ >= t) {
118 fvec<T, N> next_point = curve.evaluate(t);
119 result += length(next_point - prev_point);
120 break;
121 }
122
123 fvec<T, N> next_point = curve.evaluate(t_);
124 result += length(next_point - prev_point);
125 prev_point = next_point;
126 }
127
128 return result;
