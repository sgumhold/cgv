162void overlay::update_layout() {
163 container_.position = margin_;
164 ivec2 max_size = last_viewport_size_ - 2 * margin_;
165
166 switch(stretch_) {
167 case SO_HORIZONTAL:
168 container_.w() = max_size.x();
169 break;
170 case SO_VERTICAL:
171 container_.h() = max_size.y();
172 break;
173 case SO_BOTH:
174 container_.size = max_size;
175 break;
176 case SO_PERCENTUAL:
177 container_.size = percentual_size_ * max_size;
178 break;
179 default:
180 break;
181 }
182
183 const auto compute_aligned_position = [this, &max_size](AlignmentOption alignment, int axis) {
184 switch(alignment) {
185 case AO_CENTER: return (last_viewport_size_[axis] - container_.size[axis]) / 2;
186 case AO_END: return last_viewport_size_[axis] - container_.size[axis] - margin_[axis];
187 case AO_PERCENTUAL: return static_cast<int32_t>(margin_[axis] + percentual_offset_[axis] * max_size[axis]);
188 case AO_START:
189 case AO_FREE:
190 default: return margin_[axis];
191 }
192 };
193
194 container_.x() = compute_aligned_position(horizontal_alignment_, 0);
195 container_.y() = compute_aligned_position(vertical_alignment_, 1);
196}
