112  bool cone_renderer::enable(context& ctx)
113  {
114  if (!surface_renderer::enable(ctx))
115  return false;
116 
117  if(!ref_prog().is_linked())
118  return false;
119 
120  const cone_render_style& crs = get_style<cone_render_style>();
121  if(!has_radii)
122  ref_prog().set_attribute(ctx, "radius", crs.radius);
123 
124  ref_prog().set_uniform(ctx, "radius_scale", crs.radius_scale);
125 
126  if(crs.enable_texturing && !albedo_texture)
127  return false;
128 
129  if(crs.enable_ambient_occlusion && !density_texture)
130  return false;
131 
132  if(crs.enable_ambient_occlusion) {
133  ref_prog().set_uniform(ctx, "ao_offset", crs.ao_offset);
134  ref_prog().set_uniform(ctx, "ao_distance", crs.ao_distance);
135  ref_prog().set_uniform(ctx, "ao_strength", crs.ao_strength);
136  ref_prog().set_uniform(ctx, "density_tex_offset", crs.tex_offset);
137  ref_prog().set_uniform(ctx, "density_tex_scaling", crs.tex_scaling);
138  ref_prog().set_uniform(ctx, "tex_coord_scaling", crs.tex_coord_scaling);
139  ref_prog().set_uniform(ctx, "texel_size", crs.texel_size);
140  ref_prog().set_uniform(ctx, "cone_angle_factor", crs.cone_angle_factor);
141  ref_prog().set_uniform_array(ctx, "sample_dirs", crs.sample_dirs);
142  }
143 
144  if(albedo_texture) {
145  ref_prog().set_uniform(ctx, "texture_blend_factor", crs.texture_blend_factor);
146  ref_prog().set_uniform(ctx, "texture_offset", crs.texture_offset);
147  ref_prog().set_uniform(ctx, "texture_tiling", crs.texture_tiling);
148  ref_prog().set_uniform(ctx, "texture_reference_length", crs.texture_reference_length);
149  albedo_texture->enable(ctx, 0);
150  }
151  if(density_texture) density_texture->enable(ctx, 1);
152 
153  return true;
154  }
156  bool cone_renderer::disable(context& ctx)
157  {
158  if(albedo_texture) albedo_texture->disable(ctx);
159  if(density_texture) density_texture->disable(ctx);
160 
161  if(!attributes_persist()) {
162  has_radii = false;
163  }
164  return surface_renderer::disable(ctx);
165  }
